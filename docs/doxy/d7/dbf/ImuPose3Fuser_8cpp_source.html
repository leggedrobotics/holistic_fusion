81void ImuPose3Fuser::pose3Callback_(const nav_msgs::Odometry::ConstPtr& odomPtr) {
82 // Static members
83 static int odometryCallbackCounter__ = -1;
84
85 // Counter
86 ++odometryCallbackCounter__;
87
88 Eigen::Isometry3d T_W_Ik;
89 graph_msf::odomMsgToEigen(*odomPtr, T_W_Ik.matrix());
90
91 // Transform to IMU frame
92 double odometryTimeK = odomPtr->header.stamp.toSec();
93
94 // Measurement
95 graph_msf::UnaryMeasurementXDAbsolute<Eigen::Isometry3d, 6> unary6DMeasurement(
96 "Pose3Unary6D", int(pose3OdometryRate_), unaryPose3Frame_, unaryPose3Frame_, graph_msf::RobustNorm::None(), odometryTimeK, 1.0,
97 T_W_Ik, pose3UnaryNoise_, staticTransformsPtr_->getWorldFrame(), staticTransformsPtr_->getWorldFrame());
98
99 // Only add measurement once every second in beginning
100 bool addMeasurementFlag = false;
101 if (odometryCallbackCounter__ < 4000) {
102 if ((odometryCallbackCounter__ % 200) == 0) {
103 addMeasurementFlag = true;
104 }
105 } else { // And more rarely later
106 if ((odometryCallbackCounter__ % 2000) == 0) {
107 addMeasurementFlag = true;
108 }
109 }
110
111 // Add measurement or initialize
112 if (odometryCallbackCounter__ <= 2) {
113 return;
114 } else if (!areYawAndPositionInited()) { // Initializing
115 REGULAR_COUT << GREEN_START << " Odometry callback is setting global yaw, as it was not set so far." << COLOR_END << std::endl;
116 this->initYawAndPosition(unary6DMeasurement);
117 } else if (addMeasurementFlag) { // Already initialized --> unary factor
118 this->addUnaryPose3AbsoluteMeasurement(unary6DMeasurement);
119 }
120
121 // Visualization ----------------------------
122 // Add to path message
123 addToPathMsg(measPose3_worldImuPathPtr_, staticTransformsPtr_->getWorldFrame(), odomPtr->header.stamp, T_W_Ik.matrix().block<3, 1>(0, 3),
124 graphConfigPtr_->imuBufferLength_ * 4);
125
126 // Publish Path
127 pubMeasPose3Path_.publish(measPose3_worldImuPathPtr_);
128}
