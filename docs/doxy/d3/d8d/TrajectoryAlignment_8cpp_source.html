54bool TrajectoryAlignment::associateTrajectories(Trajectory& trajectoryA, Trajectory& trajectoryB, Trajectory& newTrajectoryA,
55 Trajectory& newTrajectoryB) {
56 // matching trajectories with their timestamps
57 bool swapped = false;
58 if (trajectoryA.poses().size() < trajectoryB.poses().size()) {
59 std::swap(trajectoryA, trajectoryB);
60 swapped = true;
61 }
62
63 int indexB = 0;
64 int matchingIndexA = -1;
65 int lastMatchingIndexA = 0;
66 bool matched = false;
67 for (const auto& poseB : trajectoryB.poses()) { // shorter trajectory.
68 int indexA = 0;
69 double diff = std::numeric_limits<double>::max();
70 for (auto it = trajectoryA.poses().begin(); it != trajectoryA.poses().end(); ++it) {
71 if ((abs(poseB.time() - it->time())) < diff && abs((poseB.time() - it->time())) < 0.1) {
72 diff = abs(poseB.time() - it->time());
73 matchingIndexA = indexA;
74 // lastMatchingIndexA = matchingIndexA;
75 matched = true;
76 }
77 ++indexA;
78 }
79
80 if (matched) {
81 newTrajectoryA.addPose(trajectoryA.poses().at(matchingIndexA));
82 newTrajectoryB.addPose(trajectoryB.poses().at(indexB));
83 matched = false;
84 }
85 ++indexB;
86 }
87
88 if (swapped) {
89 std::swap(newTrajectoryA, newTrajectoryB);
90 }
91
92 return true;
93}

95bool TrajectoryAlignment::trajectoryAlignment(Trajectory& trajectoryA, Trajectory& trajectoryB, Eigen::Isometry3d& returnTransform) {
96 // Fill matrices to use Eigen Umeyama function
97 const int numberOfMeasurements = std::min(trajectoryA.poses().size(), trajectoryB.poses().size());
98 if (numberOfMeasurements < 2) return false;
99
100 Eigen::MatrixXd posesA;
101 Eigen::MatrixXd posesB;
102 posesA.resize(3, numberOfMeasurements);
103 posesB.resize(3, numberOfMeasurements);
104
105 for (unsigned i = 0; i < numberOfMeasurements; ++i) {
106 posesA.col(i) = trajectoryA.poses().at(i).position();
107 posesB.col(i) = trajectoryB.poses().at(i).position();
108 }
109
110 // Umeyama Alignment
111 Eigen::Matrix4d transform = umeyama(posesB, posesA, false);
112 returnTransform = Eigen::Isometry3d(transform);
113 std::cout << YELLOW_START << "Trajectory Alignment" << GREEN_START << " Umeyama transform: " << std::endl
114 << COLOR_END << returnTransform.matrix() << std::endl;
115
116 return true;
117}

119bool TrajectoryAlignment::alignTrajectories(double& yaw, Eigen::Isometry3d& returnTransform) {
120 // Containers for running data processing
121 Trajectory copySe3Trajectory;
122 Trajectory copyR3Trajectory;
123 {
124 // Mutex for alignment Flag
125 const std::lock_guard<std::mutex> alignmentLock(alignmentMutex);
126 // Copy Trajectories
127 copySe3Trajectory = se3Trajectory_;
128 copyR3Trajectory = r3Trajectory_;
129 }
130
131 // Check whether we are standing still
132 if (copySe3Trajectory.isStanding(se3Rate_, noMovementTime_, noMovementDistance_) ||
133 copyR3Trajectory.isStanding(r3Rate_, noMovementTime_, noMovementDistance_) || firstAlignmentTryFlag_) {
134 // Mutex for alignment Flag
135 const std::lock_guard<std::mutex> alignmentLock(alignmentMutex);
136 se3Trajectory_.poses().clear();
137 r3Trajectory_.poses().clear();
138 if (r3Trajectory_.poses().size() % 10 == 0) {
139 std::cout << YELLOW_START << "Trajectory Alignment" << GREEN_START << " No movement detected. Clearing trajectories." << COLOR_END
140 << std::endl;
141 }
142 if (firstAlignmentTryFlag_) {
143 std::cout << YELLOW_START << "Trajectory Alignment" << GREEN_START << " First Alignment Try." << COLOR_END << std::endl;
144 firstAlignmentTryFlag_ = false;
145 }
146 return false;
147 }
148
149 // Status
150 if (copyR3Trajectory.poses().size() % 10 == 0) {
151 std::cout << YELLOW_START << "Trajectory Alignment" << GREEN_START << " Current Distance of Se3 / R3 [m]: " << COLOR_END
152 << copySe3Trajectory.distance() << " / " << copyR3Trajectory.distance() << " of required [m] " << minDistanceHeadingInit_
153 << std::endl;
154 }
155
156 // Perform Checks
157 if (copySe3Trajectory.distance() < minDistanceHeadingInit_) {
158 return false;
159 }
160 if (copyR3Trajectory.distance() < minDistanceHeadingInit_) {
161 return false;
162 }
163
164 // Status that all checks passed
165 std::cout << YELLOW_START << "Trajectory Alignment" << GREEN_START << " All checks passed. Trying to align." << COLOR_END << std::endl;
166
167 // Cutting of Timestamps Outside the Joint Time Window
168 double startTime = std::max(copySe3Trajectory.poses().front().time(), copyR3Trajectory.poses().front().time());
169 double endTime = std::min(copySe3Trajectory.poses().back().time(), copyR3Trajectory.poses().back().time());
170 // Print
171 std::cout << YELLOW_START << "Trajectory Alignment" << GREEN_START << " Trajectories A and B before cutting of size: " << COLOR_END
172 << copySe3Trajectory.poses().size() << " / " << copyR3Trajectory.poses().size() << std::endl;
173 // Cut
174 copySe3Trajectory.cutTrajectory(startTime, endTime);
175 copyR3Trajectory.cutTrajectory(startTime, endTime);
176 // Print
177 std::cout << YELLOW_START << "Trajectory Alignment" << GREEN_START
178 << " Trajectories A and B after cutting of size and time: " << COLOR_END << copySe3Trajectory.poses().size() << " / "
179 << copyR3Trajectory.poses().size() << std::endl;
180 // Downsample the longer trajectory to the shorter one
181 int downsampleFactor = 1;
182 if (copySe3Trajectory.poses().size() > copyR3Trajectory.poses().size()) {
183 downsampleFactor = std::floor(copySe3Trajectory.poses().size() / copyR3Trajectory.poses().size());
184 copySe3Trajectory.downsample(downsampleFactor);
185 } else {
186 downsampleFactor = std::floor(copyR3Trajectory.poses().size() / copySe3Trajectory.poses().size());
187 copyR3Trajectory.downsample(downsampleFactor);
188 }
189 std::cout << YELLOW_START << "Trajectory Alignment" << GREEN_START << " Downsample Factor: " << COLOR_END << downsampleFactor
190 << std::endl;
191 // Print
192 std::cout << YELLOW_START << "Trajectory Alignment" << GREEN_START << " Trajectories A and B after downsampling: " << COLOR_END
193 << copySe3Trajectory.poses().size() << " / " << copyR3Trajectory.poses().size() << std::endl;
194
195 // Align Trajectories
196 Trajectory newSe3Trajectory;
197 Trajectory newR3Trajectory;
198 // Associate Trajectories
199 // Difference of trajectory size should not be larger than 10%
200 bool sizeDeviatesTooMuch = (std::abs(static_cast<int>(copySe3Trajectory.poses().size() - copyR3Trajectory.poses().size())) >
201 0.1 * std::min(copySe3Trajectory.poses().size(), copyR3Trajectory.poses().size()));
202 // If the trajectories deviate too much, associate them
203 if (sizeDeviatesTooMuch) {
204 associateTrajectories(copySe3Trajectory, copyR3Trajectory, newSe3Trajectory, newR3Trajectory);
205 std::cout << YELLOW_START << "Trajectory Alignment" << GREEN_START
206 << " Trajectories Associated, as after downsampling they were different in size." << COLOR_END << std::endl;
207 }
208 // If the trajectories do not deviate too much, just copy them
209 else {
210 newSe3Trajectory = copySe3Trajectory;
211 newR3Trajectory = copyR3Trajectory;
212 std::cout << YELLOW_START << "Trajectory Alignment" << GREEN_START
213 << " Trajectories copied, as after downsampling they were similar in size." << COLOR_END << std::endl;
214 }
215
216 // Update Trajectories
217 copySe3Trajectory = newSe3Trajectory;
218 copyR3Trajectory = newR3Trajectory;
219 Eigen::Isometry3d alignmentTransform;
220 // Trajectory Alignment
221 if (!trajectoryAlignment(newR3Trajectory, newSe3Trajectory, alignmentTransform)) {
222 std::cout << "TrajectoryAlignment::initializeYaw trajectoryAlignment failed." << std::endl;
223 return false;
224 } else {
225 std::cout << YELLOW_START << "Trajectory Alignment" << GREEN_START << " Trajectories Aligned." << COLOR_END << std::endl;
226 }
227 // Math --> Rotation to Yaw
228 Eigen::Matrix3d rotation = alignmentTransform.rotation();
229 double pitch = -asin(rotation(2, 0));
230 double roll = atan2(rotation(2, 1), rotation(2, 2));
231 yaw = atan2(rotation(1, 0) / cos(pitch), rotation(0, 0) / cos(pitch));
232 std::cout << YELLOW_START << "Trajectory Alignment" << GREEN_START << " Initial Roll/Pitch/Yaw(deg):" << COLOR_END << roll * 180 / M_PI
233 << "," << pitch * 180 / M_PI << "," << yaw * 180 / M_PI << std::endl;
234 // Return
235 returnTransform = alignmentTransform;
236 return true;
237}
