30void GraphMsfClassic::addUnaryYawAbsoluteMeasurement(const UnaryMeasurementXDAbsolute<double, 1>& yaw_W_frame) {
31 // Only take actions if graph has been initialized
32 if (!initedGraphFlag_) {
33 return;
34 }
35
36 // Check for covariance violation
37 bool covarianceViolatedFlag =
38 isCovarianceViolated_<1>(yaw_W_frame.unaryMeasurementNoiseDensity(), yaw_W_frame.covarianceViolationThreshold());
39
40 // Transform yaw to imu frame
41 gtsam::Rot3 yawR_W_frame = gtsam::Rot3::Yaw(yaw_W_frame.unaryMeasurement());
42 gtsam::Rot3 yawR_W_I =
43 yawR_W_frame *
44 gtsam::Rot3(staticTransformsPtr_->rv_T_frame1_frame2(yaw_W_frame.sensorFrameName(), staticTransformsPtr_->getImuFrame()).rotation());
45
46 // Add factor
47 if (!covarianceViolatedFlag) {
48 graphMgrPtr_->addUnaryFactorInImuFrame<double, 1, YawFactor, gtsam::symbol_shorthand::X>(
49 yawR_W_I.yaw(), yaw_W_frame.unaryMeasurementNoiseDensity(), yaw_W_frame.timeK());
50 {
51 // Mutex for optimizeGraph Flag
52 const std::lock_guard<std::mutex> optimizeGraphLock(optimizeGraphMutex_);
53 optimizeGraphFlag_ = true;
54 }
55 } else {
56 std::cout << YELLOW_START << "GMsf-Classic" << RED_START
57 << " Covariance violation detected in yaw unary measurement at time " << yaw_W_frame.timeK()
58 << ". Measurement not added to graph." << COLOR_END << std::endl;
59 }
60}

128void GraphMsfClassic::addBinaryPose3Measurement(const BinaryMeasurementXD<Eigen::Isometry3d, 6>& deltaMeasurement) {
129 // Valid measurement received
130 if (!validFirstMeasurementReceivedFlag_) {
131 validFirstMeasurementReceivedFlag_ = true;
132 }
133
134 // Only take actions if graph has been initialized
135 if (!initedGraphFlag_) {
136 return;
137 }
138
139 Eigen::Isometry3d T_fkm1_fk = deltaMeasurement.deltaMeasurement();
140
141 // Check frame of measuremnts
142 if (deltaMeasurement.measurementName() != staticTransformsPtr_->getImuFrame()) {
143 T_fkm1_fk = staticTransformsPtr_->rv_T_frame1_frame2(staticTransformsPtr_->getImuFrame(), deltaMeasurement.sensorFrameName()) *
144 T_fkm1_fk *
145 staticTransformsPtr_->rv_T_frame1_frame2(deltaMeasurement.sensorFrameName(), staticTransformsPtr_->getImuFrame());
146 }
147
148 static_cast<void>(graphMgrPtr_->addPoseBetweenFactor(
149 gtsam::Pose3(T_fkm1_fk.matrix()), deltaMeasurement.measurementNoiseDensity(), deltaMeasurement.timeKm1(), deltaMeasurement.timeK(),
150 deltaMeasurement.measurementRate(), deltaMeasurement.robustNormEnum(), deltaMeasurement.robustNormConstant()));
151
152 // Optimize
153 {
154 // Mutex for optimizeGraph Flag
155 const std::lock_guard<std::mutex> optimizeGraphLock(optimizeGraphMutex_);
156 optimizeGraphFlag_ = true;
157 }
158}
