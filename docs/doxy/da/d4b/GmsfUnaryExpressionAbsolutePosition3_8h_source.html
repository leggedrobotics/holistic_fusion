48 // i) Generate Expression for Basic IMU State in World Frame at Key -------------------------------------
49 void generateImuStateInWorldFrameAtKey(const gtsam::Key& closestGeneralKey) final {
50 // Translation (core part)
51 exp_fixedFrame_t_fixedFrame_sensorFrame_ =
52 gtsam::translation(gtsam::Expression<gtsam::Pose3>(gtsam::symbol_shorthand::X(closestGeneralKey))); // W_t_W_I at this point
53
54 // Rotation (needed for the transformation to the sensor frame)
55 exp_R_fixedFrame_I_ =
56 gtsam::rotation(gtsam::Expression<gtsam::Pose3>(gtsam::symbol_shorthand::X(closestGeneralKey))); // R_W_I at this point
57 }
58
59 // Interface with three cases (non-exclusive):
60 // ii) Holistically Optimize over Fixed Frames -----------------------------------------------------------
61 gtsam::Pose3 computeT_W_fixedFrame_initial(const gtsam::NavState& W_currentPropagatedState) final {
62 gtsam::Pose3 T_fixedFrame_sensorFrame_meas_noOrientation =
63 gtsam::Pose3(gtsam::Rot3::Identity(), positionUnaryMeasurementPtr_->unaryMeasurement());
64 gtsam::Pose3 T_W_sensorFrame_est = W_currentPropagatedState.pose() * gtsam::Pose3(T_I_sensorFrameInit_.matrix());
65 return T_W_sensorFrame_est * T_fixedFrame_sensorFrame_meas_noOrientation.inverse(); // TODO: Does not yet include orientation
66 }
67
68 const Eigen::Vector3d getMeasurementPosition() final {

77
78 void transformStateToReferenceFrameMeasurement(const gtsam::Pose3_& exp_T_W_fixedFrame) override {
79 // Transform state to fixed frame
80 exp_fixedFrame_t_fixedFrame_sensorFrame_ =
81 gtsam::transformTo(exp_T_W_fixedFrame, exp_fixedFrame_t_fixedFrame_sensorFrame_); // T_fixedFrame_I at this point
82
83 // Transform rotation from world to fixed frame
84 exp_R_fixedFrame_I_ = inverseRot3(gtsam::rotation(exp_T_W_fixedFrame)) * exp_R_fixedFrame_I_; // R_fixedFrame_I at this point
85 }
86
87 // iii) Transform Measurement to Core Imu Frame -----------------------------------------------------------
88 void transformImuStateToSensorFrameState() final {
89 // Get relative translation
90 Eigen::Vector3d I_t_I_sensorFrame = T_I_sensorFrameInit_.translation();
91
92 // Rotation of IMU frame
93 exp_fixedFrame_t_fixedFrame_sensorFrame_ =
94 exp_fixedFrame_t_fixedFrame_sensorFrame_ +
95 gtsam::rotate(exp_R_fixedFrame_I_, I_t_I_sensorFrame); // fixedFrame_t_fixedFrame_sensorFrameInit
96 }
97
98 // iv) Extrinsic Calibration ---------------------------------------------------------------------
99 void applyExtrinsicCalibrationCorrection(const gtsam::Point3_& exp_sensorFrame_t_sensorFrame_correctSensorFrame) final {
100 // Apply Correction
101 gtsam::Rot3_ exp_R_fixedFrame_sensorFrame = exp_R_fixedFrame_I_ * gtsam::Rot3_(gtsam::Rot3(T_I_sensorFrameInit_.rotation()));
102 exp_fixedFrame_t_fixedFrame_sensorFrame_ =
103 exp_fixedFrame_t_fixedFrame_sensorFrame_ +
104 gtsam::rotate(exp_R_fixedFrame_sensorFrame,
105 exp_sensorFrame_t_sensorFrame_correctSensorFrame); // fixedFrame_t_fixedFrame_sensorFrameCorrected
106 }
107
108 gtsam::Pose3 convertToPose3(const gtsam::Point3& measurement) final { return gtsam::Pose3(gtsam::Rot3::Identity(), measurement); }
109
110 gtsam::Point3 convertFromPose3(const gtsam::Pose3& measurement) final { return measurement.translation(); }
111
112 // Return Expression
113 [[nodiscard]] const gtsam::Expression<gtsam::Point3> getGtsamExpression() const override {
114 return exp_fixedFrame_t_fixedFrame_sensorFrame_;
115 }
116
117 private:
118 // Full Measurement Type
119 std::shared_ptr<UnaryMeasurementXDAbsolute<Eigen::Vector3d, 3>> positionUnaryMeasurementPtr_;
120
121 // Expression
122 gtsam::Expression<gtsam::Point3> exp_fixedFrame_t_fixedFrame_sensorFrame_; // Translation
123 gtsam::Expression<gtsam::Rot3> exp_R_fixedFrame_I_; // Rotation
124};
