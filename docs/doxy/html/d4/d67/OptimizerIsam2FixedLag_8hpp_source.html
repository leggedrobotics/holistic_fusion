34  isam2Params_); // std::make_shared<gtsam::ISAM2>(isamParams_);

44  bool update(const gtsam::NonlinearFactorGraph& newGraphFactors, const gtsam::Values& newGraphValues,
45  const std::map<gtsam::Key, double>& newGraphKeysTimeStampMap) override {
46  // Make copy of the fixedLagSmootherPtr_ to avoid changing the original graph
47  gtsam::IncrementalFixedLagSmoother fixedLagSmootherCopy = *fixedLagSmootherPtr_;
48 
49  // Try to update
50  try {
51  fixedLagSmootherPtr_->update(newGraphFactors, newGraphValues, newGraphKeysTimeStampMap);
52  }
53  // Case 1: Catching the failure of marginalization (as some states which should be marginalized have not been optimized before)
54  catch (const std::out_of_range& outOfRangeException) {
55  std::cerr << YELLOW_START << "GMsf-ISAM2" << RED_START << " Out of Range exception while optimizing graph: '"
56  << outOfRangeException.what() << "'." << COLOR_END << std::endl;
57  std::cout << YELLOW_START << "GMsf-ISAM2" << RED_START
58  << " This happens if the FixedLagSmoother tries to marginalize out states that have never been optimized before. This e.g. "
59  "happens if the graph hasn't been optimized for the duration of the smoother lag. Increase the lag or optimization "
60  "frequency in this case."
61  << COLOR_END << std::endl;
62 
63  // Detect all factors that contain a timestamp older than the smoother lag
64  double latestTimeStamp = 0.0;
65  for (auto factor : newGraphFactors) {
66  for (auto key : factor->keys()) {
67  // Check whether key is in the keyTimestampMap
68  if (newGraphKeysTimeStampMap.find(key) != newGraphKeysTimeStampMap.end()) {
69  if (newGraphKeysTimeStampMap.at(key) > latestTimeStamp) {
70  latestTimeStamp = newGraphKeysTimeStampMap.at(key);
71  }
72  }
73  }
74  }
75  std::cout << YELLOW_START << "GMsf-ISAM2" << RED_START << " Latest timestamp in new factors: " << std::setprecision(14)
76  << latestTimeStamp << COLOR_END << std::endl;
77  // Filter out the factor that caused the error -------------------------
78  bool filteredOutAtLeastOneKey = false;
79  // Containers
80  gtsam::NonlinearFactorGraph newGraphFactorsFiltered = gtsam::NonlinearFactorGraph();
81  // Filtering the keyTimestampMap
82  std::map<gtsam::Key, double> newGraphKeysTimeStampMapFiltered = newGraphKeysTimeStampMap;
83  // For loop
84  for (auto factor : newGraphFactors) {
85  bool factorOnlyContainsExistentKeys = true;
86  for (auto key : factor->keys()) {
87  // If Exists
88  if (newGraphKeysTimeStampMap.find(key) != newGraphKeysTimeStampMap.end()) {
89  double timestampAtKey = newGraphKeysTimeStampMap.at(key);
90  // Case 1: Check if key is older than smoother lag
91  if (latestTimeStamp - timestampAtKey > graphConfigPtr_->realTimeSmootherLag_) {
92  std::cout << YELLOW_START << "GMsf-ISAM2" << RED_START
93  << " Factor contains key older than smoother lag: " << gtsam::Symbol(key) << ", which is "
94  << latestTimeStamp - timestampAtKey << "s old." << COLOR_END << std::endl;
95  factorOnlyContainsExistentKeys = false;
96  filteredOutAtLeastOneKey = true;
97  }
98  // Case 2: Check if key is from the future --> hence might not be optimized
99  else if (timestampAtKey > latestTimeStamp) {
100  std::cout << YELLOW_START << "GMsf-ISAM2" << RED_START << " Factor contains key from the future: " << gtsam::Symbol(key)
101  << ", which is " << timestampAtKey << "s in the future." << COLOR_END << std::endl;
102  factorOnlyContainsExistentKeys = false;
103  filteredOutAtLeastOneKey = true;
104  }
105  }
106  }
107  // Add factor if it does not contain any key older than the smoother lag
108  if (factorOnlyContainsExistentKeys) {
109  newGraphFactorsFiltered.add(factor);
110  }
111  }
112 
113  // Limit depth of recursion
114  // Try again
115  if (filteredOutAtLeastOneKey) {
116  std::cout << YELLOW_START << "GMsf-ISAM2" << GREEN_START
117  << " Filtered out factors that are either older than the smoother lag or coming from the future. Trying to optimize "
118  "again."
119  << COLOR_END << std::endl;
120  return update(newGraphFactorsFiltered, newGraphValues, newGraphKeysTimeStampMapFiltered);
121  }
122  // Nothing changed, hence abort
123  else {
124  // Show all values and corresponding timestamps that are not in timestamp map
125  std::cout << YELLOW_START << "GMsf-ISAM2" << RED_START
126  << " Could not filter out any factors that are either older than the smoother lag or coming from the future. Aborting "
127  "optimization."
128  << COLOR_END << std::endl;
129  return false;
130  }
131  }
132  // Case 2: Catching the addition of a bad factor (pointing to a state that does not exist)
133  catch (const gtsam::ValuesKeyDoesNotExist& valuesKeyDoesNotExistException) {
134  std::cout << "----------------------------------------------------------" << std::endl;
135  std::cerr << YELLOW_START << "GMsf-ISAM2" << RED_START << " ValuesKeyDoesNotExist exception while optimizing graph: '" << COLOR_END
136  << valuesKeyDoesNotExistException.what() << "'" << std::endl;
137  std::cout << YELLOW_START << "GMsf-ISAM2" << COLOR_END
138  << " This happens if a factor is added to the graph that contains a non-existent state. The most common case is a factor "
139  "containing a graph state that was marginalized out before. To avoid this, increase the factor graph lag or reduce "
140  "the delay of your measurements."
141  << std::endl;
142 
143  // Filter out the factor that caused the error -------------------------
144  gtsam::NonlinearFactorGraph newGraphFactorsFiltered;
145  gtsam::Key valuesKeyNotExistent = valuesKeyDoesNotExistException.key();
146  bool removedAtLeastOneFactorOrKey = false;
147  for (auto factor : newGraphFactors) {
148  bool factorContainsExistentKeys = true;
149  for (auto key : factor->keys()) {
150  if (key == valuesKeyNotExistent) {
151  std::cout << YELLOW_START << "GMsf-ISAM2" << RED_START << " Factor contains non-existent key: " << gtsam::Symbol(key)
152  << ". -> Removed." << COLOR_END << std::endl;
153  factorContainsExistentKeys = false;
154  removedAtLeastOneFactorOrKey = true;
155  break; // factor contains dangerous key --> break inner loop
156  }
157  }
158  if (factorContainsExistentKeys) {
159  newGraphFactorsFiltered.add(factor);
160  }
161  }
162 
163  // Filter out the key from the keyTimestampMap -------------------------
164  std::map<gtsam::Key, double> newGraphKeysTimeStampMapFiltered;
165  for (auto keyTimeStamp : newGraphKeysTimeStampMap) {
166  if (keyTimeStamp.first != valuesKeyNotExistent) {
167  newGraphKeysTimeStampMapFiltered.insert(keyTimeStamp);
168  } else {
169  std::cout << YELLOW_START << "GMsf-ISAM2" << RED_START
170  << " GraphKeysTimeStampMap contains non-existent key: " << gtsam::Symbol(keyTimeStamp.first) << ". -> Removed."
171  << COLOR_END << std::endl;
172  removedAtLeastOneFactorOrKey = true;
173  }
174  }
175 
176  // Keys in values
177  for (auto key : newGraphValues.keys()) {
178  if (key == valuesKeyNotExistent) {
179  std::cout << YELLOW_START << "GMsf-ISAM2" << RED_START << " Values contains non-existent key: " << gtsam::Symbol(key)
180  << ". -> Removed." << COLOR_END << std::endl;
181  removedAtLeastOneFactorOrKey = true;
182  }
183  }
184 
185  // Potentially try to optimize again
186  if (removedAtLeastOneFactorOrKey) {
187  std::cout << YELLOW_START << "GMsf-ISAM2" << GREEN_START
188  << " Filtered out the factor or key that caused the error (i.e. containing the key "
189  << gtsam::Symbol(valuesKeyNotExistent) << "). Trying to optimize again." << COLOR_END << std::endl;
190  std::cout << "----------------------------------------------------------" << std::endl;
191  // Copy back
192  *fixedLagSmootherPtr_ = fixedLagSmootherCopy;
193  // Try again
194  return update(newGraphFactorsFiltered, newGraphValues, newGraphKeysTimeStampMapFiltered);
195  } else {
196  std::cout << YELLOW_START << "GMsf-ISAM2" << RED_START
197  << " Could not filter out any factors that caused the error (i.e. containing the key "
198  << gtsam::Symbol(valuesKeyNotExistent) << "). Aborting optimization." << COLOR_END << std::endl;
199  std::cout << "----------------------------------------------------------" << std::endl;
200  return false;
201  }
202  }
203  // Case 3: ValuesKeyAlreadyExists
204  catch (const gtsam::ValuesKeyAlreadyExists& valuesKeyAlreadyExistsException) {
205  std::cerr << YELLOW_START << "GMsf-ISAM2" << RED_START << " ValuesKeyAlreadyExists exception while optimizing graph: '"
206  << valuesKeyAlreadyExistsException.what() << "'" << COLOR_END << std::endl;
207  std::cout << YELLOW_START << "GMsf-ISAM2" << COLOR_END
208  << " This happens if a factor is added to the graph that contains a state that already exists. This is usually a sign of "
209  "a bug in the code. Please check the factor graph construction."
210  << std::endl;
211  throw std::runtime_error(valuesKeyAlreadyExistsException.what());
212  }
213  // Case 4: Runtime error --> can't handle explicitly
214  catch (const std::runtime_error& runtimeError) {
215  std::cout << YELLOW_START << "GMsf-ISAM2" << RED_START << " Runtime error while optimizing graph: " << runtimeError.what()
216  << COLOR_END << std::endl;
217  throw std::runtime_error(runtimeError.what());
218  }
219  if (!optimizedAtLeastOnceFlag_) {
220  optimizedAtLeastOnceFlag_ = true;
221  }
222  return true;
223  }
