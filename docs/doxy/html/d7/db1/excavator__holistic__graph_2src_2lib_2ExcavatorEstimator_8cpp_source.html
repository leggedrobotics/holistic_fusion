97 void ExcavatorEstimator::lidarOdometryCallback_(const nav_msgs::Odometry::ConstPtr& odomLidarPtr) {
98  // Static members
99  static int lidarOdometryCallbackCounter__ = -1;
100  static Eigen::Isometry3d lio_T_M_LKm1__;
101  static double lidarOdometryTimeKm1__ = 0;
102 
103  // Counter
104  ++lidarOdometryCallbackCounter__;
105 
106  Eigen::Isometry3d lio_T_M_Lk;
107  graph_msf::odomMsgToEigen(*odomLidarPtr, lio_T_M_Lk.matrix());
108 
109  // Transform to IMU frame
110  double lidarOdometryTimeK = odomLidarPtr->header.stamp.toSec();
111 
112  // Measurement
113  // Frame Names
114  const std::string& sensorFrameName = dynamic_cast<ExcavatorStaticTransforms*>(staticTransformsPtr_.get())->getLioOdometryFrame();
115  const std::string& fixedFrameName = odomLidarPtr->header.frame_id;
116  // Measurement
117  graph_msf::UnaryMeasurementXDAbsolute<Eigen::Isometry3d, 6> unary6DMeasurement(
118  "LioUnary6D", int(lioOdometryRate_), sensorFrameName, sensorFrameName + sensorFrameCorrectedNameId, graph_msf::RobustNorm::None(),
119  lidarOdometryTimeK, 1.0, lio_T_M_Lk, lioPoseUnaryNoise_, fixedFrameName, staticTransformsPtr_->getWorldFrame(),
120  initialSe3AlignmentNoiseDensity_, lioSe3AlignmentRandomWalk_);
121 
122  if (lidarOdometryCallbackCounter__ <= 2) {
123  return;
124  } else if (areYawAndPositionInited()) { // Already initialized --> unary factor
125  this->addUnaryPose3AbsoluteMeasurement(unary6DMeasurement);
126  } else if (!(useLeftGnssFlag_ || useRightGnssFlag_) || secondsSinceStart() > 15) { // Initializing
127  REGULAR_COUT << GREEN_START << " LiDAR odometry callback is setting global yaw, as it was not set so far." << COLOR_END << std::endl;
128  this->initYawAndPosition(unary6DMeasurement);
129  }
130 
131  // Wrap up iteration
132  lio_T_M_LKm1__ = lio_T_M_Lk;
133  lidarOdometryTimeKm1__ = lidarOdometryTimeK;
134 
135  // Visualization ----------------------------
136  // Add to path message
137  addToPathMsg(measLio_mapImuPathPtr_, odomLidarPtr->header.frame_id, odomLidarPtr->header.stamp, (lio_T_M_Lk.matrix()).block<3, 1>(0, 3),
138  graphConfigPtr_->imuBufferLength_ * 4);
139 
140  // Publish Path
141  pubMeasMapLioPath_.publish(measLio_mapImuPathPtr_);
142 }

144 void ExcavatorEstimator::gnssCallback_(const sensor_msgs::NavSatFix::ConstPtr& leftGnssMsgPtr,
145  const sensor_msgs::NavSatFix::ConstPtr& rightGnssMsgPtr) {
146  // Static method variables
147  static Eigen::Vector3d accumulatedLeftCoordinates__(0.0, 0.0, 0.0);
148  static Eigen::Vector3d accumulatedRightCoordinates__(0.0, 0.0, 0.0);
149  // static Eigen::Vector3d W_t_W_GnssL_km1__, W_t_W_GnssR_km1__;
150  static int gnssCallbackCounter__ = 0;
151 
152  // Start
153  ++gnssCallbackCounter__;
154  Eigen::Vector3d leftGnssCoord = Eigen::Vector3d(leftGnssMsgPtr->latitude, leftGnssMsgPtr->longitude, leftGnssMsgPtr->altitude);
155  Eigen::Vector3d rightGnssCoord = Eigen::Vector3d(rightGnssMsgPtr->latitude, rightGnssMsgPtr->longitude, rightGnssMsgPtr->altitude);
156  Eigen::Vector3d leftGnssCovarianceXYZ(leftGnssMsgPtr->position_covariance[0], leftGnssMsgPtr->position_covariance[4],
157  leftGnssMsgPtr->position_covariance[8]);
158  Eigen::Vector3d rightGnssCovarianceXYZ(rightGnssMsgPtr->position_covariance[0], rightGnssMsgPtr->position_covariance[4],
159  rightGnssMsgPtr->position_covariance[8]);
160 
161  // Workflow
162  if (gnssCallbackCounter__ <= NUM_GNSS_CALLBACKS_UNTIL_START) { // Accumulate measurements at the beginning
163  // Wait until measurements got accumulated
164  accumulatedLeftCoordinates__ += leftGnssCoord;
165  accumulatedRightCoordinates__ += rightGnssCoord;
166  if (!(gnssCallbackCounter__ % 10)) {
167  REGULAR_COUT << " NOT ENOUGH Gnss MESSAGES ARRIVED!" << std::endl;
168  }
169  return;
170  } else if (gnssCallbackCounter__ ==
171  NUM_GNSS_CALLBACKS_UNTIL_START + 1) { // Initialize the GNSS handler with the initial position and initialize the graph
172  Eigen::Vector3d meanLeftCoordinates = accumulatedLeftCoordinates__ / NUM_GNSS_CALLBACKS_UNTIL_START;
173  Eigen::Vector3d meanRightCoordinates = accumulatedRightCoordinates__ / NUM_GNSS_CALLBACKS_UNTIL_START;
174  gnssHandlerPtr_->initHandler(meanLeftCoordinates, meanRightCoordinates);
175  }
176 
177  // Convert to cartesian coordinates --> Needed for every ramaining step
178  Eigen::Vector3d W_t_W_GnssL, W_t_W_GnssR;
179  gnssHandlerPtr_->convertNavSatToPositions(leftGnssCoord, rightGnssCoord, W_t_W_GnssL, W_t_W_GnssR);
180  double yaw_W_C = gnssHandlerPtr_->computeYaw(W_t_W_GnssL, W_t_W_GnssR);
181 
182  std::string fixedFrameName = staticTransformsPtr_->getWorldFrame(); // + "_ENU";
183 
184  // State Machine
185  if (!areYawAndPositionInited() && areRollAndPitchInited()) { // Try to initialize yaw and position if not done already
186  if (this->initYawAndPositionInWorld(yaw_W_C, W_t_W_GnssL,
187  dynamic_cast<ExcavatorStaticTransforms*>(staticTransformsPtr_.get())->getCabinFrame(),
188  dynamic_cast<ExcavatorStaticTransforms*>(staticTransformsPtr_.get())->getLeftGnssFrame())) {
189  REGULAR_COUT << " Set yaw and position successfully." << std::endl;
190  }
191  } else { // Already initialized --> add to graph
192  // Measurement type 1: GNSS Yaw ----------------------------------------------------
193  if (useGnssYawFlag_) {
194  // Sensor Frame Name
195  const std::string& sensorFrameName = dynamic_cast<ExcavatorStaticTransforms*>(staticTransformsPtr_.get())->getCabinFrame();
196  // Create yaw measurement and add it
197  Eigen::Matrix<double, 1, 1> gnssHeadingUnaryNoise(std::max(5.0 * leftGnssCovarianceXYZ.maxCoeff(), gnssHeadingUnaryNoise_));
198  graph_msf::UnaryMeasurementXDAbsolute<double, 1> meas_yaw_W_C(
199  "GnssYaw", int(gnssRate_), sensorFrameName, sensorFrameName + sensorFrameCorrectedNameId, graph_msf::RobustNorm::None(),
200  leftGnssMsgPtr->header.stamp.toSec(), 1.0, yaw_W_C, gnssHeadingUnaryNoise, fixedFrameName, staticTransformsPtr_->getWorldFrame());
201  this->addUnaryYawAbsoluteMeasurement(meas_yaw_W_C);
202  }
203  // Measurement type 2: GNSS Left Position ----------------------------------------------------
204  if (useLeftGnssFlag_) {
205  // Sensor Frame Name
206  const std::string& sensorFrameName = dynamic_cast<ExcavatorStaticTransforms*>(staticTransformsPtr_.get())->getLeftGnssFrame();
207  // Create left position measurement and add it
208  leftGnssCovarianceXYZ = Eigen::Vector3d(std::max(leftGnssCovarianceXYZ(0), gnssPositionUnaryNoise_),
209  std::max(leftGnssCovarianceXYZ(1), gnssPositionUnaryNoise_),
210  std::max(leftGnssCovarianceXYZ(2), gnssPositionUnaryNoise_));
211  graph_msf::UnaryMeasurementXDAbsolute<Eigen::Vector3d, 3> meas_W_t_W_GnssL(
212  "GnssLeftPosition", int(gnssRate_), sensorFrameName, sensorFrameName + sensorFrameCorrectedNameId, graph_msf::RobustNorm::None(),
213  leftGnssMsgPtr->header.stamp.toSec(), 1.0, W_t_W_GnssL, leftGnssCovarianceXYZ, fixedFrameName,
214  staticTransformsPtr_->getWorldFrame());
215  this->addUnaryPosition3AbsoluteMeasurement(meas_W_t_W_GnssL);
216  }
217  // Measurement type 3: GNSS Right Position ----------------------------------------------------
218  if (useRightGnssFlag_) {
219  // Sensor Frame Name
220  const std::string& sensorFrameName = dynamic_cast<ExcavatorStaticTransforms*>(staticTransformsPtr_.get())->getRightGnssFrame();
221  // Create right position measurement and add it
222  rightGnssCovarianceXYZ = Eigen::Vector3d(std::max(rightGnssCovarianceXYZ(0), gnssPositionUnaryNoise_),
223  std::max(rightGnssCovarianceXYZ(1), gnssPositionUnaryNoise_),
224  std::max(rightGnssCovarianceXYZ(2), gnssPositionUnaryNoise_));
225  graph_msf::UnaryMeasurementXDAbsolute<Eigen::Vector3d, 3> meas_W_t_W_GnssR(
226  "GnssRightPosition", int(gnssRate_), sensorFrameName, sensorFrameName + sensorFrameCorrectedNameId, graph_msf::RobustNorm::None(),
227  rightGnssMsgPtr->header.stamp.toSec(), 1.0, W_t_W_GnssR, rightGnssCovarianceXYZ, fixedFrameName,
228  staticTransformsPtr_->getWorldFrame());
229  this->addUnaryPosition3AbsoluteMeasurement(meas_W_t_W_GnssR);
230  }
231  // Visualizations ------------------------------------------------------------
232  // Left GNSS
233  if (useLeftGnssFlag_) {
234  addToPathMsg(measGnss_worldGnssLPathPtr_, staticTransformsPtr_->getWorldFrame(), leftGnssMsgPtr->header.stamp, W_t_W_GnssL,
235  graphConfigPtr_->imuBufferLength_ * 4);
236  pubMeasWorldGnssLPath_.publish(measGnss_worldGnssLPathPtr_);
237  }
238  // Right GNSS
239  if (useRightGnssFlag_) {
240  addToPathMsg(measGnss_worldGnssRPathPtr_, staticTransformsPtr_->getWorldFrame(), rightGnssMsgPtr->header.stamp, W_t_W_GnssR,
241  graphConfigPtr_->imuBufferLength_ * 4);
242  pubMeasWorldGnssRPath_.publish(measGnss_worldGnssRPathPtr_);
243  }
244  }

247 void ExcavatorEstimator::publishState(
248  const std::shared_ptr<graph_msf::SafeIntegratedNavState>& preIntegratedNavStatePtr,
249  const std::shared_ptr<graph_msf::SafeNavStateWithCovarianceAndBias>& optimizedStateWithCovarianceAndBiasPtr) {
250  // Lookup I->B, also influenced by rotation of cabin
251  static tf::StampedTransform transform_I_B;
252  tfListener_.waitForTransform(staticTransformsPtr_->getImuFrame(),
253  dynamic_cast<ExcavatorStaticTransforms*>(staticTransformsPtr_.get())->getBaseLinkFrame(), ros::Time(0),
254  ros::Duration(0.1));
255  tfListener_.lookupTransform(staticTransformsPtr_->getImuFrame(),
256  dynamic_cast<ExcavatorStaticTransforms*>(staticTransformsPtr_.get())->getBaseLinkFrame(), ros::Time(0),
257  transform_I_B);
258  // Update Imu->Base transformation
259  graph_msf::tfToIsometry3(transform_I_B, staticTransformsPtr_->lv_T_frame1_frame2(staticTransformsPtr_->getImuFrame(),
260  staticTransformsPtr_->getBaseLinkFrame()));
261  // Updaate Base->Imu transformation
262  staticTransformsPtr_->lv_T_frame1_frame2(staticTransformsPtr_->getBaseLinkFrame(), staticTransformsPtr_->getImuFrame()) =
263  staticTransformsPtr_->rv_T_frame1_frame2(staticTransformsPtr_->getImuFrame(), staticTransformsPtr_->getBaseLinkFrame()).inverse();
264 
265  // Publish state
266  graph_msf::GraphMsfRos::publishState(preIntegratedNavStatePtr, optimizedStateWithCovarianceAndBiasPtr);
267 }
