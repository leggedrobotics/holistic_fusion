158  static std::vector<double> linspace(double start, double end, int num) {
159  std::vector<double> result;
160  if (num <= 0) return result; // Empty vector if num is 0 or negative
161  if (num == 1) { // Single point case
162  result.push_back(start);
163  return result;
164  }
165 
166  double step = (end - start) / (num - 1);
167  for (int i = 0; i < num; ++i) {
168  result.push_back(start + i * step);
169  }
170 
171  return result;
172  }

174  static int filterFactorsConnectedToKey(gtsam::NonlinearFactorGraph& graph, const gtsam::Key& keyToFilter) {
175  gtsam::NonlinearFactorGraph filteredGraph;
176  int numFilteredFactors = 0;
177  for (const auto& factor : graph) {
178  bool foundKeyInFactor = false;
179  for (const gtsam::Key& key : factor->keys()) {
180  if (key == keyToFilter) {
181  foundKeyInFactor = true;
182  break;
183  }
184  }
185  if (!foundKeyInFactor) {
186  filteredGraph.add(factor);
187  } else {
188  numFilteredFactors++;
189  }
190  }
191  graph = filteredGraph;
192  return numFilteredFactors;
193  }

195  static gtsam::Marginals computeMarginals(gtsam::NonlinearFactorGraph& graph, gtsam::Values& values) {
196  // // Create linearized factor graph
197  // gtsam::GaussianFactorGraph linearizedGraph = *graph.linearize(values);
198 
199  // Try to create marginals, filter out ill-conditioned states if necessary
200  gtsam::Marginals marginals;
201  try {
202  marginals = gtsam::Marginals(graph, values);
203  } catch (const gtsam::IndeterminantLinearSystemException& e) {
204  std::cout << "GraphMSF: OptimizerLMBatch: computing marginals failed with error: " << e.what() << std::endl;
205  // Filter out ill-conditioned states
206  std::cout << "GraphMSF: OptimizerLMBatch: filtering out ill-conditioned state: " << gtsam::Symbol(e.nearbyVariable()) << std::endl;
207  // Values
208  bool foundInValues = false;
209  for (const gtsam::Key& key : values.keys()) {
210  if (key == e.nearbyVariable()) {
211  values.erase(key);
212  foundInValues = true;
213  break;
214  }
215  }
216  // Factors
217  int numFilteredFactors = filterFactorsConnectedToKey(graph, e.nearbyVariable());
218 
219  // Filtered out
220  if (foundInValues) {
221  std::cout << "GraphMSF: OptimizerLMBatch: Filtered out state from values (" << gtsam::Symbol(e.nearbyVariable()) << ")."
222  << std::endl;
223  std::cout << "GraphMSF: OptimizerLMBatch: Filtered out " << numFilteredFactors << " factors. Retrying to compute marginals."
224  << std::endl;
225  // Retry
226  marginals = OptimizerLMBatch::computeMarginals(graph, values);
227  } else {
228  throw std::runtime_error("GraphMSF: OptimizerLMBatch: computing marginals failed and could not filter out ill-conditioned state.");
229  }
230  } catch (const gtsam::ValuesKeyDoesNotExist& e) {
231  std::cout << "GraphMSF: OptimizerLMBatch: computing marginals failed with error: " << e.what() << std::endl;
232 
233  // Filtering out the factors that contain the not existing key
234  int numFilteredFactors = filterFactorsConnectedToKey(graph, e.key());
235  std::cout << "GraphMSF: OptimizerLMBatch: Filtered out " << numFilteredFactors << " factors. Retrying to compute marginals."
236  << std::endl;
237  // Retry
238  marginals = OptimizerLMBatch::computeMarginals(graph, values);
239  }
240 
241  // Return
242  return marginals;

245  void divideGraphIntoSubGraphs() {
246  // Check
247  if (!optimizedAtLeastOnceFlag_) {
248  throw std::runtime_error("GraphMSF: OptimizerLMBatch: computeAndStoreAllMarginals: No optimization has been performed yet.");
249  }
250 
251  // All Keys
252  gtsam::KeyVector keysSorted = valuesLastOptimizedResult_.keys();
253 
254  // Sort keys by time (given in keyTimestampMap)
255  std::sort(keysSorted.begin(), keysSorted.end(), [&](const gtsam::Key& key1, const gtsam::Key& key2) {
256  return batchSmootherKeyTimestampMap_.at(key1) < batchSmootherKeyTimestampMap_.at(key2);
257  });
258 
259  // Get time span and number of keys
260  double startTime = batchSmootherKeyTimestampMap_.at(keysSorted.front());
261  double endTime = batchSmootherKeyTimestampMap_.at(keysSorted.back());
262  int numKeys = keysSorted.size();
263  double timeDifference = endTime - startTime;
264 
265  // Print
266  std::cout << "GraphMSF: OptimizerLMBatch: Graph consisting of " << numKeys << " keys is spanning " << timeDifference << " seconds."
267  << std::endl;
268 
269  // Determine how many overlapping sub-graphs of size windowSizeSecond to create
270  const double& windowSizeSeconds = graphConfigPtr_->windowSizeSecondsForMarginalsComputation_; // Alias
271  int numSubGraphs = std::ceil(timeDifference / windowSizeSeconds);
272  std::cout << "GraphMSF: OptimizerLMBatch: Dividing graph into " << numSubGraphs << " main windows." << std::endl;
273  numSubGraphs += (numSubGraphs - 1); // Overlapping windows
274  std::cout << "GraphMSF: OptimizerLMBatch: Dividing graph into " << numSubGraphs
275  << " overlapping windows (to make sure there are no edges)." << std::endl;
276 
277  // Placeholders
278  std::vector<double> subGraphCenterTimes;
279  // Sub-graph start and end times
280  std::vector<double> subGraphStartTimes;
281  std::vector<double> subGraphEndTimes;
282 
283  // More than one sub-graph
284  if (numSubGraphs > 1) {
285  // Sub-graph Center Times: linearly spaced from startTime + windowSizeSeconds/2 to endTime - windowSizeSeconds/2
286  subGraphCenterTimes = linspace(startTime + windowSizeSeconds / 2, endTime - windowSizeSeconds / 2, numSubGraphs);
287  for (int i = 0; i < numSubGraphs; ++i) {
288  subGraphStartTimes.push_back(subGraphCenterTimes[i] - windowSizeSeconds / 2);
289  subGraphEndTimes.push_back(subGraphCenterTimes[i] + windowSizeSeconds / 2);
290  }
291  // One sub-graph
292  } else {
293  // Only one sub-graph
294  subGraphCenterTimes.push_back((startTime + endTime) / 2);
295  subGraphStartTimes.push_back(startTime);
296  subGraphEndTimes.push_back(endTime);
297  }
298 
299  // Print Relative Times
300  for (int i = 0; i < numSubGraphs; ++i) {
301  std::cout << "Subgraph " << i << " starts at relative " << subGraphStartTimes[i] - startTime << " seconds and ends at relative "
302  << subGraphEndTimes[i] - startTime << " seconds." << std::endl;
303  std::cout << "Center time: " << subGraphCenterTimes[i] - startTime << " seconds." << std::endl;
304  }
305 
306  // Reset all containers
307  subGraphs_.clear();
308  keyToSubGraphIndexMap_.clear();
309  marginalsForSubGraphs_.clear();
310 
311  // Create sub-graphs
312  subGraphs_.resize(numSubGraphs);
313  keyToSubGraphIndexMap_.clear();
314  marginalsForSubGraphs_.resize(numSubGraphs);
315 
316  // Go through all factors and add them to the sub-graph if they are within the time window
317  for (const auto& factorPtr : graphLastOptimizedResult_) {
318  bool factorAtLeastInOneSubGraph = false;
319  // Check whether the factor contains any key within the window of all sub-graphs
320  for (int i = 0; i < numSubGraphs; ++i) {
321  // Go through all keys of factor
322  int numReferenceFrameKeysInFactor = 0;
323  for (const auto& factorKey : factorPtr->keys()) {
324  // Skip keys of reference frame alignment factors for determining timestamp (as they can span a lot of time) unless it is random
325  // walk factor with two reference frame keys
326  if (gtsam::Symbol(factorKey).chr() == 'r') {
327  numReferenceFrameKeysInFactor++;
328  if (numReferenceFrameKeysInFactor == 1) {
329  continue; // Skip timestamp of first reference frame key
330  } else if (numReferenceFrameKeysInFactor > 1) {
331  std::cout << "Found random walk factor with (at least) two reference frame keys (including " << gtsam::Symbol(factorKey)
332  << ")." << std::endl; // Do not skip
333  }
334  }
335 
336  // Check whether key is within window
337  if (batchSmootherKeyTimestampMap_[factorKey] >= subGraphStartTimes[i] &&
338  batchSmootherKeyTimestampMap_[factorKey] <= subGraphEndTimes[i]) {
339  subGraphs_[i].add(factorPtr);
340  factorAtLeastInOneSubGraph = true;
341  break;
342  }
343  }
344  }
345  // Has to be in at least one sub-graph
346  if (!factorAtLeastInOneSubGraph) {
347  throw std::runtime_error("GraphMSF: OptimizerLMBatch: Factor does not contain any key within the time window of any sub-graph.");
348  }
349  }
350 
351  // Assign each key to the sub-graph with the closest center time to know in which sub-graph to compute marginals
352  for (const auto& key : keysSorted) {
353  double keyTime = batchSmootherKeyTimestampMap_[key];
354  // Find the closest center time
355  double minTimeDifference = std::numeric_limits<double>::max();
356  int closestSubGraphIndex = -1;
357  // Go through all sub-graphs
358  for (int i = 0; i < numSubGraphs; ++i) {
359  const double timeDifference = std::abs(subGraphCenterTimes[i] - keyTime);
360  if (timeDifference < minTimeDifference) {
361  minTimeDifference = timeDifference;
362  closestSubGraphIndex = i;
363  }
364  }
365  // Assign key to sub-graph
366  keyToSubGraphIndexMap_[key] = closestSubGraphIndex;
367  }
368 
369  // Optimize each sub-graph and compute marginals
370  for (int i = 0; i < numSubGraphs; ++i) {
371  // Print sub-graph information
372  std::cout << "GraphMSF: OptimizerLMBatch: Optimizing subgraph " << i << " with " << subGraphs_[i].size() << " factors and "
373  << subGraphs_[i].keys().size() << " keys." << std::endl;
374  // Optimize
375  marginalsForSubGraphs_[i] = computeMarginals(subGraphs_[i], valuesLastOptimizedResult_);
376  // Print
377  std::cout << "GraphMSF: OptimizerLMBatch: Subgraph " << i << " optimized and marginals computed." << std::endl;
378  }
379 
380  // Set flag
381  dividedIntoSubGraphsFlag_ = true;
382 
383  // Print
384  std::cout << "GraphMSF: OptimizerLMBatch: Divided graph into subgraphs." << std::endl;
385  }

388  gtsam::Matrix calculateMarginalCovarianceMatrixAtKey(const gtsam::Key& valueKey) override {
389  if (!optimizedAtLeastOnceFlag_) {
390  throw std::runtime_error("GraphMSF: OptimizerLMBatch: marginalCovariance: No optimization has been performed yet.");
391  }
392 
393  if (!dividedIntoSubGraphsFlag_) {
394  divideGraphIntoSubGraphs();
395  }
396 
397  // Super slow check
398  if constexpr (false) {
399  // Check whether key exists in optimized result
400  if (!valuesLastOptimizedResult_.exists(valueKey)) {
401  std::cout << "Key " << gtsam::Symbol(valueKey) << " does not exist in optimized result." << std::endl;
402  throw std::runtime_error("GraphMSF: OptimizerLMBatch: marginalCovariance: Key does not exist in optimized result.");
403  }
404  // Check whether key exists in nonlinar factor graph
405  if (!graphLastOptimizedResult_.keys().exists(valueKey)) {
406  std::cout << "Key " << gtsam::Symbol(valueKey) << " does not exist in nonlinear factor graph." << std::endl;
407  throw std::runtime_error("GraphMSF: OptimizerLMBatch: marginalCovariance: Key does not exist in nonlinear factor graph.");
408  }
409  }
410 
411  // Only use window around key for marginal covariance ------------------------------------------------
412  if (graphConfigPtr_->useWindowForMarginalsComputationFlag_) {
413  // Get sub-graph index
414  const int& subGraphsIndex_ = keyToSubGraphIndexMap_[valueKey]; // Alias
415  if (subGraphsIndex_ != currentOptimizedSubgraphIndex_) {
416  std::cout << GREEN_START << "Switching to sub-graph " << subGraphsIndex_ << " for marginal covariance computation." << COLOR_END
417  << std::endl;
418  currentOptimizedSubgraphIndex_ = subGraphsIndex_;
419  }
420  // Compute marginal covariance and measure time
421  gtsam::Matrix marginalCovariance = marginalsForSubGraphs_[subGraphsIndex_].marginalCovariance(valueKey);
422  return marginalCovariance;
423  }
424  // Normal case: Compute marginals for whole graph
425  else {
426  // Have to compute all marginals (if not done already for this result)
427  if (!marginalsComputedForLastOptimizedResultFlag_) {
428  std::cout << "Getting new marginals for last optimized result." << std::endl;
429  marginalsForLastOptimizedResult_ = gtsam::Marginals(graphLastOptimizedResult_, valuesLastOptimizedResult_);
430  marginalsComputedForLastOptimizedResultFlag_ = true;
431  }
432  // Return
433  return marginalsForLastOptimizedResult_.marginalCovariance(valueKey);
434  }
435  }
