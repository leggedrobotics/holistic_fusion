149 bool GraphMsf::addCoreImuMeasurementAndGetState(
150  const Eigen::Vector3d& linearAcc, const Eigen::Vector3d& angularVel, const double imuTimeK,
151  std::shared_ptr<SafeIntegratedNavState>& returnPreIntegratedNavStatePtr,
152  std::shared_ptr<SafeNavStateWithCovarianceAndBias>& returnOptimizedStateWithCovarianceAndBiasPtr,
153  Eigen::Matrix<double, 6, 1>& returnAddedImuMeasurements) {
154  // Setup -------------------------
155  // Increase counter
156  ++imuCallbackCounter_;
157 
158  // First Iteration
159  if (preIntegratedNavStatePtr_ == nullptr) {
160  preIntegratedNavStatePtr_ = std::make_shared<SafeIntegratedNavState>();
161  preIntegratedNavStatePtr_->updateLatestMeasurementTimestamp(imuTimeK);
162  }
163 
164  // Filter out imu messages with same time stamp
165  if (std::abs(imuTimeK - preIntegratedNavStatePtr_->getTimeK()) < 1e-8 && imuCallbackCounter_ > 1) {
166  REGULAR_COUT << RED_START << " Imu time " << std::setprecision(14) << imuTimeK << " was repeated." << COLOR_END << std::endl;
167  return false;
168  }
169 
170  // Add measurement to buffer
171  returnAddedImuMeasurements = coreImuBufferPtr_->addToImuBuffer(imuTimeK, linearAcc, angularVel);
172 
173  // Locking
174  const std::lock_guard<std::mutex> initYawAndPositionLock(initYawAndPositionMutex_);
175 
176  // State Machine in form of if-else statements -----------------
177  if (!alignedImuFlag_) { // Case 1: IMU not aligned
178  // Try to align
179  double imuAttitudeRoll, imuAttitudePitch = 0.0;
180  if (!alignImu_(imuAttitudeRoll, imuAttitudePitch)) { // Case 1.1: IMU alignment failed --> try again next time
181  // Print only once per second
182  if (imuCallbackCounter_ % int(graphConfigPtr_->imuRate_) == 0) {
183  REGULAR_COUT << " NOT ENOUGH IMU MESSAGES TO INITIALIZE POSE. WAITING FOR MORE..." << std::endl;
184  }
185  return false;
186  } else { // Case 1.2: IMU alignment succeeded --> continue next call iteration
187  Eigen::Matrix3d R_W_I0_attitude = gtsam::Rot3::Ypr(0.0, imuAttitudePitch, imuAttitudeRoll).matrix();
188  gtsam::Rot3 R_W_Init0_attitude(
189  R_W_I0_attitude *
190  staticTransformsPtr_->rv_T_frame1_frame2(staticTransformsPtr_->getImuFrame(), staticTransformsPtr_->getInitializationFrame())
191  .rotation()
192  .matrix());
193  // Set yaw of base frame to zero
194  // Set yaw of base frame to zero
195  R_W_Init0_attitude = gtsam::Rot3::Ypr(0.0, R_W_Init0_attitude.pitch(), R_W_Init0_attitude.roll());
196  R_W_I0_attitude =
197  R_W_Init0_attitude.matrix() *
198  staticTransformsPtr_->rv_T_frame1_frame2(staticTransformsPtr_->getInitializationFrame(), staticTransformsPtr_->getImuFrame())
199  .rotation()
200  .matrix();
201  Eigen::Isometry3d T_O_Ik_attitude = Eigen::Isometry3d::Identity();
202  T_O_Ik_attitude.matrix().block<3, 3>(0, 0) = R_W_I0_attitude;
203  Eigen::Vector3d O_t_O_Ik =
204  Eigen::Vector3d(0, 0, 0) -
205  R_W_I0_attitude *
206  staticTransformsPtr_->rv_T_frame1_frame2(staticTransformsPtr_->getImuFrame(), staticTransformsPtr_->getInitializationFrame())
207  .translation();
208  T_O_Ik_attitude.matrix().block<3, 1>(0, 3) = O_t_O_Ik;
209  REGULAR_COUT << " Setting zero position of " << staticTransformsPtr_->getInitializationFrame()
210  << ", hence iniital position of IMU is: " << O_t_O_Ik.transpose() << std::endl;
211  Eigen::Vector3d zeroPVeloctiy = Eigen::Vector3d(0, 0, 0);
212  preIntegratedNavStatePtr_ = std::make_shared<SafeIntegratedNavState>(T_O_Ik_attitude, zeroPVeloctiy, zeroPVeloctiy, imuTimeK);
213  REGULAR_COUT << GREEN_START << " IMU aligned. Initial pre-integrated state in odom frame: "
214  << preIntegratedNavStatePtr_->getT_O_Ik_gravityAligned().matrix() << COLOR_END << std::endl;
215  alignedImuFlag_ = true;
216  return false;
217  }
218  } else if (!areYawAndPositionInited()) { // Case 2: IMU aligned, but yaw and position not initialized, waiting for external
219  // initialization, meanwhile publishing initial roll and pitch
220  // Printing every second
221  if (imuCallbackCounter_ % int(graphConfigPtr_->imuRate_) == 0) {
222  REGULAR_COUT << " IMU callback waiting for initialization of global yaw and initial position." << std::endl;
223  }
224  // Publish state with correct roll and pitch, nothing has changed compared to Case 1.2
225  preIntegratedNavStatePtr_->updateLatestMeasurementTimestamp(imuTimeK);
226  returnPreIntegratedNavStatePtr = std::make_shared<SafeIntegratedNavState>(*preIntegratedNavStatePtr_);
227  return true;
228  } else if (!validFirstMeasurementReceivedFlag_) { // Case 3: No valid measurement received yet
229  if (imuCallbackCounter_ % int(graphConfigPtr_->imuRate_) == 0) {
230  REGULAR_COUT << RED_START << " IMU callback waiting for first valid measurement before initializing graph." << COLOR_END << std::endl;
231  }
232  preIntegratedNavStatePtr_->updateLatestMeasurementTimestamp(imuTimeK);
233  returnPreIntegratedNavStatePtr = std::make_shared<SafeIntegratedNavState>(*preIntegratedNavStatePtr_);
234  return true;
235  } else if (!initedGraphFlag_) { // Case 4: IMU aligned, yaw and position initialized, valid measurement received, but graph not yet
236  // initialized
237  preIntegratedNavStatePtr_->updateLatestMeasurementTimestamp(imuTimeK);
238  initGraph_(imuTimeK);
239  returnPreIntegratedNavStatePtr = std::make_shared<SafeIntegratedNavState>(*preIntegratedNavStatePtr_);
240  REGULAR_COUT << GREEN_START << " ...graph is initialized." << COLOR_END << std::endl;
241  return true;
242  }
243 
244  // Case 5: Normal operation, meaning predicting the next state via integration -------------
245  // Only create state every n-th measurements (or at first successful iteration)
246  bool createNewStateFlag = imuCallbackCounter_ % graphConfigPtr_->createStateEveryNthImuMeasurement_ == 0 || !normalOperationFlag_;
247  // Add IMU factor and return propagated & optimized state
248  graphMgrPtr_->addImuFactorAndGetState(*preIntegratedNavStatePtr_, returnOptimizedStateWithCovarianceAndBiasPtr, coreImuBufferPtr_,
249  imuTimeK, createNewStateFlag);
250  returnPreIntegratedNavStatePtr = std::make_shared<SafeIntegratedNavState>(*preIntegratedNavStatePtr_);
251 
252  // Set to normal operation
253  if (!normalOperationFlag_) {
254  normalOperationFlag_ = true;
255  }
256 
257  // Return
258  return true;
259 }

281 bool GraphMsf::alignImu_(double& imuAttitudeRoll, double& imuAttitudePitch) {

284  ++alignImuCounter__;

287  imuAttitudeRoll = R_W_I_rollPitch.roll();
288  imuAttitudePitch = R_W_I_rollPitch.pitch();
289  if (graphConfigPtr_->estimateGravityFromImuFlag_) {
290  graphConfigPtr_->gravityMagnitude_ = estimatedGravityMagnitude;

293  REGULAR_COUT << " Estimated gravity magnitude from IMU is: " << estimatedGravityMagnitude << std::endl;
294  REGULAR_COUT << " This gravity is not used, because estimateGravityFromImu is set to false. Gravity set to "
295  << graphConfigPtr_->gravityMagnitude_ << "." << std::endl;
296  gtsam::Vector3 gravityVector = gtsam::Vector3(0, 0, graphConfigPtr_->gravityMagnitude_);
297  gtsam::Vector3 estimatedGravityVector = gtsam::Vector3(0, 0, estimatedGravityMagnitude);
298  gtsam::Vector3 gravityVectorError = estimatedGravityVector - gravityVector;
299  gtsam::Vector3 gravityVectorErrorInImuFrame = R_W_I_rollPitch.inverse().rotate(gravityVectorError);
300  graphMgrPtr_->getInitAccBiasReference() = gravityVectorErrorInImuFrame;

305  } else {
306  return false;

337 void GraphMsf::optimizeGraph_() {
338  // While loop
339  REGULAR_COUT << " Thread for updating graph is ready." << std::endl;
340  double lastOptimizedTime = std::chrono::duration<double>(std::chrono::system_clock::now().time_since_epoch()).count();
341  double currentTime = std::chrono::duration<double>(std::chrono::system_clock::now().time_since_epoch()).count();
342  bool optimizedAtLeastOnce = false;
343  while (true) {
344  bool optimizeGraphFlag = false;
345  // Mutex for optimizeGraph Flag
346  {
347  // Lock
348  const std::lock_guard<std::mutex> optimizeGraphLock(optimizeGraphMutex_);
349  currentTime = std::chrono::duration<double>(std::chrono::system_clock::now().time_since_epoch()).count();
350  // Optimize at most at the rate of maxOptimizationFrequency but at least every second
351  if ((optimizeGraphFlag_ && ((currentTime - lastOptimizedTime) > (1.0 / graphConfigPtr_->maxOptimizationFrequency_))) ||
352  ((currentTime - lastOptimizedTime) > (1.0 / graphConfigPtr_->minOptimizationFrequency_) && optimizedAtLeastOnce)) {
353  optimizeGraphFlag = true;
354  lastOptimizedTime = currentTime;
355  optimizedAtLeastOnce = true;
356  optimizeGraphFlag_ = false;
357  }
358  }
359 
360  // Optimize
361  if (optimizeGraphFlag) {
362  graphMgrPtr_->updateGraph();
363  } // else just sleep for a short amount of time before polling again
364  else {
365  std::this_thread::sleep_for(std::chrono::milliseconds(1));
366  }
367  }
