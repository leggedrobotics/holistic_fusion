95 void ExcavatorEstimator::imuCallback_(const sensor_msgs::Imu::ConstPtr& imuMsgPtr) {
96  static bool firstCallbackFlag__ = true;
97  if (firstCallbackFlag__) {
98  firstCallbackFlag__ = false;
99  }
100  // Convert to Eigen
101  Eigen::Vector3d linearAcc(imuMsgPtr->linear_acceleration.x, imuMsgPtr->linear_acceleration.y, imuMsgPtr->linear_acceleration.z);
102  Eigen::Vector3d angularVel(imuMsgPtr->angular_velocity.x, imuMsgPtr->angular_velocity.y, imuMsgPtr->angular_velocity.z);
103  // Create Pointer for Carrying State
104  std::shared_ptr<graph_msf::SafeNavState> preIntegratedNavStatePtr;
105  std::shared_ptr<graph_msf::SafeNavStateWithCovarianceAndBias> optimizedStateWithCovarianceAndBiasPtr;
106  // Add Measurement and Get State
107  if (graph_msf::GraphMsf::addImuMeasurementAndGetState(linearAcc, angularVel, imuMsgPtr->header.stamp.toSec(), preIntegratedNavStatePtr,
108  optimizedStateWithCovarianceAndBiasPtr)) {
109  // Encountered delay
110  if (ros::Time::now() - ros::Time(preIntegratedNavStatePtr->getTimeK()) > ros::Duration(0.5)) {
111  std::cout << RED_START << "ExcavatorEstimator" << COLOR_END << " Encountered delay of "
112  << (ros::Time::now() - ros::Time(preIntegratedNavStatePtr->getTimeK())).toSec() << " seconds." << std::endl;
113  }
114  // Publish Odometry
115  this->publishState_(preIntegratedNavStatePtr, optimizedStateWithCovarianceAndBiasPtr);
116  }

119 void ExcavatorEstimator::lidarOdometryCallback_(const nav_msgs::Odometry::ConstPtr& odomLidarPtr) {
120  // Static members
121  static int odometryCallbackCounter__ = -1;
122  static std::shared_ptr<graph_msf::UnaryMeasurement6D> odometryKm1Ptr__;
123 
124  // Counter
125  ++odometryCallbackCounter__;
126 
127  Eigen::Matrix4d compslam_T_Wl_Lk;
128  graph_msf::odomMsgToEigen(*odomLidarPtr, compslam_T_Wl_Lk);
129 
130  // Transform to IMU frame
131  double timeK = odomLidarPtr->header.stamp.toSec();
132 
133  // Measurement
134  std::shared_ptr<graph_msf::UnaryMeasurement6D> odometryKPtr;
135  // Create pseudo unary factors
136  if (graphConfigPtr_->usingGnssFlag) {
137  odometryKPtr = std::make_unique<graph_msf::UnaryMeasurement6D>(
138  "Lidar 6D", dynamic_cast<ExcavatorStaticTransforms*>(staticTransformsPtr_.get())->getLidarFrame(), int(lidarRate_), timeK,
139  compslam_T_Wl_Lk, poseUnaryNoise_);
140  if (odometryCallbackCounter__ > 0) {
141  this->addDualOdometryMeasurementAndReturnNavState(*odometryKm1Ptr__, *odometryKPtr, poseBetweenNoise_);
142  }
143  } else { // real unary factors
144  odometryKPtr = std::make_unique<graph_msf::UnaryMeasurement6D>(
145  "Lidar 6D", dynamic_cast<ExcavatorStaticTransforms*>(staticTransformsPtr_.get())->getLidarFrame(), int(lidarRate_), timeK,
146  compslam_T_Wl_Lk, poseUnaryNoise_);
147  this->addUnaryPoseMeasurement(*odometryKPtr);
148  }
149 
150  if (!this->areYawAndPositionInited() && (!graphConfigPtr_->usingGnssFlag || secondsSinceStart_() > 15)) {
151  std::cout << YELLOW_START << "ExcavatorEstimator" << GREEN_START
152  << " LiDAR odometry callback is setting global cabin yaw to 0, as it was not set so far." << COLOR_END << std::endl;
153  this->initYawAndPosition(compslam_T_Wl_Lk, dynamic_cast<ExcavatorStaticTransforms*>(staticTransformsPtr_.get())->getLidarFrame());
154  }
155 
156  // Wrap up iteration
157  odometryKm1Ptr__ = odometryKPtr;
158 
159  // Add to path message
160  addToPathMsg(measWorldLidarPathPtr_, dynamic_cast<ExcavatorStaticTransforms*>(staticTransformsPtr_.get())->getWorldFrame(),
161  odomLidarPtr->header.stamp, compslam_T_Wl_Lk.block<3, 1>(0, 3), graphConfigPtr_->imuBufferLength * 4);
162 
163  // Publish Path
164  pubMeasWorldLidarPath_.publish(measWorldLidarPathPtr_);
165 }

167 void ExcavatorEstimator::gnssCallback_(const sensor_msgs::NavSatFix::ConstPtr& leftGnssMsgPtr,
168  const sensor_msgs::NavSatFix::ConstPtr& rightGnssMsgPtr) {
169  // Static method variables
170  static Eigen::Vector3d accumulatedLeftCoordinates__(0.0, 0.0, 0.0);
171  static Eigen::Vector3d accumulatedRightCoordinates__(0.0, 0.0, 0.0);
172  static Eigen::Vector3d W_t_W_GnssL_km1__, W_t_W_GnssR_km1__;
173  static int gnssCallbackCounter__ = 0;
174 
175  // Start
176  ++gnssCallbackCounter__;
177  Eigen::Vector3d leftGnssCoord = Eigen::Vector3d(leftGnssMsgPtr->latitude, leftGnssMsgPtr->longitude, leftGnssMsgPtr->altitude);
178  Eigen::Vector3d rightGnssCoord = Eigen::Vector3d(rightGnssMsgPtr->latitude, rightGnssMsgPtr->longitude, rightGnssMsgPtr->altitude);
179  Eigen::Vector3d estCovarianceXYZ(leftGnssMsgPtr->position_covariance[0], leftGnssMsgPtr->position_covariance[4],
180  leftGnssMsgPtr->position_covariance[8]);
181  if (!graphConfigPtr_->usingGnssFlag) {
182  ROS_WARN("Received Gnss message, but usage is set to false.");
183  this->activateFallbackGraph();
184  return;
185  } else if (gnssCallbackCounter__ < NUM_GNSS_CALLBACKS_UNTIL_START + 1) {
186  // Wait until measurements got accumulated
187  accumulatedLeftCoordinates__ += leftGnssCoord;
188  accumulatedRightCoordinates__ += rightGnssCoord;
189  if (!(gnssCallbackCounter__ % 10)) {
190  std::cout << YELLOW_START << "ExcavatorEstimator" << COLOR_END << " NOT ENOUGH Gnss MESSAGES ARRIVED!" << std::endl;
191  }
192  return;
193  } else if (gnssCallbackCounter__ == NUM_GNSS_CALLBACKS_UNTIL_START + 1) {
194  gnssHandlerPtr_->initHandler(accumulatedLeftCoordinates__ / NUM_GNSS_CALLBACKS_UNTIL_START,
195  accumulatedRightCoordinates__ / NUM_GNSS_CALLBACKS_UNTIL_START);
196  }
197 
198  // Convert to cartesian coordinates
199  Eigen::Vector3d W_t_W_GnssL, W_t_W_GnssR;
200  gnssHandlerPtr_->convertNavSatToPositions(leftGnssCoord, rightGnssCoord, W_t_W_GnssL, W_t_W_GnssR);
201  double yaw_W_C = gnssHandlerPtr_->computeYaw(W_t_W_GnssL, W_t_W_GnssR);
202 
203  // Initialization
204  if (gnssCallbackCounter__ == NUM_GNSS_CALLBACKS_UNTIL_START + 1) {
205  if (!this->initYawAndPosition(yaw_W_C, dynamic_cast<ExcavatorStaticTransforms*>(staticTransformsPtr_.get())->getCabinFrame(),
206  W_t_W_GnssL, dynamic_cast<ExcavatorStaticTransforms*>(staticTransformsPtr_.get())->getLeftGnssFrame())) {
207  // Decrease counter if not successfully initialized
208  --gnssCallbackCounter__;
209  }
210  } else {
211  graph_msf::UnaryMeasurement1D meas_yaw_W_C("Gnss yaw",
212  dynamic_cast<ExcavatorStaticTransforms*>(staticTransformsPtr_.get())->getCabinFrame(),
213  int(gnssLeftRate_), leftGnssMsgPtr->header.stamp.toSec(), yaw_W_C, gnssHeadingUnaryNoise_);
214  this->addGnssHeadingMeasurement(meas_yaw_W_C);
215  graph_msf::UnaryMeasurement3D meas_W_t_W_GnssL(
216  "Gnss left", dynamic_cast<ExcavatorStaticTransforms*>(staticTransformsPtr_.get())->getLeftGnssFrame(), gnssLeftRate_,
217  leftGnssMsgPtr->header.stamp.toSec(), W_t_W_GnssL,
218  Eigen::Vector3d(gnssPositionUnaryNoise_, gnssPositionUnaryNoise_, gnssPositionUnaryNoise_));
219  this->addDualGnssPositionMeasurement(meas_W_t_W_GnssL, W_t_W_GnssL_km1__, estCovarianceXYZ, true, true);
220  }
221  W_t_W_GnssL_km1__ = W_t_W_GnssL;
222  W_t_W_GnssR_km1__ = W_t_W_GnssR;
223 
224  // Left GNSS
226  addToPathMsg(measWorldLeftGnssPathPtr_, dynamic_cast<ExcavatorStaticTransforms*>(staticTransformsPtr_.get())->getWorldFrame(),
227  leftGnssMsgPtr->header.stamp, W_t_W_GnssL, graphConfigPtr_->imuBufferLength * 4);
229  pubMeasWorldGnssLPath_.publish(measWorldLeftGnssPathPtr_);
230  // Right GNSS
232  addToPathMsg(measWorldRightGnssPathPtr_, dynamic_cast<ExcavatorStaticTransforms*>(staticTransformsPtr_.get())->getWorldFrame(),
233  rightGnssMsgPtr->header.stamp, W_t_W_GnssR, graphConfigPtr_->imuBufferLength * 4);
235  pubMeasWorldGnssRPath_.publish(measWorldRightGnssPathPtr_);
236 }

238 void ExcavatorEstimator::publishState_(
239  const std::shared_ptr<graph_msf::SafeNavState>& navStatePtr,
240  const std::shared_ptr<graph_msf::SafeNavStateWithCovarianceAndBias>& optimizedStateWithCovarianceAndBiasPtr) {
241  // Lookup I->B, also influenced by rotation of cabin
242  static tf::StampedTransform transform_I_B;
243  tfListener_.waitForTransform(staticTransformsPtr_->getImuFrame(),
244  dynamic_cast<ExcavatorStaticTransforms*>(staticTransformsPtr_.get())->getBaseLinkFrame(), ros::Time(0),
245  ros::Duration(0.1));
246  listener_.lookupTransform(staticTransformsPtr_->getImuFrame(),
247  dynamic_cast<ExcavatorStaticTransforms*>(staticTransformsPtr_.get())->getBaseLinkFrame(), ros::Time(0),
248  transform_I_B);
249  // Update Imu->Base transformation
250  graph_msf::tfToIsometry3(transform_I_B, staticTransformsPtr_->lv_T_frame1_frame2(staticTransformsPtr_->getImuFrame(),
251  staticTransformsPtr_->getBaseLinkFrame()));
252  // Updaate Base->Imu transformation
253  staticTransformsPtr_->lv_T_frame1_frame2(staticTransformsPtr_->getBaseLinkFrame(), staticTransformsPtr_->getImuFrame()) =
254  staticTransformsPtr_->rv_T_frame1_frame2(staticTransformsPtr_->getImuFrame(), staticTransformsPtr_->getBaseLinkFrame()).inverse();
255 
256  // Publish state
257  graph_msf::GraphMsfRos::publishState_(navStatePtr, optimizedStateWithCovarianceAndBiasPtr);
258 }
