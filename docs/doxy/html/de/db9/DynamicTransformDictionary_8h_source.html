22 // Explanation:
23 // 1) global variables are constant along the entire operation, hence they are not removed and will get the same gtsam key upon return
24 // 2) reference frames can change, but are modelled according to random walk, hence they are only removed once a successor is initialized
25 // 3) landmarks are initialized and removed as needed, hence they are removed once they are not needed anymore
26 // Frames that are not needed but also not removed are set to be "inactive"

42  bool removeTransform(const std::string& frame1, const std::string& frame2,
43  DynamicFactorGraphStateKey<GTSAM_DYNAMIC_STATE_TYPE>& returnRemovedKey) override {
44  // Removed main element
45  bool removedTransformFlag =
46  TransformsDictionary<DynamicFactorGraphStateKey<GTSAM_DYNAMIC_STATE_TYPE>>::removeTransform(frame1, frame2, returnRemovedKey);
47  // If found, we also remove the key from the key-to-frame pair map
48  if (removedTransformFlag) {
49  auto keyFramePairMapIterator = transformGtsamKeyToFramePairMap_.find(returnRemovedKey.key());
50  // Still in map --> remove
51  if (keyFramePairMapIterator != transformGtsamKeyToFramePairMap_.end()) {
52  // transformGtsamKeyToFramePairMap_.erase(keyFramePairMapIterator);
53  // TODO: Remove if offline optimization is disabled, keep otherwise
54  }
55  // Not in map --> throw error
56  else {
57  REGULAR_COUT << RED_START << " Key " << gtsam::Symbol(returnRemovedKey.key()) << " not found in map." << COLOR_END << std::endl;
58  throw std::runtime_error("Key not found in map.");
59  }
60  }
61  return removedTransformFlag;
62  }

64  // Same but with only two arguments
65  bool removeOrDeactivateTransform(const std::string& frame1, const std::string& frame2) {
66  // Check if key is in map
67  const bool isFramePairInDictionary =
68  TransformsDictionary<DynamicFactorGraphStateKey<GTSAM_DYNAMIC_STATE_TYPE>>::isFramePairInDictionary(frame1, frame2);
69 
70  // Case: Frame pair is in dictionary
71  if (isFramePairInDictionary) {
72  // Retrieve key and variable type
73  DynamicFactorGraphStateKey<GTSAM_DYNAMIC_STATE_TYPE>& keyToRemoveOrDeactivate =
74  TransformsDictionary<DynamicFactorGraphStateKey<GTSAM_DYNAMIC_STATE_TYPE>>::lv_T_frame1_frame2(frame1, frame2);
75  const DynamicVariableTypeEnum& variableTypeEnum = keyToRemoveOrDeactivate.getVariableTypeEnum();
76 
77  // Three different cases
78  switch (variableTypeEnum) {
79  // Case 1: Global Measurement --> Never remove, only deactivate, as it might come back
80  case DynamicVariableTypeEnum::Global:
81  // Currently active, so we can deactivate
82  if (keyToRemoveOrDeactivate.isVariableActive()) {
83  keyToRemoveOrDeactivate.deactivateVariable();
84  return true;
85  }
86  // Already deactivated, so we do not need to do anything
87  else {
88  return false;
89  }
90  // Case 2: Reference Frame --> Do not remove for now, as we will need it for creating the random walk
91  case DynamicVariableTypeEnum::RefFrame:
92  // Currently active, so we can deactivate
93  if (keyToRemoveOrDeactivate.isVariableActive()) {
94  keyToRemoveOrDeactivate.deactivateVariable();
95  return true;
96  }
97  // Already deactivated, so we do not need to do anything
98  else {
99  return false;
100  }
101  // Case 3: Landmark --> Remove, as the landmark is not needed anymore
102  case DynamicVariableTypeEnum::Landmark:
103  return removeTransform(frame1, frame2, keyToRemoveOrDeactivate);
104  // Has to be one of the three cases
105  default:
106  throw std::logic_error("Variable Type not found.");
107  }
108  }
109  // Case: Frame pair is not in dictionary
110  else {
111  return false;
112  }
113  }

159  // Safe addition of new frame pair to dictionary --> checks whether already present
160  template <char SYMBOL_CHAR>
161  bool addNewFramePairSafelyToDictionary(DynamicFactorGraphStateKey<GTSAM_DYNAMIC_STATE_TYPE>& returnStateKey, const std::string& frame1,
162  const std::string& frame2, const double timeK,
163  const gtsam::Pose3& approximateTransformationBeforeOptimization,
164  const DynamicVariableType& variableType) {
165  // Check and modify content --> acquire lock
166  std::lock_guard<std::mutex> lock(internalDictionaryModifierMutex_);
167 
168  // Logic
169  // CASE 1: Frame pair is already in dictionary, hence keyframe is not new (measurement either active or not)
170  if (TransformsDictionary<DynamicFactorGraphStateKey<GTSAM_DYNAMIC_STATE_TYPE>>::isFramePairInDictionary(frame1, frame2)) {
171  DynamicFactorGraphStateKey<GTSAM_DYNAMIC_STATE_TYPE>& factorGraphStateKey =
172  TransformsDictionary<DynamicFactorGraphStateKey<GTSAM_DYNAMIC_STATE_TYPE>>::lv_T_frame1_frame2(frame1, frame2);
173  // Update Timestamp and approximate transformation if newer
174  if (timeK > factorGraphStateKey.getTime()) {
175  // std::cout << " Updating timestamp of key " << gtsam::Symbol(factorGraphStateKey.key()) << " from " <<
176  // std::setprecision(14)
177  // << factorGraphStateKey.getTime() << " to " << timeK << "." << std::endl;
178  factorGraphStateKey.setTimeStamp(timeK);
179  factorGraphStateKey.setApproximateTransformationBeforeOptimization(approximateTransformationBeforeOptimization);
180  }
181  returnStateKey = factorGraphStateKey;
182  return false;
183  }
184  // CASE 2: Frame pair is not in dictionary --> Newly added
185  else {
186  returnStateKey =
187  addNewFactorGraphStateKey<SYMBOL_CHAR>(frame1, frame2, timeK, approximateTransformationBeforeOptimization, variableType);
188  return true;
189  }
190  }

193  template <char SYMBOL_CHAR>
194  DynamicFactorGraphStateKey<GTSAM_DYNAMIC_STATE_TYPE> addNewFactorGraphStateKey(
195  const std::string& frame1, const std::string& frame2, const double timeK,
196  const gtsam::Pose3& approximateTransformationBeforeOptimization, const DynamicVariableType& variableType) {
197  // Compile Time Checks and Variables
198  checkSymbol<SYMBOL_CHAR>();
199  constexpr int symbolIndex = getSymbolIndex<SYMBOL_CHAR>();
200  static_assert(symbolIndex >= 0, "Symbol not found in admissible symbols.");
201 
202  // Create New Key
203  gtsam::Key gtsamKey = gtsam::Symbol(SYMBOL_CHAR, numStoredTransformsPerSymbol_[symbolIndex]);
204 
205  // Print out for calibrations and reference frames
206  if (variableType.getVariableTypeEnum() != DynamicVariableTypeEnum::Landmark) {
207  REGULAR_COUT << GREEN_START << " New key " << gtsam::Symbol(gtsamKey) << " created for frame pair " << frame1 << " and " << frame2
208  << "." << COLOR_END << std::endl;
209  }
210 
211  // Add to main dictionary
212  DynamicFactorGraphStateKey factorGraphStateKey(gtsamKey, timeK, 0, approximateTransformationBeforeOptimization, variableType, frame1,
213  frame2);
214  TransformsDictionary<DynamicFactorGraphStateKey<GTSAM_DYNAMIC_STATE_TYPE>>::set_T_frame1_frame2(frame1, frame2, factorGraphStateKey);
215 
216  // Increase the counter of specific symbol
217  ++numStoredTransformsPerSymbol_[symbolIndex];
218 
219  // Add to key-to-frame pair map
220  // TODO: Check if necessary for non reference frames
221  transformGtsamKeyToFramePairMap_[gtsamKey] = std::make_pair(frame1, frame2);
222  // Add to key-to-keyframe position map
223  transformGtsamKeyToKeyframePositionMap_[gtsamKey] = variableType.getReferenceFrameKeyframePosition();
224 
225  // Return
226  return factorGraphStateKey;
227  }
