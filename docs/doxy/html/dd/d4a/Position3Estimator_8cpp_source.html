143 bool Position3Estimator::srvOfflineSmootherOptimizeCallback(graph_msf_ros_msgs::OfflineOptimizationTrigger::Request& req,
144  graph_msf_ros_msgs::OfflineOptimizationTrigger::Response& res) {
145  std::cout << "Position3Estimator: srvOfflineSmootherOptimizeCallback (derived class) called." << std::endl;
146  // Call parent class and create most of the files already
147  bool success = graph_msf::GraphMsfRos::srvOfflineSmootherOptimizeCallback(req, res);
148 
149  // File streams
150  std::map<std::string, std::ofstream> fileStreams;
151 
152  // Write T_totalStation_totalStationOld_ to file ----------------------------
153  // Get time string by finding latest directory in optimizationResultLoggingPath
154  std::string timeString = graph_msf::getLatestSubdirectory(optimizationResultLoggingPath);
155  if (timeString.empty()) {
156  REGULAR_COUT << " Could not find latest directory in " << optimizationResultLoggingPath << std::endl;
157  return false;
158  } else {
159  REGULAR_COUT << " Found latest directory in " << optimizationResultLoggingPath << ": " << timeString << std::endl;
160  }
161  // Create file stream
162  std::string transformIdentifier = "R_6D_transform_" + totalStationReferenceFrame_ + "_to_" + totalStationReferenceFrame_ + "Old";
163  graph_msf::createPose3CsvFileStream(fileStreams, optimizationResultLoggingPath, transformIdentifier, timeString, false);
164  // Add to file
165  graph_msf::writePose3ToCsvFile(fileStreams, T_totalStation_totalStationOld_, transformIdentifier, ros::Time::now().toSec(), false);
166  REGULAR_COUT << " Wrote T_totalStation_totalStationOld to file." << std::endl;
167 
168  // Depending on whether GNSS or Prism was used for initialization, write the corresponding T_W_R identity to file ------------------------
169  if (initializeUsingGnssFlag_) {
170  // Write T_totalStation_totalStationOld_ to file
171  std::string transformIdentifier = "R_6D_transform_" + staticTransformsPtr_->getWorldFrame() + "_to_" + gnssReferenceFrame_;
172  graph_msf::createPose3CsvFileStream(fileStreams, optimizationResultLoggingPath, transformIdentifier, timeString, false);
173  // Add to file
174  graph_msf::writePose3ToCsvFile(fileStreams, Eigen::Isometry3d::Identity(), transformIdentifier, ros::Time::now().toSec(), false);
175  REGULAR_COUT << " Wrote T_W_gnssReferenceFrame to file." << std::endl;
176  } else {
177  // Write T_totalStation_enu_ to file
178  std::string transformIdentifier = "R_6D_transform_" + staticTransformsPtr_->getWorldFrame() + "_to_" + totalStationReferenceFrame_;
179  graph_msf::createPose3CsvFileStream(fileStreams, optimizationResultLoggingPath, transformIdentifier, timeString, false);
180  // Add to file
181  graph_msf::writePose3ToCsvFile(fileStreams, T_totalStation_enu_, transformIdentifier, ros::Time::now().toSec(), false);
182  REGULAR_COUT << " Wrote T_W_totalStationReferenceFrame to file." << std::endl;
183  }
184 
185  // Close all file streams
186  for (auto& pair : fileStreams) {
187  pair.second.close();
188  }
189  // Return
190  return success;
191 }

194 void Position3Estimator::prismPositionCallback_(const geometry_msgs::PointStamped::ConstPtr& leicaPositionPtr) {
195  // Counter
196  prismPositionCallbackCounter_++;
197 
198  // Fixed Frame
199  std::string fixedFrame;
200  if (!initializeUsingGnssFlag_) {
201  fixedFrame = staticTransformsPtr_->getWorldFrame();
202  } else {
203  fixedFrame = leicaPositionPtr->header.frame_id;
204  }
205 
206  // Translate to Eigen
207  Eigen::Vector3d positionMeas = Eigen::Vector3d(leicaPositionPtr->point.x, leicaPositionPtr->point.y, leicaPositionPtr->point.z);
208  Eigen::Vector3d positionCovarianceXYZ(prismPositionMeasUnaryNoise_, prismPositionMeasUnaryNoise_, prismPositionMeasUnaryNoise_);
209 
210  // Case: First call
211  if (prismPositionCallbackCounter_ == 1) {
212  REGULAR_COUT << " First prism position measurement received. Setting initial position to " << positionMeas.transpose() << std::endl;
213  // Set initial position
214  initialPrismPosition_ = positionMeas;
215  // Set Reference Frame
216  totalStationReferenceFrame_ = leicaPositionPtr->header.frame_id;
217  }
218 
219  // Case: Not moved enough --> check if moved enough in meanwhile, only relevant if not initialized by GNSS
220  if (!initializeUsingGnssFlag_ && !prismMovedEnoughFlag_) {
221  if ((positionMeas - initialPrismPosition_).norm() > PRISM_MOVED_ENOUGH_THRESHOLD) {
222  prismMovedEnoughFlag_ = true;
223  REGULAR_COUT << " Prism moved enough from initial position of " << initialPrismPosition_.transpose() << " to "
224  << positionMeas.transpose() << std::endl;
225  }
226  }
227 
228  // Flag
229  bool addToGraphFlag = true;
230 
231  // Case: Graph is not initialized by this callback, but instead by GNSS
232  if (initializeUsingGnssFlag_ && !alignedPrismAndGnssFlag_) {
233  trajectoryAlignmentHandler_->addR3Position(positionMeas, leicaPositionPtr->header.stamp.toSec());
234  // In radians
235  double yaw{0};
236  if (!(trajectoryAlignmentHandler_->alignTrajectories(yaw, T_totalStation_enu_))) {
237  if (prismPositionCallbackCounter_ % 10 == 0) {
238  REGULAR_COUT << " Graph was initialized by GNSS, hence trying to align. Trajectory alignment not ready. Waiting for more motion."
239  << std::endl;
240  }
241  addToGraphFlag = false;
242  } else {
243  T_totalStation_totalStationOld_ = T_totalStation_enu_.inverse();
244  // Only keep yaw of the orientation part of the transformation
245  REGULAR_COUT << GREEN_START << " Total Station to Old Total Station Transformation: " << COLOR_END
246  << T_totalStation_totalStationOld_.matrix() << std::endl;
247  graph_msf::inPlaceRemoveRollPitch(T_totalStation_totalStationOld_);
248  REGULAR_COUT << GREEN_START << " Total Station to Old Total Station Transformation after removing roll and pitch: " << COLOR_END
249  << T_totalStation_totalStationOld_.matrix() << std::endl;
250  // Prepare for using this transformation in the graph
251  alignedPrismAndGnssFlag_ = true;
252  return; // Do not add unary measurement this round
253  }
254  }
255 
256  // State Machine
257  if (!areYawAndPositionInited() && areRollAndPitchInited() && !initializeUsingGnssFlag_) {
258  // Try to initialize yaw and position if not done already
259  if (this->initYawAndPositionInWorld(
260  0.0, positionMeas, staticTransformsPtr_->getBaseLinkFrame(),
261  dynamic_cast<Position3StaticTransforms*>(staticTransformsPtr_.get())->getPrismPositionMeasFrame())) {
262  std::cout << "-------------------" << std::endl;
263  REGULAR_COUT << RED_START << " Prism callback is setting global yaw." << COLOR_END << std::endl;
264  std::cout << "-------------------" << std::endl;
265  REGULAR_COUT << " Set yaw and position successfully." << std::endl;
266  } else {
267  REGULAR_COUT << " Could not set yaw and position." << std::endl;
268  }
269  }
270  // Else if active
271  else if (usePrismPositionUnaryFlag_ && addToGraphFlag) {
272  const std::string& positionMeasFrame =
273  dynamic_cast<Position3StaticTransforms*>(staticTransformsPtr_.get())->getPrismPositionMeasFrame();
274  // Consider whether graph was initialized by GNSS or not ----------------------------
275  // if (initializedByGnssFlag_) {
276  // fixedFrame = leicaPositionPtr->header.frame_id;
277  // } else {
278  // fixedFrame = staticTransformsPtr_->getWorldFrame();
279  // }
280  // Potentially transform measurement to start ENU frame (if world origin was set to coincide with GNSS)
281  if (alignedPrismAndGnssFlag_) {
282  positionMeas = T_totalStation_totalStationOld_ * positionMeas;
283  }
284 
285  // Already initialized --> add position measurement to graph ----------------------------
286  graph_msf::UnaryMeasurementXDAbsolute<Eigen::Vector3d, 3> meas_W_t_W_P(
287  "LeicaPosition", int(prismPositionRate_), positionMeasFrame, positionMeasFrame + sensorFrameCorrectedNameId,
288  graph_msf::RobustNorm::None(), leicaPositionPtr->header.stamp.toSec(), POS_COVARIANCE_VIOLATION_THRESHOLD, positionMeas,
289  positionCovarianceXYZ, fixedFrame, staticTransformsPtr_->getWorldFrame(), initialSe3AlignmentNoise_, prismSe3AlignmentRandomWalk_);
290  this->addUnaryPosition3AbsoluteMeasurement(meas_W_t_W_P);
291  }
292  // Else if not active
293  else if ((prismPositionCallbackCounter_ % 10) == 0 && addToGraphFlag) {
294  REGULAR_COUT << " Prism unary measurement is turned off." << std::endl;
295  }
296 
297  // Visualizations
298  // Fixed Frame
299  std::string msgFixedFrame = fixedFrame;
300  if (msgFixedFrame != staticTransformsPtr_->getWorldFrame()) {
301  msgFixedFrame += referenceFrameAlignedNameId;
302  }
303  // Add to Path
304  if (!initializeUsingGnssFlag_ || alignedPrismAndGnssFlag_) {
305  addToPathMsg(measPosition_worldPrismPositionPathPtr_, msgFixedFrame, leicaPositionPtr->header.stamp, positionMeas,
306  graphConfigPtr_->imuBufferLength_ * 4);
307  pubMeasWorldPrismPositionPath_.publish(measPosition_worldPrismPositionPathPtr_);
308  }
309 }

312 void Position3Estimator::gnssPositionCallback_(const sensor_msgs::NavSatFix::ConstPtr& gnssPositionPtr) {
313  throw std::runtime_error("Position3Estimator: outdated in terms of logic.");
314 
315  // Counter
316  gnssPositionCallbackCounter_++;
317 
318  // Translate to Eigen
319  Eigen::Vector3d gnssCoord = Eigen::Vector3d(gnssPositionPtr->latitude, gnssPositionPtr->longitude, gnssPositionPtr->altitude);
320  Eigen::Vector3d estStdDevXYZ(sqrt(gnssPositionPtr->position_covariance[0]), sqrt(gnssPositionPtr->position_covariance[4]),
321  sqrt(gnssPositionPtr->position_covariance[8]));
322 
323  // Initialize GNSS Handler
324  if (gnssPositionCallbackCounter_ < NUM_GNSS_CALLBACKS_UNTIL_START) { // Accumulate measurements
325  // Wait until measurements got accumulated
326  accumulatedGnssCoordinates_ += gnssCoord;
327  if ((gnssPositionCallbackCounter_ % 10) == 0) {
328  REGULAR_COUT << " NOT ENOUGH GNSS MESSAGES ARRIVED!" << std::endl;
329  }
330  return;
331  } else if (gnssPositionCallbackCounter_ == NUM_GNSS_CALLBACKS_UNTIL_START) { // Initialize GNSS Handler
332  gnssHandlerPtr_->initHandler(accumulatedGnssCoordinates_ / NUM_GNSS_CALLBACKS_UNTIL_START);
333  REGULAR_COUT << " GNSS Handler initialized." << std::endl;
334  return;
335  }
336 
337  // Convert to Cartesian Coordinates
338  Eigen::Vector3d W_t_W_Gnss;
339  gnssHandlerPtr_->convertNavSatToPosition(gnssCoord, W_t_W_Gnss);
340  // std::string fixedFrame = staticTransformsPtr_->getWorldFrame();
341  std::string fixedFrame = gnssPositionPtr->header.frame_id;
342 
343  // Regular Operation
344  // Initialize if needed and no prism is used
345  if (!areYawAndPositionInited() && areRollAndPitchInited() && initializeUsingGnssFlag_ && !constexprUsePrismPositionUnaryFlag_) {
346  // Try to initialize yaw and position if not done already
347  if (this->initYawAndPositionInWorld(0.0, W_t_W_Gnss, staticTransformsPtr_->getBaseLinkFrame(),
348  dynamic_cast<Position3StaticTransforms*>(staticTransformsPtr_.get())->getGnssPositionMeasFrame())) {
349  REGULAR_COUT << " GNSS set yaw and position successfully, as there is no prism." << std::endl;
350  } else {
351  REGULAR_COUT << " Could not set yaw and position." << std::endl;
352  }
353  }
354  // Else if inited or prism is used
355  else if (areRollAndPitchInited() && useGnssPositionUnaryFlag_) {
356  const std::string& positionMeasFrame = dynamic_cast<Position3StaticTransforms*>(staticTransformsPtr_.get())->getGnssPositionMeasFrame();
357  // Already initialized --> add position measurement to graph
358  graph_msf::UnaryMeasurementXDAbsolute<Eigen::Vector3d, 3> meas_W_t_W_P(
359  "GnssPosition", int(gnssPositionRate_), positionMeasFrame, positionMeasFrame + sensorFrameCorrectedNameId,
360  graph_msf::RobustNorm::None(), gnssPositionPtr->header.stamp.toSec(), POS_COVARIANCE_VIOLATION_THRESHOLD, W_t_W_Gnss, estStdDevXYZ,
361  fixedFrame, staticTransformsPtr_->getWorldFrame(), initialSe3AlignmentNoise_, gnssSe3AlignmentRandomWalk_);
362  this->addUnaryPosition3AbsoluteMeasurement(meas_W_t_W_P);
363  }
364  // Status
365  else if ((gnssPositionCallbackCounter_ % 10) == 0) {
366  REGULAR_COUT << " GNSS unary measurement is turned off or roll/pitch not yet inited." << std::endl;
367  }
368 
369  // Adjust frame name
370  if (fixedFrame != staticTransformsPtr_->getWorldFrame()) {
371  fixedFrame += referenceFrameAlignedNameId;
372  }
373 
374  // Visualizations
375  addToPathMsg(measPosition_worldGnssPositionPathPtr_, fixedFrame, gnssPositionPtr->header.stamp, W_t_W_Gnss,
376  graphConfigPtr_->imuBufferLength_ * 4);
377  // Publish Path
378  pubMeasWorldGnssPositionPath_.publish(measPosition_worldGnssPositionPathPtr_);
379 }

382 void Position3Estimator::gnssOfflinePoseCallback_(const nav_msgs::Odometry::ConstPtr& gnssOfflinePosePtr) {
383  // Counter
384  gnssOfflinePoseCallbackCounter_++;
385 
386  // Fixed Frame
387  std::string fixedFrame;
388  if (initializeUsingGnssFlag_) {
389  fixedFrame = staticTransformsPtr_->getWorldFrame();
390  } else {
391  fixedFrame = gnssOfflinePosePtr->header.frame_id;
392  }
393 
394  // Case: First call
395  if (gnssOfflinePoseCallbackCounter_ == 1) {
396  REGULAR_COUT << " First GNSS offline pose measurement received." << std::endl;
397  // Set Reference Frame
398  gnssReferenceFrame_ = gnssOfflinePosePtr->header.frame_id;
399  }
400 
401  // Prepare Data
402  Eigen::Isometry3d T_ENU_Gk = Eigen::Isometry3d::Identity();
403  graph_msf::odomMsgToEigen(*gnssOfflinePosePtr, T_ENU_Gk.matrix());
404  double gnssUnaryTimeK = gnssOfflinePosePtr->header.stamp.toSec();
405 
406  // If initialized by gnss --> make sure we can do the alignment with prism
407  if (initializeUsingGnssFlag_ && usePrismPositionUnaryFlag_ && !alignedPrismAndGnssFlag_) {
408  trajectoryAlignmentHandler_->addSe3Position(T_ENU_Gk.translation(), gnssOfflinePosePtr->header.stamp.toSec());
409  }
410 
411  // Get Uncertainty from message
412  auto estGnssOfflinePoseMeasUnaryNoiseList = gnssOfflinePosePtr->pose.covariance;
413  Eigen::Matrix<double, 6, 1> estGnssOfflinePoseMeasUnaryNoise;
414  estGnssOfflinePoseMeasUnaryNoise << sqrt(estGnssOfflinePoseMeasUnaryNoiseList[0]), sqrt(estGnssOfflinePoseMeasUnaryNoiseList[7]),
415  sqrt(estGnssOfflinePoseMeasUnaryNoiseList[14]), sqrt(estGnssOfflinePoseMeasUnaryNoiseList[21]),
416  sqrt(estGnssOfflinePoseMeasUnaryNoiseList[28]), sqrt(estGnssOfflinePoseMeasUnaryNoiseList[35]);
417 
418  // Measurement
419  const std::string& gnssFrameName =
420  dynamic_cast<Position3StaticTransforms*>(staticTransformsPtr_.get())->getGnssOfflinePoseMeasFrame(); // alias
421  graph_msf::UnaryMeasurementXDAbsolute<Eigen::Isometry3d, 6> unary6DMeasurement(
422  "Lidar_unary_6D", int(gnssOfflinePoseRate_), gnssFrameName, gnssFrameName + sensorFrameCorrectedNameId, graph_msf::RobustNorm::None(),
423  gnssUnaryTimeK, 1.0, T_ENU_Gk, gnssOfflinePoseMeasUnaryNoise_, fixedFrame, staticTransformsPtr_->getWorldFrame(),
424  initialSe3AlignmentNoise_, gnssSe3AlignmentRandomWalk_);
425 
426  // Wait until enough measurements have arrived
427  if (gnssOfflinePoseCallbackCounter_ <= 2) {
428  return;
429  }
430  // Initialize if needed and no prism is used
431  else if (!areYawAndPositionInited() && areRollAndPitchInited() && initializeUsingGnssFlag_) { // Initializing if no GNSS
432  std::cout << "-------------------" << std::endl;
433  REGULAR_COUT << RED_START << " GNSS offline odometry callback is setting global yaw." << COLOR_END << std::endl;
434  std::cout << "-------------------" << std::endl;
435  this->initYawAndPosition(unary6DMeasurement);
436  }
437  // Otherwise just add measurement
438  else if (areYawAndPositionInited()) { // Already initialized --> unary factor
439  // If prism has not moved enough, do not add GNSS measurements
440  bool addToOnlineSmootherFlag = true;
441  if (!prismMovedEnoughFlag_ && !initializeUsingGnssFlag_) {
442  if ((gnssOfflinePoseCallbackCounter_ % 100) == 0) {
443  REGULAR_COUT << " PRISM HAS NOT MOVED ENOUGH YET! Not adding GNSS measurements to online graph." << std::endl;
444  }
445  addToOnlineSmootherFlag = false;
446  }
447  this->addUnaryPose3AbsoluteMeasurement(unary6DMeasurement, addToOnlineSmootherFlag);
448  }
449 
450  // Visualization ----------------------------
451  // Add to path message
452  addToPathMsg(measPosition_worldGnssOfflinePosePathPtr_, fixedFrame, gnssOfflinePosePtr->header.stamp, T_ENU_Gk.translation(),
453  graphConfigPtr_->imuBufferLength_ * 4);
454 
455  // Publish Path
456  pubMeasWorldGnssOfflinePosePath_.publish(measPosition_worldGnssOfflinePosePathPtr_);
457 }
