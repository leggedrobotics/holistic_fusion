100 bool GraphManager::initPoseVelocityBiasGraph(const double timeStamp, const gtsam::Pose3& T_W_I0, const gtsam::Pose3& T_O_I0) {
101  // Create Prior factor ----------------------------------------------------
103  auto priorPoseNoise = gtsam::noiseModel::Diagonal::Sigmas(
104  (gtsam::Vector(6) << graphConfigPtr_->initialOrientationNoiseDensity_, graphConfigPtr_->initialOrientationNoiseDensity_,
105  graphConfigPtr_->initialOrientationNoiseDensity_, graphConfigPtr_->initialPositionNoiseDensity_,
106  graphConfigPtr_->initialPositionNoiseDensity_, graphConfigPtr_->initialPositionNoiseDensity_)
107  .finished()); // rad,rad,rad,m, m, m
108  auto priorVelocityNoise = gtsam::noiseModel::Isotropic::Sigma(3, graphConfigPtr_->initialVelocityNoiseDensity_); // m/s
109  auto priorBiasNoise = gtsam::noiseModel::Diagonal::Sigmas((gtsam::Vector(6) << graphConfigPtr_->initialAccBiasNoiseDensity_, // m/s^2
110  graphConfigPtr_->initialAccBiasNoiseDensity_, // m/s^2
111  graphConfigPtr_->initialAccBiasNoiseDensity_, // m/s^2
112  graphConfigPtr_->initialGyroBiasNoiseDensity_, // rad/s
113  graphConfigPtr_->initialGyroBiasNoiseDensity_, // rad/s
114  graphConfigPtr_->initialGyroBiasNoiseDensity_) // rad/s
115  .finished()); // acc, acc, acc, gyro, gyro, gyro
116 
117  // Pre-allocate
118  gtsam::NonlinearFactorGraph newRtGraphFactors, newBatchGraphFactors;
119  gtsam::Values newRtGraphValues, newBatchGraphValues;
120  std::shared_ptr<std::map<gtsam::Key, double>> priorKeyTimestampMapPtr = std::make_shared<std::map<gtsam::Key, double>>();
121 
122  // Looking up from IMU buffer --> acquire mutex (otherwise values for key might not be set)
123  {
124  const std::lock_guard<std::mutex> operateOnGraphDataLock(operateOnGraphDataMutex_);
125 
126  // Initial estimate
127  gtsam::Values valuesEstimate;
128  REGULAR_COUT << " Initial Pose of imu in world frame: " << T_W_I0 << std::endl;
129  valuesEstimate.insert(gtsam::symbol_shorthand::X(propagatedStateKey_), T_W_I0);
130  REGULAR_COUT << " Initial velocity assumed to be: " << gtsam::Vector3(0, 0, 0) << std::endl;
131  valuesEstimate.insert(gtsam::symbol_shorthand::V(propagatedStateKey_), gtsam::Vector3(0, 0, 0));
132  REGULAR_COUT << " Initial bias set to: " << *imuBiasPriorPtr_ << std::endl;
133  valuesEstimate.insert(gtsam::symbol_shorthand::B(propagatedStateKey_), *imuBiasPriorPtr_);
135  rtGraphValuesBufferPtr_->insert(valuesEstimate);
136  batchGraphValuesBufferPtr_->insert(valuesEstimate); // TODO: Check whether velocity can be removed from batch
137  writeValueKeysToKeyTimeStampMap_(valuesEstimate, timeStamp, priorKeyTimestampMapPtr);
138 
139  // Initialize graph -------------------------------------------------
140  rtFactorGraphBufferPtr_->resize(0);
141  batchFactorGraphBufferPtr_->resize(0);
142  rtFactorGraphBufferPtr_->emplace_shared<gtsam::PriorFactor<gtsam::Pose3>>(
143  gtsam::symbol_shorthand::X(propagatedStateKey_), T_W_I0,
144  priorPoseNoise); // POSE - PriorFactor format is (key,value,matrix) value
145  batchFactorGraphBufferPtr_->emplace_shared<gtsam::PriorFactor<gtsam::Pose3>>(
146  gtsam::symbol_shorthand::X(propagatedStateKey_), T_W_I0,
147  priorPoseNoise); // POSE - PriorFactor format is (key,value,matrix) value
148  // is same type as type of PriorFactor
149  rtFactorGraphBufferPtr_->emplace_shared<gtsam::PriorFactor<gtsam::Vector3>>(gtsam::symbol_shorthand::V(propagatedStateKey_),
150  gtsam::Vector3(0, 0, 0),
151  priorVelocityNoise); // VELOCITY
152  batchFactorGraphBufferPtr_->emplace_shared<gtsam::PriorFactor<gtsam::Vector3>>(gtsam::symbol_shorthand::V(propagatedStateKey_),
153  gtsam::Vector3(0, 0, 0),
154  priorVelocityNoise); // VELOCITY
155  rtFactorGraphBufferPtr_->emplace_shared<gtsam::PriorFactor<gtsam::imuBias::ConstantBias>>(
156  gtsam::symbol_shorthand::B(propagatedStateKey_), *imuBiasPriorPtr_,
157  priorBiasNoise); // BIAS
158  batchFactorGraphBufferPtr_->emplace_shared<gtsam::PriorFactor<gtsam::imuBias::ConstantBias>>(
159  gtsam::symbol_shorthand::B(propagatedStateKey_), *imuBiasPriorPtr_,
160  priorBiasNoise); // BIAS
161 
162  // Add to Time Key Buffer
163  timeToKeyBufferPtr_->addToBuffer(timeStamp, propagatedStateKey_);
164 
165  // Copy over
166  // Rt
167  newRtGraphFactors = *rtFactorGraphBufferPtr_;
168  newRtGraphValues = *rtGraphValuesBufferPtr_;
169  rtFactorGraphBufferPtr_->resize(0);
170  rtGraphValuesBufferPtr_->clear();
171  // Batch
172  newBatchGraphFactors = *batchFactorGraphBufferPtr_;
173  newBatchGraphValues = *batchGraphValuesBufferPtr_;
174  batchFactorGraphBufferPtr_->resize(0);
175  batchGraphValuesBufferPtr_->clear();
176  }
177 
179  // Rt Smoother
180  addFactorsToRtSmootherAndOptimize(newRtGraphFactors, newRtGraphValues, *priorKeyTimestampMapPtr, graphConfigPtr_, 0);
181  // Batch Smoother
182  addFactorsToBatchSmootherAndOptimize(newBatchGraphFactors, newBatchGraphValues, *priorKeyTimestampMapPtr, graphConfigPtr_);
183 
184  // Update Current State ---------------------------------------------------
185  const std::lock_guard<std::mutex> operateOnGraphDataLock(operateOnGraphDataMutex_);
186  optimizedGraphState_.updateNavStateAndBias(propagatedStateKey_, timeStamp, gtsam::NavState(T_W_I0, gtsam::Vector3(0, 0, 0)),
187  gtsam::Vector3(0, 0, 0), *imuBiasPriorPtr_);
188  O_imuPropagatedState_ = gtsam::NavState(T_O_I0, gtsam::Vector3(0, 0, 0));
189  W_imuPropagatedState_ = gtsam::NavState(T_W_I0, gtsam::Vector3(0, 0, 0));
190  T_W_O_ = (T_W_I0.inverse() * T_O_I0).matrix();
191  return true;
192 }

194 // IMU at the core --------------------------------------------------------------
195 void GraphManager::addImuFactorAndGetState(SafeIntegratedNavState& returnPreIntegratedNavState,
196  std::shared_ptr<SafeNavStateWithCovarianceAndBias>& newOptimizedNavStatePtr,
197  const std::shared_ptr<ImuBuffer>& imuBufferPtr, const double imuTimeK, bool createNewStateFlag) {
198  // Logging of latency
199  if (graphConfigPtr_->logLatencyAndUpdateDurationToMemoryFlag_) {
200  latencyStartTime_ = std::chrono::high_resolution_clock::now();
201  }
202 
203  // Looking up from IMU buffer --> acquire mutex (otherwise values for key might not be set)
204  const std::lock_guard<std::mutex> operateOnGraphDataLock(operateOnGraphDataMutex_);
205 
206  // Part 1 (ALWAYS): Propagate state and imu measurement to pre-integrator -----------------------
207  // 1.1 Get last two measurements from buffer to determine dt
208  TimeToImuMap imuMeas;
209  imuBufferPtr->getLastTwoMeasurements(imuMeas);
210  propagatedStateTime_ = imuTimeK;
211  currentAngularVelocity_ = imuMeas.rbegin()->second.angularVelocity;
212 
213  // 1.2 Update IMU Pre-integrator
214  updateImuIntegrators_(imuMeas);
215 
216  // 1.3 Predict propagated state via forward integration
217  if (graphConfigPtr_->usingBiasForPreIntegrationFlag_) {
218  O_imuPropagatedState_ =
219  imuBufferPtr->integrateNavStateFromTimestamp(imuMeas.begin()->first, imuMeas.rbegin()->first, O_imuPropagatedState_,
220  optimizedGraphState_.imuBias(), graphConfigPtr_->W_gravityVector_);
221  W_imuPropagatedState_ =
222  imuBufferPtr->integrateNavStateFromTimestamp(imuMeas.begin()->first, imuMeas.rbegin()->first, W_imuPropagatedState_,
223  optimizedGraphState_.imuBias(), graphConfigPtr_->W_gravityVector_);
224  } else {
225  O_imuPropagatedState_ =
226  imuBufferPtr->integrateNavStateFromTimestamp(imuMeas.begin()->first, imuMeas.rbegin()->first, O_imuPropagatedState_,
227  gtsam::imuBias::ConstantBias(), graphConfigPtr_->W_gravityVector_);
228  W_imuPropagatedState_ =
229  imuBufferPtr->integrateNavStateFromTimestamp(imuMeas.begin()->first, imuMeas.rbegin()->first, W_imuPropagatedState_,
230  gtsam::imuBias::ConstantBias(), graphConfigPtr_->W_gravityVector_);
231  }
232 
233  // 1.4 Fill in optimized state container
234  if (optimizedGraphState_.isOptimized()) {
235  newOptimizedNavStatePtr = std::make_shared<SafeNavStateWithCovarianceAndBias>(optimizedGraphState_);
236  } else {
237  newOptimizedNavStatePtr = nullptr;
238  }
239 
240  // 1.5 Return pre-integrated state
241  gtsam::NavState& T_O_Ik_nav = O_imuPropagatedState_; // Alias
242  // Assign poses and velocities
243  returnPreIntegratedNavState.update(T_W_O_, Eigen::Isometry3d(T_O_Ik_nav.pose().matrix()), T_O_Ik_nav.bodyVelocity(),
244  optimizedGraphState_.imuBias().correctGyroscope(imuMeas.rbegin()->second.angularVelocity), imuTimeK,
245  false);
246 
247  // Check whether new state can be created (in case fixed lag smoother has a short window)
248  if (createNewStateFlag && imuTimeK - lastOptimizedStateTime_ > graphConfigPtr_->realTimeSmootherLag_ - WORST_CASE_OPTIMIZATION_TIME &&
249  lastOptimizedStateTime_ > 0.0) {
250  createNewStateFlag = false;
251  REGULAR_COUT << RED_START
252  << " The current measurement would fall outside of the real-time smoother lag, hence skipping the creation of a new"
253  << "IMU measurement. Not creating new state." << COLOR_END << std::endl;
254  }
255 
256  // Part 2 (OPTIONAL): Create new state and add IMU factor to graph ------------------------------
257  if (createNewStateFlag) {
258  // Get new key
259  const gtsam::Key oldKey = propagatedStateKey_;
260  const gtsam::Key newKey = newPropagatedStateKey_();
261 
262  // Add to time key buffer
263  timeToKeyBufferPtr_->addToBuffer(imuTimeK, newKey);
264 
265  // Add IMU Factor to graph
266  gtsam::CombinedImuFactor imuFactor(gtsam::symbol_shorthand::X(oldKey), gtsam::symbol_shorthand::V(oldKey),
267  gtsam::symbol_shorthand::X(newKey), gtsam::symbol_shorthand::V(newKey),
268  gtsam::symbol_shorthand::B(oldKey), gtsam::symbol_shorthand::B(newKey), *imuStepPreintegratorPtr_);
269  addFactorToRtAndBatchGraph_<const gtsam::CombinedImuFactor*>(&imuFactor, imuTimeK, "imu");
270 
271  // Add IMU values
272  gtsam::Values valuesEstimate;
273  valuesEstimate.insert(gtsam::symbol_shorthand::X(newKey), W_imuPropagatedState_.pose());
274  valuesEstimate.insert(gtsam::symbol_shorthand::V(newKey), W_imuPropagatedState_.velocity());
275  valuesEstimate.insert(gtsam::symbol_shorthand::B(newKey), optimizedGraphState_.imuBias());
276  rtGraphValuesBufferPtr_->insert(valuesEstimate);
277  if (graphConfigPtr_->useAdditionalSlowBatchSmootherFlag_) {
278  batchGraphValuesBufferPtr_->insert(valuesEstimate);
279  }
280 
281  // Add timestamps
282  writeValueKeysToKeyTimeStampMap_(valuesEstimate, imuTimeK, rtGraphKeysTimestampsMapBufferPtr_);
283  if (graphConfigPtr_->useAdditionalSlowBatchSmootherFlag_) {
284  writeValueKeysToKeyTimeStampMap_(valuesEstimate, imuTimeK, batchGraphKeysTimestampsMapBufferPtr_);
285  }
286 
287  // After adding this factor we can again empty the step integrator
288  // Reset IMU Step Pre-integration
289  if (graphConfigPtr_->usingBiasForPreIntegrationFlag_) {
290  imuStepPreintegratorPtr_->resetIntegrationAndSetBias(optimizedGraphState_.imuBias());
291  } else {
292  imuStepPreintegratorPtr_->resetIntegrationAndSetBias(gtsam::imuBias::ConstantBias());
293  }
294  } // End of create new state
295 
296  // Potentially log real-time state to container
297  if (graphConfigPtr_->logRealTimeStateToMemoryFlag_) {
298  realTimeWorldPoseContainer_[imuTimeK] = W_imuPropagatedState_.pose();
299  realTimeOdomPoseContainer_[imuTimeK] = O_imuPropagatedState_.pose();
300  }
301 
302  // Logging of latency
303  if (graphConfigPtr_->logLatencyAndUpdateDurationToMemoryFlag_) {
304  latencyEndTime_ = std::chrono::high_resolution_clock::now();
305  // Duration in seconds
306  std::chrono::duration<double> latencyDuration = latencyEndTime_ - latencyStartTime_;
307  latencyContainer_[imuTimeK] = latencyDuration.count();
308  }
309 }

350 // Robust Aware Between factors ------------------------------------------------------------------------------------------------------
351 gtsam::Key GraphManager::addPoseBetweenFactor(const gtsam::Pose3& deltaPose, const Eigen::Matrix<double, 6, 1>& poseBetweenNoiseDensity,
352  const double timeKm1, const double timeK, const double rate,
353  const RobustNormEnum& robustNormEnum, const double robustNormConstant) {
354  // Find corresponding keys in graph
355  // Find corresponding keys in graph
356  const double maxLidarTimestampDistance = (1.0 / rate) + (2.0 * graphConfigPtr_->maxSearchDeviation_);
357  gtsam::Key closestKeyKm1, closestKeyK;
358  double keyTimeStampDistance{0.0};
359 
360  if (!findGraphKeys_(closestKeyKm1, closestKeyK, keyTimeStampDistance, maxLidarTimestampDistance, timeKm1, timeK, "pose between")) {
361  REGULAR_COUT << RED_START << " Current propagated key: " << propagatedStateKey_ << " , PoseBetween factor not added between keys "
362  << closestKeyKm1 << " and " << closestKeyK << COLOR_END << std::endl;
363  return closestKeyK;
364  }
365 
366  // Scale delta pose according to timeStampDistance
367  const double scale = keyTimeStampDistance / (timeK - timeKm1);
368  if (graphConfigPtr_->verboseLevel_ > 3) {
369  REGULAR_COUT << " Scaling factor in tangent space for pose between delta pose: " << scale << std::endl;
370  }
371  gtsam::Pose3 scaledDeltaPose = gtsam::Pose3::Expmap(scale * gtsam::Pose3::Logmap(deltaPose));
372 
373  // Create noise model
374  assert(poseBetweenNoiseDensity.size() == 6);
375  auto noise = gtsam::noiseModel::Diagonal::Sigmas((gtsam::Vector(poseBetweenNoiseDensity))); // rad,rad,rad,m,m,m
376  boost::shared_ptr<gtsam::noiseModel::Robust> robustErrorFunction;
377  // Pick Robust Error Function
378  switch (robustNormEnum) {
379  case RobustNormEnum::Huber:
380  robustErrorFunction = gtsam::noiseModel::Robust::Create(gtsam::noiseModel::mEstimator::Huber::Create(robustNormConstant), noise);
381  break;
382  case RobustNormEnum::Cauchy:
383  robustErrorFunction = gtsam::noiseModel::Robust::Create(gtsam::noiseModel::mEstimator::Cauchy::Create(robustNormConstant), noise);
384  break;
385  case RobustNormEnum::Tukey:
386  robustErrorFunction = gtsam::noiseModel::Robust::Create(gtsam::noiseModel::mEstimator::Tukey::Create(robustNormConstant), noise);
387  break;
388  }
389 
390  // Create pose between factor and add it
391  gtsam::BetweenFactor<gtsam::Pose3> poseBetweenFactor;
392  if (robustNormEnum == RobustNormEnum::None) {
393  poseBetweenFactor = gtsam::BetweenFactor<gtsam::Pose3>(gtsam::symbol_shorthand::X(closestKeyKm1),
394  gtsam::symbol_shorthand::X(closestKeyK), scaledDeltaPose, noise);
395  } else {
396  poseBetweenFactor = gtsam::BetweenFactor<gtsam::Pose3>(gtsam::symbol_shorthand::X(closestKeyKm1),
397  gtsam::symbol_shorthand::X(closestKeyK), scaledDeltaPose, robustErrorFunction);
398  }
399 
400  // Write to graph
401  addFactorSafelyToRtAndBatchGraph_<const gtsam::BetweenFactor<gtsam::Pose3>*>(&poseBetweenFactor, timeKm1);
402 
403  // Print summary
404  if (graphConfigPtr_->verboseLevel_ > 3) {
405  REGULAR_COUT << " Current propagated key: " << propagatedStateKey_ << ", " << YELLOW_START << " PoseBetween factor added between key "
406  << closestKeyKm1 << " and key " << closestKeyK << COLOR_END << std::endl;
407  }
408 
409  return closestKeyK;
410 }

437  std::optional<const gtsam::NavState> optionalNavState) {
438  // Pose Covariance in Tangent Space
439  std::chrono::high_resolution_clock::time_point start = std::chrono::high_resolution_clock::now();
440  gtsam::Matrix66 resultPoseCovarianceBodyFrame = graphPtr->calculateMarginalCovarianceMatrixAtKey(graphKey);
441  std::chrono::high_resolution_clock::time_point end = std::chrono::high_resolution_clock::now();
442  auto duration = std::chrono::duration_cast<std::chrono::microseconds>(end - start).count();
443  if (!optionalNavState.has_value() && duration > 400) {
444  std::cout << RED_START << "Time to calculate marginal covariance: " << duration << " microseconds, which is too much." << std::endl;
445  }
446  // Transform covariance from I_S_I in body frame, to W_S_W in world frame
447  // Get NavState
448  gtsam::NavState navState;
449  if (optionalNavState.has_value()) {
450  navState = optionalNavState.value();
451  } else {
452  gtsam::Pose3 resultPose = graphPtr->calculateEstimatedPose3(graphKey); // auto result = mainGraphPtr_->estimate();
453  gtsam::Point3 resultVelocity = gtsam::Point3::Identity(); // Assume Zero Velocity, but adjoint below is still fine to map to world
454  navState = gtsam::NavState(resultPose, resultVelocity);
455  }
456  // Compute adjoint matrix
457  gtsam::Matrix66 adjointMatrix = navState.pose().AdjointMap();
458  // Transform covariance
459  gtsam::Matrix66 resultPoseCovarianceWorldFrame = adjointMatrix * resultPoseCovarianceBodyFrame * adjointMatrix.transpose();
460  // Return
461  return resultPoseCovarianceWorldFrame;
462 }
