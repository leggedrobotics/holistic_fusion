38  // ii).A Holistically Optimize over Fixed Frames
39  void transformImuStateFromWorldToReferenceFrame(DynamicDictionaryContainer& gtsamDynamicExpressionKeys,
40  const gtsam::NavState& W_currentPropagatedState,
41  const bool centerMeasurementsAtKeyframePositionBeforeAlignmentFlag) final {
42  if (gmsfUnaryAbsoluteMeasurementPtr_->fixedFrameName() == gmsfUnaryAbsoluteMeasurementPtr_->worldFrameName()) {
43  // Do nothing as this is a world frame measurement
44  return;
45  }
46 
47  // Mutex because we are changing the dynamically allocated graphKeys
48  std::lock_guard<std::mutex> modifyGraphKeysLock(gtsamDynamicExpressionKeys.get<gtsam::Pose3>().mutex());
49 
50  // Run through steps needed for absolute measurements
51  // If it should be centered --> create keyframe for measurement
52  Eigen::Vector3d measurementOriginPosition = Eigen::Vector3d::Zero();
53  if (centerMeasurementsAtKeyframePositionBeforeAlignmentFlag) {
54  measurementOriginPosition = this->getMeasurementPosition();
55  }
56 
57  // Initial Guess
58  gtsam::Pose3 T_W_fixedFrame_initial = this->computeT_W_fixedFrame_initial(W_currentPropagatedState);
59 
60  // A. Search for the new graph key of T_fixedFrame_W -----------------------------------------------------
61  bool newGraphKeyAddedFlag = false;
62  const DynamicVariableType dynamicVariableType =
63  DynamicVariableType::RefFrame(measurementOriginPosition, gmsfUnaryAbsoluteMeasurementPtr_->timeK());
64  // Search for new graph key, if not found, add it
65  DynamicFactorGraphStateKey<gtsam::Pose3> graphKey = gtsamDynamicExpressionKeys.get<gtsam::Pose3>().getTransformationKey<'r'>(
66  newGraphKeyAddedFlag, gmsfUnaryAbsoluteMeasurementPtr_->worldFrameName(), gmsfUnaryAbsoluteMeasurementPtr_->fixedFrameName(),
67  gmsfUnaryAbsoluteMeasurementPtr_->timeK(), T_W_fixedFrame_initial, dynamicVariableType);
68  // Get age of keyframe
69  const double keyframeAge = graphKey.computeKeyframeAge(gmsfUnaryAbsoluteMeasurementPtr_->timeK());
70 
71  // B: Take care of the old keyframe -----------------------------------------------------
72  // If deactivated or too old, we should remove the old keyframe and create a new one (with or without random walk)
73  // In any case, we have to add relation between old and new keyframe
74  // Offline graph: we can always add the delta as all states are part of the optimization (random walk or deterministic keyframe
75  // displacement).
76  // Online graph: we can only add the delta if the old keyframe is still active (random walk or deterministic keyframe displacement. If
77  // it is not active, add prior from old keyframe to new keyframe (including the displacement).
78  bool introducedNewKeyframeDisplacement = false;
79  // Old GTSAM key --> needed for adding constraints between previous and new keyframe
80  const gtsam::Key oldGtsamKey = graphKey.key();
81  const Eigen::Vector3d oldKeyframePosition = graphKey.getReferenceFrameKeyframePosition();
82  bool oldVariableWasActive = graphKey.isVariableActive();
83  // If needed, have a copy of the old key to add back a prior to the online graph
84  std::unique_ptr<DynamicFactorGraphStateKey<gtsam::Pose3>> oldKeyPtr = nullptr;
85  if (!oldVariableWasActive) {
86  REGULAR_COUT << " Old keyframe is not active anymore. Hence keeping the belief and uncertainty of previous key." << std::endl;
87  oldKeyPtr = std::make_unique<DynamicFactorGraphStateKey<gtsam::Pose3>>(graphKey);
88  }
89 
90  // C: Check if we need to create a new keyframe -----------------------------------------------------
91  // Case 1: Old keyframe not active --> just reactivate the keyframe (independent of whether new keyframe was added or not)
92  if (!oldVariableWasActive) {
93  // Check whether no new keyframe was added
94  if (newGraphKeyAddedFlag) {
95  throw std::logic_error("GmsfUnaryExpressionAbsolut: Old keyframe was not active, but new keyframe was added.");
96  }
97  // Reactivate the keyframe (have to use reference instead of copy "graphKey" above)
98  gtsamDynamicExpressionKeys.get<gtsam::Pose3>()
99  .lv_T_frame1_frame2(gmsfUnaryAbsoluteMeasurementPtr_->worldFrameName(), gmsfUnaryAbsoluteMeasurementPtr_->fixedFrameName())
100  .activateVariable();
101  // Set number of steps optimized to zero
102  gtsamDynamicExpressionKeys.get<gtsam::Pose3>()
103  .lv_T_frame1_frame2(gmsfUnaryAbsoluteMeasurementPtr_->worldFrameName(), gmsfUnaryAbsoluteMeasurementPtr_->fixedFrameName())
104  .resetNumberStepsOptimized();
105  // Print out
106  REGULAR_COUT << GREEN_START << " Reactivated old keyframe " << gtsam::Symbol(graphKey.key())
107  << " as it was deactivated, but is needed again. Also set number of optimization steps to 0." << COLOR_END << std::endl;
108 
109  // If the old keyframe was not active, then it is not part of the optimization anymore (or has never been) --> prior to online graph
110  // Make sure that old key ptr is not null
111  if (oldKeyPtr == nullptr) {
112  throw std::logic_error("GmsfUnaryExpressionAbsolut: Old keyframe was not active, but old key ptr is null.");
113  }
114  // Container for belief of old keyframe
115  gtsam::Pose3 T_W_fixedFrameOld;
116  // Noise model container
117  boost::shared_ptr<gtsam::noiseModel::Base> noiseModelPtr;
118  // Case 1: Has been optimized before
119  if (oldKeyPtr->getNumberStepsOptimized() > 0) {
120  T_W_fixedFrameOld = oldKeyPtr->getTransformationAfterOptimization();
121  Eigen::Matrix<double, 6, 6> oldKeyframeCovariance = oldKeyPtr->getCovarianceAfterOptimization();
122  // Noise model from 6x6 covariance
123  noiseModelPtr = gtsam::noiseModel::Gaussian::Covariance(oldKeyframeCovariance);
124  // Print
125  REGULAR_COUT << " Old keyframe has been optimized before. Adding previous optimization outcome again to online graph." << std::endl;
126  }
127  // Case 2: Has not been optimized before
128  else {
129  T_W_fixedFrameOld = oldKeyPtr->getApproximateTransformationBeforeOptimization();
130  Eigen::Matrix<double, 6, 1> oldKeyframeCovariance = gmsfUnaryAbsoluteMeasurementPtr_->initialSe3AlignmentNoise();
131  // Noise model for prior
132  noiseModelPtr = gtsam::noiseModel::Diagonal::Sigmas(gmsfUnaryAbsoluteMeasurementPtr_->initialSe3AlignmentNoise());
133  // Print
134  REGULAR_COUT << " Old keyframe has not been optimized before. Adding very initial guess that should have been added last time."
135  << std::endl;
136  }
137  // Add information to online graph only (as has been marginalized out), offline still has it
138  // Prior belief
139  this->newOnlinePosePriorFactors_.emplace_back(oldGtsamKey, T_W_fixedFrameOld, noiseModelPtr);
140  // State value
141  this->newOnlineStateValues_.insert(oldGtsamKey, T_W_fixedFrameOld);
142  // Print
143  REGULAR_COUT << GREEN_START << " Added prior to old keyframe " << gtsam::Symbol(oldGtsamKey)
144  << " to the online graph, as it was not active anymore." << COLOR_END << std::endl;
145  }
146 
147  // Case 2: Keyframe is too old --> create a new keyframe to model the displacement
148  if (keyframeAge > createReferenceAlignmentKeyframeEveryNSeconds_) {
149  // Remove the old keyframe from memory
150  gtsamDynamicExpressionKeys.get<gtsam::Pose3>().removeTransform(gmsfUnaryAbsoluteMeasurementPtr_->worldFrameName(),
151  gmsfUnaryAbsoluteMeasurementPtr_->fixedFrameName(), graphKey);
152  // Create a new keyframe
153  graphKey = gtsamDynamicExpressionKeys.get<gtsam::Pose3>().getTransformationKey<'r'>(
154  newGraphKeyAddedFlag, gmsfUnaryAbsoluteMeasurementPtr_->worldFrameName(), gmsfUnaryAbsoluteMeasurementPtr_->fixedFrameName(),
155  gmsfUnaryAbsoluteMeasurementPtr_->timeK(), T_W_fixedFrame_initial, dynamicVariableType);
156  // Assert that new keyframe was added and that it is active
157  assert(newGraphKeyAddedFlag && graphKey.isVariableActive());
158  // Set flag that we introduced a new keyframe
159  introducedNewKeyframeDisplacement = true;
160  }
161 
162  // D: Shift the measurement to the robot position and recompute initial guess if we create keyframes -----------------------------------
163  // Has to be done here, as we did not know the keyframe position before
164  if (centerMeasurementsAtKeyframePositionBeforeAlignmentFlag) {
165  // Shift the measurement to the robot position
166  this->setMeasurementPosition(this->getMeasurementPosition() - graphKey.getReferenceFrameKeyframePosition());
167  // Recompute initial guess
168  T_W_fixedFrame_initial = this->computeT_W_fixedFrame_initial(W_currentPropagatedState);
169  // Update the initial guess
170  gtsamDynamicExpressionKeys.get<gtsam::Pose3>()
171  .lv_T_frame1_frame2(gmsfUnaryAbsoluteMeasurementPtr_->worldFrameName(), gmsfUnaryAbsoluteMeasurementPtr_->fixedFrameName())
172  .setApproximateTransformationBeforeOptimization(T_W_fixedFrame_initial);
173  }
174 
175  // E: Define expression for T_fixedFrame_W -----------------------------------------------------
176  gtsam::Pose3_ exp_T_W_fixedFrame(graphKey.key()); // T_fixedFrame_W
177  // Call main child function
178  transformStateToReferenceFrameMeasurement(exp_T_W_fixedFrame);
179 
180  // F: Initial values and priors if needed -----------------------------------------------------
181  if (newGraphKeyAddedFlag) {
182  REGULAR_COUT << " Initial Guess for T_W_" << gmsfUnaryAbsoluteMeasurementPtr_->fixedFrameName()
183  << ", RPY (deg): " << T_W_fixedFrame_initial.rotation().rpy().transpose() * (180.0 / M_PI)
184  << ", t (x, y, z): " << T_W_fixedFrame_initial.translation().transpose() << std::endl;
185  // Insert Values
186  this->newOnlineStateValues_.insert(graphKey.key(), T_W_fixedFrame_initial);
187  this->newOfflineStateValues_.insert(graphKey.key(), T_W_fixedFrame_initial);
188 
189  // Case 1: Entirely new keyframe has been added (not just a displacement) --> add prior to online graph
190  if (!introducedNewKeyframeDisplacement) {
191  // Insert Prior ONLY for online graph (offline is observable regardless)
192  this->newOnlinePosePriorFactors_.emplace_back(
193  graphKey.key(), T_W_fixedFrame_initial,
194  gtsam::noiseModel::Diagonal::Sigmas(gmsfUnaryAbsoluteMeasurementPtr_->initialSe3AlignmentNoise()));
195  REGULAR_COUT << GREEN_START << " Adding prior to new keyframe " << gtsam::Symbol(graphKey.key()) << " with noise "
196  << gmsfUnaryAbsoluteMeasurementPtr_->initialSe3AlignmentNoise().transpose()
197  << " to the online graph, as it is an entirely new keyframe." << COLOR_END << std::endl;
198  }
199  // Case 2: New keyframe has been added, but one existed before already --> add relative constraint from old to new keyframe
200  else {
201  // Add relative constraint from old keyframe to new keyframe
202  gtsam::Point3 relativeKeyframeTranslation(graphKey.getReferenceFrameKeyframePosition() - oldKeyframePosition);
203  // Transform between old and new keyframe
204  gtsam::Pose3 T_fixedFrameOld_fixedFrame(gtsam::Rot3::Identity(), relativeKeyframeTranslation);
205  // Define noise model --> either random walk or deterministic displacement
206 
207  // a): Random Walk between the two
208  if (gmsfUnaryAbsoluteMeasurementPtr_->modelAsRandomWalkFlag()) {
209  boost::shared_ptr<gtsam::noiseModel::Diagonal> noiseModelPtr =
210  gtsam::noiseModel::Diagonal::Sigmas(gmsfUnaryAbsoluteMeasurementPtr_->se3AlignmentRandomWalk());
211  this->newOnlineAndOfflinePoseBetweenFactors_.emplace_back(oldGtsamKey, graphKey.key(), T_fixedFrameOld_fixedFrame, noiseModelPtr);
212 
213  // Add prior from old keyframe to new keyframe
214  REGULAR_COUT << GREEN_START << " Adding relative RANDOM WALK constraint from key " << gtsam::Symbol(oldGtsamKey) << " to key "
215  << gtsam::Symbol(graphKey.key()) << ": " << T_fixedFrameOld_fixedFrame << COLOR_END << std::endl;
216  }
217  // b): Deterministic displacement modelled as equality constraint
218  else {
219  boost::shared_ptr<gtsam::noiseModel::Constrained> noiseModelPtr =
220  gtsam::noiseModel::Constrained::MixedSigmas(gmsfUnaryAbsoluteMeasurementPtr_->se3AlignmentRandomWalk());
221  this->newOnlineAndOfflinePoseBetweenFactors_.emplace_back(oldGtsamKey, graphKey.key(), T_fixedFrameOld_fixedFrame, noiseModelPtr);
222 
223  // Add prior from old keyframe to new keyframe
224  REGULAR_COUT << GREEN_START << " Adding relative DETERMINISTIC constraint from key " << gtsam::Symbol(oldGtsamKey) << " to key "
225  << gtsam::Symbol(graphKey.key()) << ": " << T_fixedFrameOld_fixedFrame << COLOR_END << std::endl;
226  }
227  }
228  }
229  }
