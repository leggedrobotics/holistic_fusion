108 void SmbEstimator::imuCallback(const sensor_msgs::Imu::ConstPtr& imuPtr) {
109  // Check whether any of the measurements is available, otherwise do pure imu integration
110  if (graph_msf::GraphMsf::areRollAndPitchInited() && !graph_msf::GraphMsf::areYawAndPositionInited() && !useLioOdometryFlag_ &&
111  !useWheelOdometryBetweenFlag_ && !useWheelLinearVelocitiesFlag_ && !useVioOdometryFlag_) {
112  // Initialization
113  REGULAR_COUT << RED_START << " IMU callback is setting global yaw and position, as no other odometry is available. Initializing..."
114  << COLOR_END << std::endl;
115  // Create dummy measurement for initialization
116  graph_msf::UnaryMeasurementXD<Eigen::Isometry3d, 6> unary6DMeasurement(
117  "IMU_init_6D", int(graphConfigPtr_->imuRate_), staticTransformsPtr_->getImuFrame(),
118  staticTransformsPtr_->getImuFrame() + sensorFrameCorrectedNameId, graph_msf::RobustNorm::None(), imuPtr->header.stamp.toSec(), 1.0,
119  Eigen::Isometry3d::Identity(), Eigen::MatrixXd::Identity(6, 1));
120  // Initialize
121  graph_msf::GraphMsf::initYawAndPosition(unary6DMeasurement);
122  REGULAR_COUT << RED_START << " ...initialized yaw and position to identity." << COLOR_END << std::endl;
123  // Pretend that we received first measurement --> In order to allow for optimization
124  graph_msf::GraphMsf::pretendFirstMeasurementReceived();
125  }
126 
127  // Super class
128  graph_msf::GraphMsfRos::imuCallback(imuPtr);

131 void SmbEstimator::lidarOdometryCallback_(const nav_msgs::Odometry::ConstPtr& odomLidarPtr) {
132  // Static members
133  static int lidarOdometryCallbackCounter__ = -1;
134 
135  // Counter
136  ++lidarOdometryCallbackCounter__;
137 
138  Eigen::Isometry3d lio_T_M_Lk;
139  graph_msf::odomMsgToEigen(*odomLidarPtr, lio_T_M_Lk.matrix());
140 
141  // Transform to IMU frame
142  double lidarOdometryTimeK = odomLidarPtr->header.stamp.toSec();
143 
144  // Frame Name
145  const std::string& lioOdometryFrame = dynamic_cast<SmbStaticTransforms*>(staticTransformsPtr_.get())->getLioOdometryFrame(); // alias
146 
147  // Measurement
148  graph_msf::UnaryMeasurementXDAbsolute<Eigen::Isometry3d, 6> unary6DMeasurement(
149  "Lidar_unary_6D", int(lioOdometryRate_), lioOdometryFrame, lioOdometryFrame + sensorFrameCorrectedNameId,
150  graph_msf::RobustNorm::None(), lidarOdometryTimeK, 1.0, lio_T_M_Lk, lioPoseUnaryNoise_, odomLidarPtr->header.frame_id,
151  staticTransformsPtr_->getWorldFrame(), initialSe3AlignmentNoise_, lioSe3AlignmentRandomWalk_);
152 
153  // Add measurement or initialize
154  if (lidarOdometryCallbackCounter__ <= 2) {
155  return;
156  } else if (areYawAndPositionInited()) { // Already initialized --> unary factor
157  this->addUnaryPose3AbsoluteMeasurement(unary6DMeasurement);
158  } else { // Initializing
159  REGULAR_COUT << GREEN_START << " LiDAR odometry callback is setting global yaw, as it was not set so far." << COLOR_END << std::endl;
160  this->initYawAndPosition(unary6DMeasurement);
161  }
162 
163  // Visualization ----------------------------
164  // Add to path message
165  addToPathMsg(measLio_mapImuPathPtr_, odomLidarPtr->header.frame_id, odomLidarPtr->header.stamp,
166  (lio_T_M_Lk * staticTransformsPtr_
167  ->rv_T_frame1_frame2(dynamic_cast<SmbStaticTransforms*>(staticTransformsPtr_.get())->getLioOdometryFrame(),
168  staticTransformsPtr_->getImuFrame())
169  .matrix())
170  .block<3, 1>(0, 3),
171  graphConfigPtr_->imuBufferLength_ * 4);
172 
173  // Publish Path
174  pubMeasMapLioPath_.publish(measLio_mapImuPathPtr_);
175 }

177 void SmbEstimator::wheelOdometryPoseCallback_(const nav_msgs::Odometry::ConstPtr& wheelOdometryKPtr) {
178  if (!areRollAndPitchInited()) {
179  return;
180  }
181 
182  // Counter
183  ++wheelOdometryCallbackCounter_;
184 
185  // Eigen Type
186  Eigen::Isometry3d T_O_Bw_k;
187  graph_msf::odomMsgToEigen(*wheelOdometryKPtr, T_O_Bw_k.matrix());
188  double wheelOdometryTimeK = wheelOdometryKPtr->header.stamp.toSec();
189 
190  // At start
191  if (wheelOdometryCallbackCounter_ == 0) {
192  T_O_Bw_km1_ = T_O_Bw_k;
193  wheelOdometryTimeKm1_ = wheelOdometryTimeK;
194  return;
195  }
196 
197  // Frame name
198  const std::string& wheelOdometryFrame =
199  dynamic_cast<SmbStaticTransforms*>(staticTransformsPtr_.get())->getWheelOdometryBetweenFrame(); // alias
200 
201  // State Machine
202  if (!areYawAndPositionInited()) {
203  // If no LIO, then initialize the graph here
204  if (!useLioOdometryFlag_) {
205  REGULAR_COUT << GREEN_START << " Wheel odometry callback is setting global yaw and position, as lio is all set to false." << COLOR_END
206  << std::endl;
207  graph_msf::UnaryMeasurementXD<Eigen::Isometry3d, 6> unary6DMeasurement(
208  "Lidar_unary_6D", int(wheelOdometryBetweenRate_), wheelOdometryFrame, wheelOdometryFrame + sensorFrameCorrectedNameId,
209  graph_msf::RobustNorm::None(), wheelOdometryTimeK, 1.0, Eigen::Isometry3d::Identity(), Eigen::MatrixXd::Identity(6, 1));
210  graph_msf::GraphMsf::initYawAndPosition(unary6DMeasurement);
211  REGULAR_COUT << " Initialized yaw and position to identity in the wheel odometry callback, as lio and vio are all set to false."
212  << std::endl;
213  }
214  } else {
215  // Only add every 5th measurement
216  int measurementRate = static_cast<int>(wheelOdometryBetweenRate_ / 5);
217  // Check
218  if (wheelOdometryCallbackCounter_ % 5 == 0 && wheelOdometryCallbackCounter_ > 0) {
219  // Compute Delta
220  Eigen::Isometry3d T_Bkm1_Bk = T_O_Bw_km1_.inverse() * T_O_Bw_k;
221  // Create measurement
222  graph_msf::BinaryMeasurementXD<Eigen::Isometry3d, 6> delta6DMeasurement(
223  "Wheel_odometry_6D", measurementRate, wheelOdometryFrame, wheelOdometryFrame + sensorFrameCorrectedNameId,
224  graph_msf::RobustNorm::Tukey(1.0), wheelOdometryTimeKm1_, wheelOdometryTimeK, T_Bkm1_Bk, wheelPoseBetweenNoise_);
225  // Add to graph
226  this->addBinaryPose3Measurement(delta6DMeasurement);
227 
228  // Prepare for next iteration
229  T_O_Bw_km1_ = T_O_Bw_k;
230  wheelOdometryTimeKm1_ = wheelOdometryTimeK;
231  }
232  }

235 void SmbEstimator::wheelLinearVelocitiesCallback_(const std_msgs::Float64MultiArray::ConstPtr& wheelsSpeedsPtr) {
236  if (!areRollAndPitchInited()) {
237  return;
238  }
239 
240  // Get the 3 values
241  const double timeK = wheelsSpeedsPtr->data[0];
242  const double leftWheelSpeedRps = wheelsSpeedsPtr->data[1];
243  const double rightWheelSpeedRps = wheelsSpeedsPtr->data[2];
244  const double leftWheelSpeedMs = leftWheelSpeedRps * wheelRadiusMeter_;
245  const double rightWheelSpeedMs = rightWheelSpeedRps * wheelRadiusMeter_;
246 
247  // Print
248  if (graphConfigPtr_->verboseLevel_ > 1) {
249  REGULAR_COUT << " Wheel linear velocities callback: " << std::setprecision(14) << timeK << " " << leftWheelSpeedMs << " [m/s] "
250  << rightWheelSpeedMs << " [m/s]" << std::endl;
251  }
252 
253  // Frame name
254  const std::string& wheelLinearVelocityLeftFrame =
255  dynamic_cast<SmbStaticTransforms*>(staticTransformsPtr_.get())->getWheelLinearVelocityLeftFrame(); // alias
256  const std::string& wheelLinearVelocityRightFrame =
257  dynamic_cast<SmbStaticTransforms*>(staticTransformsPtr_.get())->getWheelLinearVelocityRightFrame(); // alias
258 
259  // State Machine
260  if (!areYawAndPositionInited()) {
261  // If no LIO, then initialize the graph here
262  if (!useLioOdometryFlag_ && !useWheelOdometryBetweenFlag_) {
263  REGULAR_COUT << GREEN_START << " Wheel linear velocities callback is setting global yaw and position, as lio is all set to false."
264  << COLOR_END << std::endl;
265  graph_msf::UnaryMeasurementXD<Eigen::Isometry3d, 6> unary6DMeasurement(
266  "Lidar_unary_6D", int(wheelLinearVelocitiesRate_), wheelLinearVelocityLeftFrame,
267  wheelLinearVelocityLeftFrame + sensorFrameCorrectedNameId, graph_msf::RobustNorm::None(), timeK, 1.0,
268  Eigen::Isometry3d::Identity(), Eigen::MatrixXd::Identity(6, 1));
269  graph_msf::GraphMsf::initYawAndPosition(unary6DMeasurement);
270  REGULAR_COUT
271  << " Initialized yaw and position to identity in the wheel linear velocities callback, as lio and vio are all set to false."
272  << std::endl;
273  }
274  } else {
275  // Left Wheel
276  graph_msf::UnaryMeasurementXD<Eigen::Vector3d, 3> leftWheelLinearVelocityMeasurement(
277  "Wheel_linear_velocity_left", int(wheelLinearVelocitiesRate_), wheelLinearVelocityLeftFrame,
278  wheelLinearVelocityLeftFrame + sensorFrameCorrectedNameId, graph_msf::RobustNorm::Tukey(1.0), timeK, 1.0,
279  Eigen::Vector3d(leftWheelSpeedMs, 0.0, 0.0), wheelLinearVelocitiesNoise_);
280  this->addUnaryVelocity3LocalMeasurement(leftWheelLinearVelocityMeasurement);
281 
282  // Right Wheel
283  graph_msf::UnaryMeasurementXD<Eigen::Vector3d, 3> rightWheelLinearVelocityMeasurement(
284  "Wheel_linear_velocity_right", int(wheelLinearVelocitiesRate_), wheelLinearVelocityRightFrame,
285  wheelLinearVelocityRightFrame + sensorFrameCorrectedNameId, graph_msf::RobustNorm::Tukey(1.0), timeK, 1.0,
286  Eigen::Vector3d(rightWheelSpeedMs, 0.0, 0.0), wheelLinearVelocitiesNoise_);
287  this->addUnaryVelocity3LocalMeasurement(rightWheelLinearVelocityMeasurement);
288  }
289 }
