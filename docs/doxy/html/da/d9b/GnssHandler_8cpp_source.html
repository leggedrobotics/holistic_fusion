94 // Heading is defined as the orthogonal vector pointing from gnssPos2 to gnssPos1, projected to x,y-plane
95 Eigen::Vector3d GnssHandler::computeHeading_(const Eigen::Vector3d& gnssPos1, const Eigen::Vector3d& gnssPos2) {
96  // Compute connecting unity vector
97  Eigen::Vector3d W_t_GnssR_GnssL = (gnssPos1 - gnssPos2).normalized();
98 
99  // Compute forward pointing vector
100  Eigen::Vector3d zUnityVector = Eigen::Vector3d::UnitZ();
101  Eigen::Vector3d W_t_heading = W_t_GnssR_GnssL.cross(zUnityVector).normalized();
102 
103  return W_t_heading;
104 }
