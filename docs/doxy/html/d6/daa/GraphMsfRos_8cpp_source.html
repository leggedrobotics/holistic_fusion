168 void GraphMsfRos::addToOdometryMsg(const nav_msgs::OdometryPtr& msgPtr, const std::string& fixedFrame, const std::string& movingFrame,
169  const ros::Time& stamp, const Eigen::Isometry3d& T, const Eigen::Vector3d& F_v_W_F,
170  const Eigen::Vector3d& F_w_W_F, const Eigen::Matrix<double, 6, 6>& poseCovariance,
171  const Eigen::Matrix<double, 6, 6>& twistCovariance) {
172  msgPtr->header.frame_id = fixedFrame;
173  msgPtr->child_frame_id = movingFrame;
174  msgPtr->header.stamp = stamp;
175  tf::poseTFToMsg(isometry3ToTf(T), msgPtr->pose.pose);
176  msgPtr->twist.twist.linear.x = F_v_W_F(0);
177  msgPtr->twist.twist.linear.y = F_v_W_F(1);
178  msgPtr->twist.twist.linear.z = F_v_W_F(2);
179  msgPtr->twist.twist.angular.x = F_w_W_F(0);
180  msgPtr->twist.twist.angular.y = F_w_W_F(1);
181  msgPtr->twist.twist.angular.z = F_w_W_F(2);
182 
183  for (int i = 0; i < 6; i++) {
184  for (int j = 0; j < 6; j++) {
185  msgPtr->pose.covariance[6 * i + j] = poseCovariance(i, j);
186  msgPtr->twist.covariance[6 * i + j] = twistCovariance(i, j);
187  }
188  }
189 }

320  // Convert to Eigen
321  Eigen::Vector3d linearAcc(imuMsgPtr->linear_acceleration.x, imuMsgPtr->linear_acceleration.y, imuMsgPtr->linear_acceleration.z);
322  Eigen::Vector3d angularVel(imuMsgPtr->angular_velocity.x, imuMsgPtr->angular_velocity.y, imuMsgPtr->angular_velocity.z);
323  Eigen::Matrix<double, 6, 1> addedImuMeasurements; // accel, gyro
324 
325  // Create pointer for carrying state
326  std::shared_ptr<SafeIntegratedNavState> preIntegratedNavStatePtr = nullptr;
327  std::shared_ptr<SafeNavStateWithCovarianceAndBias> optimizedStateWithCovarianceAndBiasPtr = nullptr;
328 
329  // Add measurement and get state

331  optimizedStateWithCovarianceAndBiasPtr, addedImuMeasurements)) {
332  // Encountered Delay

336  }
337 
338  // Publish Odometry
339  this->publishState(preIntegratedNavStatePtr, optimizedStateWithCovarianceAndBiasPtr);
340 
341  // Publish Filtered Imu Measurements
342  // this->publishAddedImuMeas_(addedImuMeasurements, imuMsgPtr->header.stamp);
343  } else if (GraphMsf::isGraphInited()) {
344  REGULAR_COUT << RED_START << " Could not add IMU measurement." << COLOR_END << std::endl;
345  }

350 void GraphMsfRos::publishState(
351  const std::shared_ptr<graph_msf::SafeIntegratedNavState>& integratedNavStatePtr,
352  const std::shared_ptr<graph_msf::SafeNavStateWithCovarianceAndBias>& optimizedStateWithCovarianceAndBiasPtr) {
353  // Covariances
354  Eigen::Matrix<double, 6, 6> poseCovarianceRos, twistCovarianceRos;
355  extractCovariancesFromOptimizedState(poseCovarianceRos, twistCovarianceRos, optimizedStateWithCovarianceAndBiasPtr);
356 
357  // Variances (only digonal elements
358  Eigen::Vector3d positionVarianceRos = poseCovarianceRos.block<3, 3>(0, 0).diagonal();
359  Eigen::Vector3d orientationVarianceRos = poseCovarianceRos.block<3, 3>(3, 3).diagonal();
360 
361  // Publish non-time critical data in a separate thread
362  std::thread publishNonTimeCriticalDataThread(&GraphMsfRos::publishNonTimeCriticalData, this, poseCovarianceRos, twistCovarianceRos,
363  positionVarianceRos, orientationVarianceRos, integratedNavStatePtr,
364  optimizedStateWithCovarianceAndBiasPtr);
365  publishNonTimeCriticalDataThread.detach();
366 }

368 // Copy the arguments in order to be thread safe
369 void GraphMsfRos::publishNonTimeCriticalData(
370  const Eigen::Matrix<double, 6, 6> poseCovarianceRos, const Eigen::Matrix<double, 6, 6> twistCovarianceRos,
371  const Eigen::Vector3d positionVarianceRos, const Eigen::Vector3d orientationVarianceRos,
372  const std::shared_ptr<const graph_msf::SafeIntegratedNavState> integratedNavStatePtr,
373  const std::shared_ptr<const graph_msf::SafeNavStateWithCovarianceAndBias> optimizedStateWithCovarianceAndBiasPtr) {
374  // Mutex for not overloading ROS
375  std::lock_guard<std::mutex> lock(rosPublisherMutex_);
376 
377  // Time
378  const double& timeK = integratedNavStatePtr->getTimeK(); // Alias
379 
380  // Odometry messages
381  publishImuOdoms(integratedNavStatePtr, poseCovarianceRos, twistCovarianceRos);
382 
383  // Publish to TF
384  // B_O
385  Eigen::Isometry3d T_B_Ok =
386  staticTransformsPtr_->rv_T_frame1_frame2(staticTransformsPtr_->getBaseLinkFrame(), staticTransformsPtr_->getImuFrame()) *
387  integratedNavStatePtr->getT_O_Ik_gravityAligned().inverse();
388  publishTfTreeTransform(staticTransformsPtr_->getBaseLinkFrame(), staticTransformsPtr_->getOdomFrame(), timeK, T_B_Ok);
389  // O_W
390  Eigen::Isometry3d T_O_W = integratedNavStatePtr->getT_W_O().inverse();
391  publishTfTreeTransform(staticTransformsPtr_->getOdomFrame(), staticTransformsPtr_->getWorldFrame(), timeK, T_O_W);
392 
393  // Publish Variances
394  publishDiagVarianceVectors(positionVarianceRos, orientationVarianceRos, timeK);
395 
396  // Publish Velocity Markers
397  publishVelocityMarkers(integratedNavStatePtr);
398 
399  // Publish paths
400  publishImuPaths(integratedNavStatePtr);
401 
402  // Optimized estimate ----------------------
403  publishOptimizedStateAndBias(optimizedStateWithCovarianceAndBiasPtr, poseCovarianceRos, twistCovarianceRos);
404 }

406 void GraphMsfRos::publishOptimizedStateAndBias(
407  const std::shared_ptr<const graph_msf::SafeNavStateWithCovarianceAndBias> optimizedStateWithCovarianceAndBiasPtr,
408  const Eigen::Matrix<double, 6, 6>& poseCovarianceRos, const Eigen::Matrix<double, 6, 6>& twistCovarianceRos) {
409  if (optimizedStateWithCovarianceAndBiasPtr != nullptr &&
410  optimizedStateWithCovarianceAndBiasPtr->getTimeK() - lastOptimizedStateTimestamp_ > 1e-03) {
411  // Time of this optimized state
412  lastOptimizedStateTimestamp_ = optimizedStateWithCovarianceAndBiasPtr->getTimeK();
413 
414  // Odometry messages
415  // world->imu
416  if (pubOptWorldImu_.getNumSubscribers() > 0) {
417  addToOdometryMsg(optWorldImuMsgPtr_, staticTransformsPtr_->getWorldFrame(), staticTransformsPtr_->getImuFrame(),
418  ros::Time(optimizedStateWithCovarianceAndBiasPtr->getTimeK()), optimizedStateWithCovarianceAndBiasPtr->getT_W_Ik(),
419  optimizedStateWithCovarianceAndBiasPtr->getI_v_W_I(), optimizedStateWithCovarianceAndBiasPtr->getI_w_W_I(),
420  poseCovarianceRos, twistCovarianceRos);
421  pubOptWorldImu_.publish(optWorldImuMsgPtr_);
422  }
423 
424  // Path
425  // world->imu
426  addToPathMsg(optWorldImuPathPtr_, staticTransformsPtr_->getWorldFrame(), ros::Time(optimizedStateWithCovarianceAndBiasPtr->getTimeK()),
427  optimizedStateWithCovarianceAndBiasPtr->getT_W_Ik().translation(), graphConfigPtr_->imuBufferLength_ * 20);
428  if (pubOptWorldImuPath_.getNumSubscribers() > 0) {
429  pubOptWorldImuPath_.publish(optWorldImuPathPtr_);
430  }
431 
432  // Biases
433  // Publish accel bias
434  if (pubAccelBias_.getNumSubscribers() > 0) {
435  Eigen::Vector3d accelBias = optimizedStateWithCovarianceAndBiasPtr->getAccelerometerBias();
436  accelBiasMsgPtr_->header.stamp = ros::Time(optimizedStateWithCovarianceAndBiasPtr->getTimeK());
437  accelBiasMsgPtr_->header.frame_id = staticTransformsPtr_->getImuFrame();
438  accelBiasMsgPtr_->vector.x = accelBias(0);
439  accelBiasMsgPtr_->vector.y = accelBias(1);
440  accelBiasMsgPtr_->vector.z = accelBias(2);
441  pubAccelBias_.publish(accelBiasMsgPtr_);
442  }
443  // Publish gyro bias
444  if (pubGyroBias_.getNumSubscribers() > 0) {
445  Eigen::Vector3d gyroBias = optimizedStateWithCovarianceAndBiasPtr->getGyroscopeBias();
446  gyroBiasMsgPtr_->header.stamp = ros::Time(optimizedStateWithCovarianceAndBiasPtr->getTimeK());
447  gyroBiasMsgPtr_->header.frame_id = staticTransformsPtr_->getImuFrame();
448  gyroBiasMsgPtr_->vector.x = gyroBias(0);
449  gyroBiasMsgPtr_->vector.y = gyroBias(1);
450  gyroBiasMsgPtr_->vector.z = gyroBias(2);
451  pubGyroBias_.publish(gyroBiasMsgPtr_);
452  }
453 
454  // Reference Frame TFs in Optimized State
455  for (const auto& framePairTransformMapIterator :
456  optimizedStateWithCovarianceAndBiasPtr->getReferenceFrameTransforms().getTransformsMap()) {
457  // Case 1: If includes world frame --> everything child of world --------------------------------
458  if (framePairTransformMapIterator.first.first == staticTransformsPtr_->getWorldFrame() ||
459  framePairTransformMapIterator.first.second == staticTransformsPtr_->getWorldFrame()) {
460  // A. Get transform
461  Eigen::Isometry3d T_W_M;
462  std::string mapFrameName;
463  const std::string& worldFrameName = staticTransformsPtr_->getWorldFrame();
464 
465  // If world is second, then map is second
466  if (framePairTransformMapIterator.first.second == worldFrameName) {
467  T_W_M = framePairTransformMapIterator.second.inverse();
468  mapFrameName = framePairTransformMapIterator.first.first;
469  }
470  // If world is first, then map is second, this is the case for holistic alignment
471  else {

473  mapFrameName = framePairTransformMapIterator.first.second;

477  geometry_msgs::PoseWithCovarianceStampedPtr poseWithCovarianceStampedMsgPtr =

482  optimizedStateWithCovarianceAndBiasPtr->getReferenceFrameTransformsCovariance().rv_T_frame1_frame2(
483  framePairTransformMapIterator.first.first, framePairTransformMapIterator.first.second));

489  }

495 
496  // Print for debug

498  std::cout << "Transformation from " << mapFrameName << " to " << worldFrameName << std::endl;
499  std::cout << "Uncertainty: " << std::endl

501  framePairTransformMapIterator.first.first, framePairTransformMapIterator.first.second)
502  << std::endl;
503  }
504

506  publishTfTreeTransform(worldFrameName, mapFrameName + referenceFrameAlignedNameId,
507  optimizedStateWithCovarianceAndBiasPtr->getTimeK(), T_W_M);
508  }
509  // Case 2: Other transformation (e.g. calibration) --> does usually not include world frame ----------------
510  else {
511  const std::string& sensorFrameName = framePairTransformMapIterator.first.first;
512  const std::string& sensorFrameNameCorrected = framePairTransformMapIterator.first.second;
513  const Eigen::Isometry3d& T_sensor_sensorCorrected = framePairTransformMapIterator.second;
514

516  std::string transformTopic = "/graph_msf/transform_" + sensorFrameName + "_to_" + sensorFrameNameCorrected;
517  geometry_msgs::PoseWithCovarianceStampedPtr poseWithCovarianceStampedMsgPtr =
518  boost::make_shared<geometry_msgs::PoseWithCovarianceStamped>();
519  addToPoseWithCovarianceStampedMsg(
520  poseWithCovarianceStampedMsgPtr, sensorFrameName, ros::Time(optimizedStateWithCovarianceAndBiasPtr->getTimeK()),
521  T_sensor_sensorCorrected,
522  optimizedStateWithCovarianceAndBiasPtr->getReferenceFrameTransformsCovariance().rv_T_frame1_frame2(sensorFrameName,
523  sensorFrameNameCorrected));
524  // Check whether publisher already exists
525  if (pubPoseStampedByTopicMap_.find(transformTopic) == pubPoseStampedByTopicMap_.end()) {
526  pubPoseStampedByTopicMap_[transformTopic] = privateNode.advertise<geometry_msgs::PoseWithCovarianceStamped>(transformTopic, 1);
527  REGULAR_COUT << GREEN_START << " Initialized publisher for " << transformTopic << COLOR_END << std::endl;
528  }
529  if (pubPoseStampedByTopicMap_[transformTopic].getNumSubscribers() > 0) {
530  pubPoseStampedByTopicMap_[transformTopic].publish(poseWithCovarianceStampedMsgPtr);
531  }
532 
533  // B. Publish TF Tree --> as children of sensor frame
534  publishTfTreeTransform(sensorFrameName, sensorFrameNameCorrected, optimizedStateWithCovarianceAndBiasPtr->getTimeK(),
535  T_sensor_sensorCorrected);
536  }
537  }
538 
539  // Landmarks
540  for (const auto& landmarkTransformMapIterator : optimizedStateWithCovarianceAndBiasPtr->getLandmarkTransforms().getTransformsMap()) {
541  // A. Get transform
542  Eigen::Isometry3d T_W_L = landmarkTransformMapIterator.second;

544  std::string landmarkFrameName = landmarkTransformMapIterator.first.second;
