16 void Gnss::calculateConversionParameters() {
17  // calculate earth radii
18  const double temp = 1.0 / (1.0 - excentricity2_ * sin(referenceLatitude_ * M_PI / 180.0) * sin(referenceLatitude_ * M_PI / 180.0));
19  const double prime_vertical_radius = equatorialRadius_ * sqrt(temp);
20  earthRadiusN_ = prime_vertical_radius * (1 - excentricity2_) * temp;
21  earthRadiusE_ = prime_vertical_radius * cos(referenceLatitude_ * M_PI / 180.0);
22 }

45 Eigen::Vector3d Gnss::gnssToCartesian(const double& latitudeInDegrees, const double& longitudeInDegrees, const double& altitude) {
46  const double cn = cos(referenceHeading_);
47  const double sn = sin(referenceHeading_);
48  const double kn = 180.0 / earthRadiusN_ / M_PI;
49  const double ke = 180.0 / earthRadiusE_ / M_PI;
50  const double lat_tmp = (latitudeInDegrees - referenceLatitude_) / kn;
51  const double lon_tmp = (longitudeInDegrees - referenceLongitude_) / ke;
52 
53  Eigen::Vector3d position;
54  position(0) = cn * lat_tmp + sn * lon_tmp;
55  position(1) = sn * lat_tmp - cn * lon_tmp;
56  position(2) = altitude - referenceAltitude_;
57 
58  return position;
59 }

61 Eigen::Vector3d Gnss::cartesianToGps(const Eigen::Matrix<double, 3, 1> position) const {
62  Eigen::Vector3d gpsCoordinates;
63  gpsCoordinates(0) =
64  referenceLatitude_ + (cos(referenceHeading_) * position(0) + sin(referenceHeading_) * position(1)) / earthRadiusN_ * 180.0 / M_PI;
65  gpsCoordinates(1) =
66  referenceLongitude_ - (-sin(referenceHeading_) * position(0) + cos(referenceHeading_) * position(1)) / earthRadiusE_ * 180.0 / M_PI;
67  gpsCoordinates(2) = referenceAltitude_ + position(2);
68 
69  return gpsCoordinates;
70 }

72 Eigen::Vector3d Gnss::besselEllipsoidToMercator(const double& latitudeInRad, const double& longitudeInRad, const double& altitude) {
73  // Ellipsoid to Sphere (Gauss)
74  const double S = alpha_ * log(tan(M_PI_4 + latitudeInRad / 2.0)) -
75  alpha_ * sqrt(e2_) / 2.0 * log((1 + sqrt(e2_) * sin(latitudeInRad)) / (1 - sqrt(e2_) * sin(latitudeInRad))) + k_;
76  const double b = 2 * (atan(exp(S)) - M_PI_4);
77  const double l = alpha_ * (longitudeInRad - lambda0_);
78 
79  // Equator to Pseudoequator (Rotation)
80  const double lHat = atan(sin(l) / (sin(b0_) * tan(b) + cos(b0_) * cos(l)));
81  const double bHat = asin(cos(b0_) * sin(b) - sin(b0_) * cos(b) * cos(l));
82 
83  // Sphere to Plane (Mercator)
84  const double Y = r_ * lHat;
85  const double X = r_ / 2.0 * log((1 + sin(bHat)) / (1 - sin(bHat)));
86 
87  return Eigen::Vector3d(Y, X, altitude) - xyzOffset_; // Yes, this is correct. A point in swiss coordinates is denoted as (y,x). See
88  // https://en.wikipedia.org/wiki/Swiss_coordinate_system
89 }

133 double Gnss::decToSexAngle(double dec) {
134  int degree = std::floor(dec);
135  int minute = std::floor((dec - degree) * 60.0);
136  double second = (((dec - degree) * 60.0) - minute) * 60.0;
137  return degree + (static_cast<double>(minute) / 100.0) + (second / 10000.0);
