39 Eigen::Matrix<double, 6, 1> ImuBuffer::addToImuBuffer(double ts, const Eigen::Vector3d& linearAcc, const Eigen::Vector3d& angularVel) {
40  // Check that imuBufferLength was set
41  if (imuBufferLength_ < 0) {
42  throw std::runtime_error("GMsfImuBuffer: imuBufferLength has to be set by the user.");
43  }
44 
45  // Copy of IMU measurements
46  Eigen::Matrix<double, 6, 1> filteredImuMeas;
47 
48  // Potentially low pass filter IMU measurements
49  if (useImuSignalLowPassFilter_) {
50  filteredImuMeas = imuSignalLowPassFilterPtr_->filter(linearAcc, angularVel);
51  } else {
52  filteredImuMeas << linearAcc, angularVel;
53  }
54 
55  // Convert to gtsam type
56  graph_msf::ImuMeasurement imuMeas;
57  imuMeas.timestamp = ts;
58  imuMeas.acceleration = filteredImuMeas.head<3>();
59  imuMeas.angularVelocity = filteredImuMeas.tail<3>();
60 
61  // Add to buffer
62  {
63  // Writing to IMU buffer --> acquire mutex
64  const std::lock_guard<std::mutex> writeInBufferLock(writeInBufferMutex_);
65  timeToImuBuffer_[ts] = imuMeas;
66  // Update latest timestamp in buffer
67  if (ts > tLatestInBuffer_) {
68  tLatestInBuffer_ = ts;
69  }
70  }
71 
72  // If IMU buffer is too large, remove oldest element
73  if (timeToImuBuffer_.size() > imuBufferLength_) {
74  timeToImuBuffer_.erase(timeToImuBuffer_.begin());
75  }
76 
77  if (timeToImuBuffer_.size() > imuBufferLength_) {
78  std::ostringstream errorStream;
79  errorStream << YELLOW_START << "GMsf-ImuBuffer" << COLOR_END << " IMU Buffer has grown too large. It contains "
80  << timeToImuBuffer_.size() << " measurements instead of " << imuBufferLength_ << ".";
81  throw std::runtime_error(errorStream.str());
82  }
83 
84  return filteredImuMeas;
85 }

115 bool ImuBuffer::estimateAttitudeFromImu(gtsam::Rot3& initAttitude, double& gravityMagnitude, Eigen::Vector3d& gyrBias) {
116  // Make sure that imuBuffer is long enough
117  if (imuBufferLength_ < (imuRate_ * imuPoseInitWaitSecs_)) {
118  throw std::runtime_error(
119  "ImuBufferLength is not large enough for "
120  "initialization. Must be at least 1 second.");
121  }
122 
123  // Get timestamp of first message for lookup
124  if (timeToImuBuffer_.size() < (imuRate_ * imuPoseInitWaitSecs_)) {
125  return false;
126  } else {
127  // Accumulate Acceleration part of IMU Messages
128  Eigen::Vector3d initAccMean(0.0, 0.0, 0.0), initGyrMean(0.0, 0.0, 0.0);
129  for (auto& itr : timeToImuBuffer_) {
130  initAccMean += itr.second.acceleration;
131  initGyrMean += itr.second.angularVelocity;
132  }
133 
134  // Average IMU measurements and set assumed gravity direction
135  initAccMean /= timeToImuBuffer_.size();
136  gravityMagnitude = initAccMean.norm();
137  Eigen::Vector3d gUnitVecInWorld = Eigen::Vector3d(0.0, 0.0, 1.0); // ROS convention
138 
139  // Normalize gravity vectors to remove the affect of gravity magnitude from
140  // place-to-place
141  initAccMean.normalize();
142  initAttitude = gtsam::Rot3(Eigen::Quaterniond().setFromTwoVectors(initAccMean, gUnitVecInWorld));
143 
144  // Gyro
145  initGyrMean /= timeToImuBuffer_.size();
146  gyrBias = initGyrMean;
147 
148  // Calculate robot initial orientation using gravity vector.
149  std::cout << YELLOW_START << "GMsf-ImuBuffer" << COLOR_END << " Gravity Magnitude: " << gravityMagnitude << std::endl;
150  std::cout << YELLOW_START << "GMsf-ImuBuffer" << COLOR_END << " Mean IMU Acceleration Vector(x,y,z): " << initAccMean.transpose()
151  << " - Gravity Unit Vector(x,y,z): " << gUnitVecInWorld.transpose() << std::endl;
152  std::cout << YELLOW_START << "GMsf-ImuBuffer" << GREEN_START << " RPY (deg): " << initAttitude.rpy().transpose() * (180.0 / M_PI)
153  << COLOR_END << std::endl;
154  std::cout << YELLOW_START << "GMsf-ImuBuffer" << COLOR_END << " Gyro bias(x,y,z): " << initGyrMean.transpose() << std::endl;
155  }
156  return true;
157 }

159 bool ImuBuffer::getIMUBufferIteratorsInInterval(const double tsStart, const double tsEnd, TimeToImuMap::iterator& startIterator,
160  TimeToImuMap::iterator& endIterator) {
161  // Check if timestamps are in correct order
162  if (tsStart >= tsEnd) {
163  std::cerr << YELLOW_START << "GMsf-ImuBuffer" << RED_START << " IMU Lookup Timestamps are not correct ts_start(" << std::fixed
164  << tsStart << ") >= ts_end(" << tsEnd << ")\n";
165  return false;
166  }
167 
168  // Get Iterator Belonging to ts_start
169  startIterator = timeToImuBuffer_.lower_bound(tsStart);
170  // Get Iterator Belonging to ts_end
171  endIterator = timeToImuBuffer_.lower_bound(tsEnd);
172 
173  // Check if it is first value in the buffer which means there is no value
174  // before to interpolate with
175  if (startIterator == timeToImuBuffer_.begin()) {
176  std::cerr << YELLOW_START << "GMsf-ImuBuffer" << RED_START
177  << " Lookup requires first message of IMU buffer, cannot "
178  "Interpolate back, "
179  "Lookup Start/End: "
180  << std::fixed << tsStart << "/" << tsEnd << ", Buffer Start/End: " << timeToImuBuffer_.begin()->first << "/"
181  << timeToImuBuffer_.rbegin()->first << std::endl;
182  return false;
183  }
184 
185  // Check if lookup start time is ahead of buffer start time
186  if (startIterator == timeToImuBuffer_.end()) {
187  std::cerr << YELLOW_START << "GMsf-ImuBuffer" << RED_START
188  << " IMU Lookup start time ahead latest IMU message in the "
189  "buffer, lookup: "
190  << tsStart << ", latest IMU: " << timeToImuBuffer_.rbegin()->first << std::endl;
191  return false;
192  }
193 
194  // Check if last value is valid
195  if (endIterator == timeToImuBuffer_.end()) {
196  std::cerr << YELLOW_START << "GMsf-ImuBuffer" << RED_START << " Lookup is past IMU buffer, with lookup Start/End: " << std::fixed
197  << tsStart << "/" << tsEnd << " and latest IMU: " << timeToImuBuffer_.rbegin()->first << std::endl;
198  --endIterator;
199  }
200 
201  // Check if two IMU messages are different
202  if (startIterator == endIterator) {
203  std::cerr << YELLOW_START << "GMsf-ImuBuffer" << RED_START
204  << " Not Enough IMU values between timestamps , with Start/End: " << std::fixed << tsStart << "/" << tsEnd
205  << ", with diff: " << tsEnd - tsStart << std::endl;
206  return false;
207  }
208 
209  // If everything is good
210  return true;
211 }

213 // This function is better suitable for finding the closest IMU timestamp than the timeToKeyBuffer_ version, as there might be fewer keys
214 // than IMU measurements
215 bool ImuBuffer::getClosestImuMeasurement(double& returnedImuTimestamp, ImuMeasurement& returnedImuMeasurement,
216  const double maxSearchDeviation, const double tK) {
217  std::_Rb_tree_iterator<std::pair<const double, ImuMeasurement>> upperIterator;
218  {
219  // Read from IMU buffer --> acquire mutex
220  const std::lock_guard<std::mutex> writeInBufferLock(writeInBufferMutex_);
221  upperIterator = timeToImuBuffer_.upper_bound(tK);
222  }
223 
224  // Empty buffer
225  if (timeToImuBuffer_.empty()) {
226  std::cerr << YELLOW_START << "GMsf-ImuBuffer: Buffer is empty!" << COLOR_END << std::endl;
227  return false;
228  }
229 
230  auto lowerIterator = upperIterator;
231  --lowerIterator;
232 
233  // Keep key which is closer to tLidar
234  returnedImuTimestamp =
235  std::abs(tK - lowerIterator->first) < std::abs(upperIterator->first - tK) ? lowerIterator->first : upperIterator->first;
236  returnedImuMeasurement =
237  std::abs(tK - lowerIterator->first) < std::abs(upperIterator->first - tK) ? lowerIterator->second : upperIterator->second;
238  double timeDeviation = returnedImuTimestamp - tK;
239 
240  if (verboseLevel_ >= 2) {
241  std::cout << YELLOW_START << "GMsf-ImuBuffer" << COLOR_END << " Searched time step: " << std::setprecision(14) << tK << std::endl;
242  std::cout << YELLOW_START << "GMsf-ImuBuffer" << COLOR_END << " Found time step: " << std::setprecision(14) << returnedImuTimestamp
243  << std::endl;
244  std::cout << YELLOW_START << "GMsf-ImuBuffer" << COLOR_END << " Time Deviation (t_graph-t_request): " << 1000 * timeDeviation << " ms"
245  << std::endl;
246  std::cout << YELLOW_START << "GMsf-TimeKeyBuffer" << COLOR_END << " Latest IMU timestamp: " << tLatestInBuffer_
247  << ", hence absolut delay of measurement is " << 1000 * (tLatestInBuffer_ - tK) << "ms." << std::endl;
248  }
249 
250  // Check for error and warn user
251  if (std::abs(timeDeviation) > maxSearchDeviation) {
252  return false;
253  }
254 
255  return true;
256 }

258 gtsam::NavState ImuBuffer::integrateNavStateFromTimestamp(const double& tsStart, const double& tsEnd, const gtsam::NavState& stateStart,
259  const gtsam::imuBias::ConstantBias& imuBias,
260  const Eigen::Vector3d& W_gravityVector) {
261  // Get iterators
262  TimeToImuMap::iterator startIterator, nextToStartIterator, endIterator;
263  // Get IMU iterators in interval
264  if (!getIMUBufferIteratorsInInterval(tsStart, tsEnd, startIterator, endIterator)) {
265  throw std::runtime_error("Could not get IMU iterators in interval");
266  }
267  nextToStartIterator = startIterator;
268  ++nextToStartIterator;
269 
270  // Propagated state
271  gtsam::NavState propagatedState = stateStart;
272 
273  // For Loop for IMU integration
274  Eigen::Vector3d i_measAcceleration, i_measAngularVelocity;
275  for (; startIterator != endIterator; ++startIterator, ++nextToStartIterator) {
276  // Get IMU measurement
277  i_measAcceleration = imuBias.correctAccelerometer(nextToStartIterator->second.acceleration);
278  i_measAngularVelocity = imuBias.correctGyroscope(nextToStartIterator->second.angularVelocity);
279 
280  // Calculate dt
281  double dt = nextToStartIterator->first - startIterator->first;
282 
283  // Update propagated state
284  Eigen::Vector3d i_gravityVector = propagatedState.R().transpose() * W_gravityVector;
285  propagatedState =
286  propagatedState.update(i_gravityVector + i_measAcceleration, i_measAngularVelocity, dt, boost::none, boost::none, boost::none);
287  }
288  return propagatedState;
289 }
