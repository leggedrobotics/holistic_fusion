20 void AnymalEstimator::readParams(const ros::NodeHandle& privateNode) {
21  // Check
22  if (!graphConfigPtr_) {
23  throw std::runtime_error("AnymalEstimator: graphConfigPtr must be initialized.");
24  }
25 
26  // Sensor Params
27  // GNSS
28  gnssRate_ = graph_msf::tryGetParam<double>("sensor_params/gnssRate", privateNode);
29  // LIO
30  lioOdometryRate_ = graph_msf::tryGetParam<double>("sensor_params/lioOdometryRate", privateNode);
31  // Legged Between
32  leggedOdometryBetweenRate_ = graph_msf::tryGetParam<double>("sensor_params/leggedOdometryBetweenRate", privateNode);
33  leggedOdometryPoseDownsampleFactor_ = graph_msf::tryGetParam<int>("sensor_params/leggedOdometryPoseDownsampleFactor", privateNode);
34  // Legged Velocity
35  leggedOdometryVelocityRate_ = graph_msf::tryGetParam<double>("sensor_params/leggedOdometryVelocityRate", privateNode);
36  leggedOdometryVelocityDownsampleFactor_ =
37  graph_msf::tryGetParam<int>("sensor_params/leggedOdometryVelocityDownsampleFactor", privateNode);
38  // Legged Kinematics
39  leggedKinematicsRate_ = graph_msf::tryGetParam<double>("sensor_params/leggedKinematicsRate", privateNode);
40  leggedKinematicsDownsampleFactor_ = graph_msf::tryGetParam<int>("sensor_params/leggedKinematicsDownsampleFactor", privateNode);
41 
42  // Alignment Parameters
43  const auto initialSe3AlignmentNoise =
44  graph_msf::tryGetParam<std::vector<double>>("alignment_params/initialSe3AlignmentNoiseDensity", privateNode);
45  initialSe3AlignmentNoise_ << initialSe3AlignmentNoise[0], initialSe3AlignmentNoise[1], initialSe3AlignmentNoise[2],
46  initialSe3AlignmentNoise[3], initialSe3AlignmentNoise[4], initialSe3AlignmentNoise[5];
47  const auto lioSe3AlignmentRandomWalk =
48  graph_msf::tryGetParam<std::vector<double>>("alignment_params/lioSe3AlignmentRandomWalk", privateNode);
49  lioSe3AlignmentRandomWalk_ << lioSe3AlignmentRandomWalk[0], lioSe3AlignmentRandomWalk[1], lioSe3AlignmentRandomWalk[2],
50  lioSe3AlignmentRandomWalk[3], lioSe3AlignmentRandomWalk[4], lioSe3AlignmentRandomWalk[5];
51 
52  // Noise Parameters ---------------------------------------------------
54  const auto poseUnaryNoise =
55  graph_msf::tryGetParam<std::vector<double>>("noise_params/lioPoseUnaryNoiseDensity", privateNode); // roll,pitch,yaw,x,y,z
56  lioPoseUnaryNoise_ << poseUnaryNoise[0], poseUnaryNoise[1], poseUnaryNoise[2], poseUnaryNoise[3], poseUnaryNoise[4], poseUnaryNoise[5];
57 
59  const auto lioPoseBetweenNoise =
60  graph_msf::tryGetParam<std::vector<double>>("noise_params/lioPoseBetweenNoiseDensity", privateNode); // roll,pitch,yaw,x,y,z
61  lioPoseBetweenNoise_ << lioPoseBetweenNoise[0], lioPoseBetweenNoise[1], lioPoseBetweenNoise[2], lioPoseBetweenNoise[3],
62  lioPoseBetweenNoise[4], lioPoseBetweenNoise[5];
63

66  graph_msf::tryGetParam<std::vector<double>>("noise_params/legPoseBetweenNoiseDensity", privateNode); // roll,pitch,yaw,x,y,z
67  legPoseBetweenNoise_ << legPoseBetweenNoise[0], legPoseBetweenNoise[1], legPoseBetweenNoise[2], legPoseBetweenNoise[3],
68  legPoseBetweenNoise[4], legPoseBetweenNoise[5];
69 
71  const auto legVelocityUnaryNoise =
72  graph_msf::tryGetParam<std::vector<double>>("noise_params/legVelocityUnaryNoiseDensity", privateNode); // vx,vy,vz
73  legVelocityUnaryNoise_ << legVelocityUnaryNoise[0], legVelocityUnaryNoise[1], legVelocityUnaryNoise[2];
74 
76  const auto legKinematicsFootPositionUnaryNoise =
77  graph_msf::tryGetParam<std::vector<double>>("noise_params/legKinematicsFootPositionUnaryNoiseDensity", privateNode); // x,y,z
78  legKinematicsFootPositionUnaryNoise_ << legKinematicsFootPositionUnaryNoise[0], legKinematicsFootPositionUnaryNoise[1],
79  legKinematicsFootPositionUnaryNoise[2];
80 
81  // Flags ---------------------------------------------------
82  // GNSS Unary
83  useGnssUnaryFlag_ = graph_msf::tryGetParam<bool>("launch/usingGnssUnary", privateNode);
84  // LIO Unary
85  useLioUnaryFlag_ = graph_msf::tryGetParam<bool>("launch/usingLioUnary", privateNode);
86  // LIO Between
87  useLioBetweenFlag_ = graph_msf::tryGetParam<bool>("launch/usingLioBetween", privateNode);
88  // Legged Between Odometry
89  useLeggedBetweenFlag_ = graph_msf::tryGetParam<bool>("launch/usingLeggedBetween", privateNode);
90  // Legged Velocity Unary
91  useLeggedVelocityUnaryFlag_ = graph_msf::tryGetParam<bool>("launch/usingLeggedVelocityUnary", privateNode);
92  // Legged Kinematics
93  useLeggedKinematicsFlag_ = graph_msf::tryGetParam<bool>("launch/usingLeggedKinematics", privateNode);
94 
95  // Gnss parameters ---------------------------------------------------
96  if (useGnssUnaryFlag_) {
97  // GNSS Handler
98  gnssHandlerPtr_ = std::make_shared<graph_msf::GnssHandler>();
99 
100  // Read Yaw initial guess options
101  gnssHandlerPtr_->setUseYawInitialGuessFromFile(graph_msf::tryGetParam<bool>("gnss/useYawInitialGuessFromFile", privateNode));
102  gnssHandlerPtr_->setUseYawInitialGuessFromAlignment(graph_msf::tryGetParam<bool>("gnss/yawInitialGuessFromAlignment", privateNode));
103 
104  // Alignment options.
105  if (gnssHandlerPtr_->getUseYawInitialGuessFromAlignment()) {
106  // Make sure no dual true
107  gnssHandlerPtr_->setUseYawInitialGuessFromFile(false);
108  trajectoryAlignmentHandler_ = std::make_shared<graph_msf::TrajectoryAlignmentHandler>();
109 
110  trajectoryAlignmentHandler_->setSe3Rate(graph_msf::tryGetParam<double>("trajectoryAlignment/lidarRate", privateNode));
111  trajectoryAlignmentHandler_->setR3Rate(graph_msf::tryGetParam<double>("trajectoryAlignment/gnssRate", privateNode));
112 
113  trajectoryAlignmentHandler_->setMinDistanceHeadingInit(
114  graph_msf::tryGetParam<double>("trajectoryAlignment/minimumDistanceHeadingInit", privateNode));
115  trajectoryAlignmentHandler_->setNoMovementDistance(
116  graph_msf::tryGetParam<double>("trajectoryAlignment/noMovementDistance", privateNode));
117  trajectoryAlignmentHandler_->setNoMovementTime(graph_msf::tryGetParam<double>("trajectoryAlignment/noMovementTime", privateNode));
118 
119  } else if (!gnssHandlerPtr_->getUseYawInitialGuessFromAlignment() && gnssHandlerPtr_->getUseYawInitialGuessFromFile()) {
120  gnssHandlerPtr_->setGlobalYawDegFromFile(graph_msf::tryGetParam<double>("gnss/initYaw", privateNode));
121  }
122 
123  // GNSS Reference
124  gnssHandlerPtr_->setUseGnssReferenceFlag(graph_msf::tryGetParam<bool>("gnss/useGnssReference", privateNode));
125 
126  if (gnssHandlerPtr_->getUseGnssReferenceFlag()) {
127  REGULAR_COUT << GREEN_START << " Using GNSS reference from parameters." << COLOR_END << std::endl;
128  gnssHandlerPtr_->setGnssReferenceLatitude(graph_msf::tryGetParam<double>("gnss/referenceLatitude", privateNode));
129  gnssHandlerPtr_->setGnssReferenceLongitude(graph_msf::tryGetParam<double>("gnss/referenceLongitude", privateNode));
130  gnssHandlerPtr_->setGnssReferenceAltitude(graph_msf::tryGetParam<double>("gnss/referenceAltitude", privateNode));
131  gnssHandlerPtr_->setGnssReferenceHeading(graph_msf::tryGetParam<double>("gnss/referenceHeading", privateNode));
132  } else {
133  REGULAR_COUT << GREEN_START << " Will wait for GNSS measurements to initialize reference coordinates." << COLOR_END << std::endl;
134  }
135  } // End GNSS Unary
136 
137  // Coordinate Frames ---------------------------------------------------
139  dynamic_cast<AnymalStaticTransforms*>(staticTransformsPtr_.get())
140  ->setLioOdometryFrame(graph_msf::tryGetParam<std::string>("extrinsics/lioOdometryFrame", privateNode));
141 
143  dynamic_cast<AnymalStaticTransforms*>(staticTransformsPtr_.get())
144  ->setLeggedOdometryFrame(graph_msf::tryGetParam<std::string>("extrinsics/leggedOdometryFrame", privateNode));
145 
147  dynamic_cast<AnymalStaticTransforms*>(staticTransformsPtr_.get())
148  ->setGnssFrame(graph_msf::tryGetParam<std::string>("extrinsics/gnssFrame", privateNode));
149 }
