132 bool AnymalEstimator::srvOfflineSmootherOptimizeCallback(graph_msf_ros_msgs::OfflineOptimizationTrigger::Request& req,
133  graph_msf_ros_msgs::OfflineOptimizationTrigger::Response& res) {
134  // Call super class
135  bool success = graph_msf::GraphMsfRos::srvOfflineSmootherOptimizeCallback(req, res);
136 
137  // if GNSS is used at all, also log the reference frame
138  if (gnssCallbackCounter_ > 0) {
139  // Get the reference coordinates of the GNSS handler
140  const double referenceLatitude = gnssHandlerPtr_->getGnssReferenceLatitude();
141  const double referenceLongitude = gnssHandlerPtr_->getGnssReferenceLongitude();
142  const double referenceAltitude = gnssHandlerPtr_->getGnssReferenceAltitude();
143  const double referenceHeading = gnssHandlerPtr_->getGnssReferenceHeading();
144  REGULAR_COUT << " GNSS reference (latitude, longitude, altitude, heading): " << referenceLatitude << ", " << referenceLongitude << ", "
145  << referenceAltitude << ", " << referenceHeading << std::endl;
146 
147  // File streams
148  std::map<std::string, std::ofstream> fileStreams;
149 
150  // Write T_totalStation_totalStationOld_ to file ----------------------------
151  // Get time string by finding latest directory in optimizationResultLoggingPath
152  std::string timeString = graph_msf::getLatestSubdirectory(optimizationResultLoggingPath);
153  if (timeString.empty()) {
154  REGULAR_COUT << " Could not find latest directory in " << optimizationResultLoggingPath << std::endl;
155  return false;
156  } else {
157  REGULAR_COUT << " Found latest directory in " << optimizationResultLoggingPath << ": " << timeString << std::endl;
158  }
159  // Create file stream
160  std::string transformIdentifier = "gnss_reference_lat_lon_alt";
161  graph_msf::createLatLonAltCsvFileStream(fileStreams, optimizationResultLoggingPath, transformIdentifier, timeString);
162  // Add to file
163  graph_msf::writeLatLonAltToCsvFile(fileStreams, Eigen::Vector3d(referenceLatitude, referenceLongitude, referenceAltitude),
164  transformIdentifier, ros::Time::now().toSec());
165  REGULAR_COUT << " Wrote GNSS reference (latitude, longitude, altitude) coordinates to file." << std::endl;
166 
167  // If optional GNSS ENU coordinates are logged, write them to pose file
168  if constexpr (logOptionalGnssEnuCoordinatesFlag_) {
169  transformIdentifier = "gnss_enu_trajectory";
170  graph_msf::createPose3CsvFileStream(fileStreams, optimizationResultLoggingPath, transformIdentifier, timeString, false);
171  // Iterate over all optional GNSS ENU coordinates
172  int index = 0;
173  for (const auto& optionalGnssEnuCoordinates : optionalGnssEnuCoordinatesTrajectory_) {
174  // Create Isometry3d with identity rotation
175  Eigen::Isometry3d optionalGnssEnuCoordinatesIsometry = Eigen::Isometry3d::Identity();
176  optionalGnssEnuCoordinatesIsometry.translation() = optionalGnssEnuCoordinates;
177  // Add to file
178  graph_msf::writePose3ToCsvFile(fileStreams, optionalGnssEnuCoordinatesIsometry, transformIdentifier,
179  optionalGnssEnuCoordinatesTimeStamps_[index], false);
180  index++;
181  }
182  }
183  }
184 
185  // Return
186  return success;
187 }

190 void AnymalEstimator::gnssUnaryCallback_(const sensor_msgs::NavSatFix::ConstPtr& gnssMsgPtr) {
191  // Counter
192  ++gnssCallbackCounter_;
193 
194  // Convert to Eigen
195  Eigen::Vector3d gnssCoord = Eigen::Vector3d(gnssMsgPtr->latitude, gnssMsgPtr->longitude, gnssMsgPtr->altitude);
196  Eigen::Vector3d estStdDevXYZ(sqrt(gnssMsgPtr->position_covariance[0]), sqrt(gnssMsgPtr->position_covariance[4]),
197  sqrt(gnssMsgPtr->position_covariance[8]));
198 
199  // Initialize GNSS Handler
200  if (gnssCallbackCounter_ < NUM_GNSS_CALLBACKS_UNTIL_START) { // Accumulate measurements
201  // Wait until measurements got accumulated
202  accumulatedGnssCoordinates_ += gnssCoord;
203  if ((gnssCallbackCounter_ % 10) == 0) {
204  REGULAR_COUT << " NOT ENOUGH GNSS MESSAGES ARRIVED!" << std::endl;
205  }
206  return;
207  } else if (gnssCallbackCounter_ == NUM_GNSS_CALLBACKS_UNTIL_START) { // Initialize GNSS Handler
208  gnssHandlerPtr_->initHandler(accumulatedGnssCoordinates_ / NUM_GNSS_CALLBACKS_UNTIL_START);
209  REGULAR_COUT << " GNSS Handler initialized." << std::endl;
210  return;
211  }
212 
213  // Convert to Cartesian Coordinates
214  Eigen::Vector3d W_t_W_Gnss;
215  // gnssHandlerPtr_->convertNavSatToPosition(gnssCoord, W_t_W_Gnss);
216  gnssHandlerPtr_->convertNavSatToPositionLV03(gnssCoord, W_t_W_Gnss);
217  std::string fixedFrame = staticTransformsPtr_->getWorldFrame(); // Alias
218  // fixedFrame = "east_north_up";
219 
220  // // For Debugging: Add Gaussian Noise with 0.1m std deviation
221  // // Random number generator
222  // std::default_random_engine generator(std::chrono::system_clock::now().time_since_epoch().count());
223  // // Add noise
224  // W_t_W_Gnss +=
225  // Eigen::Vector3d(std::normal_distribution<double>(0.0, 0.1)(generator), std::normal_distribution<double>(0.0, 0.1)(generator),
226  // std::normal_distribution<double>(0.0, 0.1)(generator));
227  // // Add to assumed standard deviation (\sigma_{GNSS+noise} = \sqrt{\sigma_{GNSS}^2 + \sigma_{noise}^2})
228  // for (int i = 0; i < 3; ++i) {
229  // estStdDevXYZ(i) = sqrt(estStdDevXYZ(i) * estStdDevXYZ(i) + 0.1 * 0.1);
230  // }
231 
232  // Initial world yaw initialization options
233  // Case 1: Initialization
234  if (!areYawAndPositionInited()) {
235  // a: Default
236  double initYaw_W_Base{0.0}; // Default is 0 yaw
237  // b: From file
238  if (gnssHandlerPtr_->getUseYawInitialGuessFromFile()) {
239  initYaw_W_Base = gnssHandlerPtr_->getGlobalYawDegFromFile() / 180.0 * M_PI;
240  }
241  // c: From alignment
242  else if (gnssHandlerPtr_->getUseYawInitialGuessFromAlignment()) {
243  // Adding the GNSS measurement
244  trajectoryAlignmentHandler_->addR3Position(W_t_W_Gnss, gnssMsgPtr->header.stamp.toSec());
245  // In radians
246  Eigen::Isometry3d T_W_Base = Eigen::Isometry3d::Identity();
247  if (!(trajectoryAlignmentHandler_->alignTrajectories(initYaw_W_Base, T_W_Base))) {
248  if (gnssCallbackCounter_ % 10 == 0) {
249  REGULAR_COUT << YELLOW_START << "Trajectory alignment not ready. Waiting for more motion." << COLOR_END << std::endl;
250  }
251  return;
252  }
253  REGULAR_COUT << GREEN_START << "Trajectory Alignment Successful. Obtained Yaw Value of T_W_Base (deg): " << COLOR_END
254  << 180.0 * initYaw_W_Base / M_PI << std::endl;
255  }
256 
257  // Actual Initialization
258  if (this->initYawAndPositionInWorld(initYaw_W_Base, W_t_W_Gnss,
259  dynamic_cast<AnymalStaticTransforms*>(staticTransformsPtr_.get())->getBaseLinkFrame(),
260  dynamic_cast<AnymalStaticTransforms*>(staticTransformsPtr_.get())->getGnssFrame())) {
261  REGULAR_COUT << GREEN_START << " GNSS initialization of yaw and position successful." << std::endl;
262 
263  } else {
264  REGULAR_COUT << RED_START << " GNSS initialization of yaw and position failed." << std::endl;
265  }
266  } else { // Case 2: Already initialized --> Unary factor
267  const std::string& gnssFrameName = dynamic_cast<AnymalStaticTransforms*>(staticTransformsPtr_.get())->getGnssFrame(); // Alias
268  // Measurement
269  graph_msf::UnaryMeasurementXDAbsolute<Eigen::Vector3d, 3> meas_W_t_W_Gnss(
270  "GnssPosition", int(gnssRate_), gnssFrameName, gnssFrameName + sensorFrameCorrectedNameId, graph_msf::RobustNorm::None(),
271  gnssMsgPtr->header.stamp.toSec(), 1.0, W_t_W_Gnss, estStdDevXYZ, fixedFrame, staticTransformsPtr_->getWorldFrame());
272  this->addUnaryPosition3AbsoluteMeasurement(meas_W_t_W_Gnss);
273  }
274 
275  // Add _gmsf to the frame
276  if (fixedFrame != staticTransformsPtr_->getWorldFrame()) {
277  fixedFrame += referenceFrameAlignedNameId;
278  }
279 
281  addToPathMsg(measGnss_worldGnssPathPtr_, fixedFrame, gnssMsgPtr->header.stamp, W_t_W_Gnss, graphConfigPtr_->imuBufferLength_ * 4);
283  pubMeasWorldGnssPath_.publish(measGnss_worldGnssPathPtr_);
284 
285  // Potentially log the GNSS ENU coordinates
286  if (logOptionalGnssEnuCoordinatesFlag_) {
287  optionalGnssEnuCoordinatesTimeStamps_.push_back(gnssMsgPtr->header.stamp.toSec());
288  optionalGnssEnuCoordinatesTrajectory_.push_back(W_t_W_Gnss);
289  }
290 }

334 void AnymalEstimator::lidarBetweenCallback_(const nav_msgs::Odometry::ConstPtr& odomLidarPtr) {
335  if (!areRollAndPitchInited()) {
336  return;
337  }
338 
339  // Counter
340  ++lidarBetweenCallbackCounter_;
341 
342  // Convert
343  Eigen::Isometry3d lio_T_M_Lk = Eigen::Isometry3d::Identity();
344  graph_msf::odomMsgToEigen(*odomLidarPtr, lio_T_M_Lk.matrix());
345  // Get the time
346  double lidarBetweenTimeK = odomLidarPtr->header.stamp.toSec();
347 
348  // At start
349  if (lidarBetweenCallbackCounter_ == 0) {
350  lio_T_M_Lkm1_ = lio_T_M_Lk;
351  lidarBetweenTimeKm1_ = lidarBetweenTimeK;
352  }
353 
354  // Add to trajectory aligner if needed.
355  if (useGnssUnaryFlag_ && gnssHandlerPtr_->getUseYawInitialGuessFromAlignment()) {
356  trajectoryAlignmentHandler_->addSe3Position(lio_T_M_Lk.translation(), lidarBetweenTimeK);
357  }
358 
359  // Frame Name
360  const std::string& lioOdomFrameName = dynamic_cast<AnymalStaticTransforms*>(staticTransformsPtr_.get())->getLioOdometryFrame(); // alias
361 
362  // State Machine
363  if (lidarBetweenCallbackCounter_ <= 2) {
364  return;
365  } else if (!areYawAndPositionInited()) { // Initializing
366  if (!useGnssUnaryFlag_ && !useLioUnaryFlag_) {
367  // Measurement
368  graph_msf::UnaryMeasurementXD<Eigen::Isometry3d, 6> unary6DMeasurement(
369  "Lidar_unary_6D", int(lioOdometryRate_), lioOdomFrameName, lioOdomFrameName + sensorFrameCorrectedNameId,
370  graph_msf::RobustNorm::None(), lidarBetweenTimeK, 1.0, lio_T_M_Lk, lioPoseUnaryNoise_);
371  // Add to graph
372  REGULAR_COUT << GREEN_START << " LiDAR odometry callback is setting global yaw, as it was not set so far." << COLOR_END << std::endl;
373  this->initYawAndPosition(unary6DMeasurement);
374  }
375  } else { // Already initialized --> Between factor
376  // Compute Delta
377  const Eigen::Isometry3d T_Lkm1_Lk = lio_T_M_Lkm1_.inverse() * lio_T_M_Lk;
378  // Create measurement
379  graph_msf::BinaryMeasurementXD<Eigen::Isometry3d, 6> delta6DMeasurement(
380  "Lidar_between_6D", int(lioOdometryRate_), lioOdomFrameName, lioOdomFrameName + sensorFrameCorrectedNameId,
381  graph_msf::RobustNorm::None(), lidarBetweenTimeKm1_, lidarBetweenTimeK, T_Lkm1_Lk, lioPoseUnaryNoise_);
382  // Add to graph
383  this->addBinaryPose3Measurement(delta6DMeasurement);
384  }
385  // Provide for next iteration
386  lio_T_M_Lkm1_ = lio_T_M_Lk;
387  lidarBetweenTimeKm1_ = lidarBetweenTimeK;
388 
389  // Visualization ----------------------------
390  // Add to path message
391  addToPathMsg(measLio_mapLidarPathPtr_, staticTransformsPtr_->getWorldFrame(), odomLidarPtr->header.stamp, lio_T_M_Lk.translation(),
392  graphConfigPtr_->imuBufferLength_ * 4);
393 
394  // Publish Path
395  pubMeasMapLioPath_.publish(measLio_mapLidarPathPtr_);
396 }

434  }

445  graph_msf::RobustNorm::None(), legOdometryTimeKm1_, legOdometryTimeK, T_Bkm1_Bk, legPoseBetweenNoise_);
446  // Add to graph
447  this->addBinaryPose3Measurement(delta6DMeasurement);
448 
449  // Prepare for next iteration
450  T_O_Bl_km1_ = T_O_Bl_k;
451  legOdometryTimeKm1_ = legOdometryTimeK;
452  }
453  }

504 void AnymalEstimator::leggedKinematicsCallback_(const anymal_msgs::AnymalState::ConstPtr& anymalStatePtr) {
505  if (!areRollAndPitchInited()) {
506  return;
507  }
508 
509  // Counter
510  ++leggedKinematicsCallbackCounter_;
511 
512  if (!areYawAndPositionInited()) {
513  if (!useGnssUnaryFlag_ && !useLioUnaryFlag_ && !useLioBetweenFlag_ && !useLeggedBetweenFlag_ && !useLeggedVelocityUnaryFlag_) {
514  // Measurement
515  graph_msf::UnaryMeasurementXD<Eigen::Isometry3d, 6> unary6DMeasurement(
516  "Leg_odometry_6D", int(leggedKinematicsRate_),
517  dynamic_cast<AnymalStaticTransforms*>(staticTransformsPtr_.get())->getLeggedOdometryFrame(),
518  dynamic_cast<AnymalStaticTransforms*>(staticTransformsPtr_.get())->getLeggedOdometryFrame() + sensorFrameCorrectedNameId,
519  graph_msf::RobustNorm::None(), anymalStatePtr->header.stamp.toSec(), 1.0, Eigen::Isometry3d::Identity(),
520  Eigen::Matrix<double, 6, 1>::Identity());
521  // Add to graph
522  REGULAR_COUT << GREEN_START << " Legged kinematics callback is setting global yaw, as it was not set so far." << COLOR_END
523  << std::endl;
524  this->initYawAndPosition(unary6DMeasurement);
525  }
526  }
527  // Normal Operation
528  else {
529  // Only add every nth measurement
530  int measurementRate = static_cast<int>(leggedKinematicsRate_) / leggedKinematicsDownsampleFactor_;
531 
532  // Check whether it is time to add a measurement
533  if ((leggedKinematicsCallbackCounter_ % leggedKinematicsDownsampleFactor_) == 0) {
534  // Alias
535  const std::string& leggedOdometryFrameName =
536  dynamic_cast<AnymalStaticTransforms*>(staticTransformsPtr_.get())->getLeggedOdometryFrame();
537 
538  // Pose of the base in odom frame
539  Eigen::Isometry3d T_O_B = Eigen::Isometry3d::Identity();
540  graph_msf::geometryPoseToEigen(anymalStatePtr->pose.pose, T_O_B.matrix());
541 
542  // Create the unary measurement for each foot (loop unrolling with constexpr arrays) ----------------------------
543  visualization_msgs::MarkerArray footContactMarkers = visualization_msgs::MarkerArray();
544  // Loop
545  for (int legIndex = 0; legIndex < legNames_.size(); ++legIndex) {
546  // Get contact state
547  bool footInContact = anymalStatePtr->contacts[legIndex].state;
548  // Foot name as specified in the message
549  const std::string legName = anymalStatePtr->contacts[legIndex].name;
550 
551  // Check whether leg is in contact ----------------------------
552  if (footInContact) {
553  // Increase debounce counter
554  ++legInContactForNSteps_[legIndex];
555 
556  // Check whether leg was at least N times in contact to count it as in contact --> debouncing
557  if (legInContactForNSteps_[legIndex] >= legInContactDebounceThreshold_) {
558  // If it was not in contact before
559  if (legInContactForNSteps_[legIndex] == legInContactDebounceThreshold_) {
560  // Increase contact counter
561  ++legTotalContactsCounter_[legIndex];
562  // Print
563  if (graphConfigPtr_->verboseLevel_ > 1) {
564  REGULAR_COUT << " Leg " << legName << " came back into contact for the " << legTotalContactsCounter_[legIndex] << ". time."
565  << std::endl;
566  }
567  }
568 
569  // Add measurement ----------------------------
570  // Position of foot in legged odometry frame
571  Eigen::Vector3d O_t_O_foot =
572  Eigen::Vector3d(anymalStatePtr->contacts[legIndex].position.x, anymalStatePtr->contacts[legIndex].position.y,
573  anymalStatePtr->contacts[legIndex].position.z);
574  Eigen::Vector3d B_t_B_foot = T_O_B.inverse() * O_t_O_foot;
575 
576  // Create the unary measurement with contact counter
577  std::string legIdentifier = legName;
578  graph_msf::UnaryMeasurementXDLandmark<Eigen::Vector3d, 3> footContactPositionMeasurement(
579  legIdentifier, measurementRate, leggedOdometryFrameName, leggedOdometryFrameName + sensorFrameCorrectedNameId,
580  graph_msf::RobustNorm::None(), anymalStatePtr->header.stamp.toSec(), 1.0, B_t_B_foot, //graph_msf::RobustNorm::Huber(1)
581  legKinematicsFootPositionUnaryNoise_, staticTransformsPtr_->getWorldFrame());
582 
583  // Add to graph
584  this->addUnaryPosition3LandmarkMeasurement(footContactPositionMeasurement, legTotalContactsCounter_[legIndex]);
585 
586  // Visualize foot contact in RViz
587  visualization_msgs::Marker footContactMarker;
588  createContactMarker(leggedOdometryFrameName, anymalStatePtr->header.stamp, B_t_B_foot, "foot_contact", legIndex,
589  Eigen::Vector3d(0.0, 1.0, 0.0), footContactMarker);
590  footContactMarkers.markers.push_back(footContactMarker);
591  }
592  }
593  // Leg not in contact ----------------------------
594  else if (legInContactForNSteps_[legIndex] > 0) {
595  // Set to 0
596  legInContactForNSteps_[legIndex] = 0;
597  // Print
598  if (graphConfigPtr_->verboseLevel_ > 1) {
599  REGULAR_COUT << " Leg " << legName << " lost contact. " << std::endl;
600  }
601  // Remove marker if it lost contact
602  visualization_msgs::Marker deleteMarker;
603  deleteMarker.header.frame_id = leggedOdometryFrameName;
604  deleteMarker.header.stamp = anymalStatePtr->header.stamp;
605  deleteMarker.ns = "foot_contact";
606  deleteMarker.id = legIndex;
607  deleteMarker.action = visualization_msgs::Marker::DELETE;
608  footContactMarkers.markers.push_back(deleteMarker);
609  }
610  }
611  // Publish foot contact markers
612  if (!footContactMarkers.markers.empty() && pubFootContactMarkers_.getNumSubscribers() > 0) {
613  pubFootContactMarkers_.publish(footContactMarkers);
614  }
615  }
616  }
617 }
