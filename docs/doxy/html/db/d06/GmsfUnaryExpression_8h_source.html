60  gtsam::Expression<GTSAM_MEASUREMENT_TYPE> createAndReturnExpression(const gtsam::Key& closestGeneralKey,
61  DynamicDictionaryContainer& gtsamDynamicExpressionKeys,
62  const gtsam::NavState& W_imuPropagatedState,
63  const bool optimizeReferenceFramePosesWrtWorldFlag,
64  const bool centerMeasurementsAtKeyframePositionBeforeAlignmentFlag,
65  const bool optimizeExtrinsicSensorToSensorCorrectedOffsetFlag) {
66  // Measurement Pointer
67  const auto& gmsfUnaryMeasurement = *getGmsfBaseUnaryMeasurementPtr();
68 
69  // A. Generate Expression for Basic IMU State in World Frame at Key --------------------------------
70  generateImuStateInWorldFrameAtKey(closestGeneralKey);
71 
72  // B.A. Holistic Fusion: Optimize over reference frame poses --------------------------------------------
73  if constexpr (TYPE == UnaryExpressionType::Absolute) {
74  if (optimizeReferenceFramePosesWrtWorldFlag) {
75  transformImuStateFromWorldToReferenceFrame(gtsamDynamicExpressionKeys, W_imuPropagatedState,
76  centerMeasurementsAtKeyframePositionBeforeAlignmentFlag);
77  }
78  }
79 
80  // B.B. Holistic Fusion: Create Landmark State in Dynamic Memory -------------------------------------
81  if constexpr (TYPE == UnaryExpressionType::Landmark) {
82  transformLandmarkInWorldToImuFrame(gtsamDynamicExpressionKeys, W_imuPropagatedState);
83  }
84 
85  // C. Transform State to Sensor Frame -----------------------------------------------------
86  if (gmsfUnaryMeasurement.sensorFrameName() != imuFrame_) {
87  transformImuStateToSensorFrameState();
88  }
89 
90  // D. Extrinsic Calibration: Add correction to sensor pose -----------------------------------------
91  if (optimizeExtrinsicSensorToSensorCorrectedOffsetFlag) {
92  transformSensorFrameStateToSensorFrameCorrectedState(gtsamDynamicExpressionKeys);
93  }
94 
95  // Return
96  return getGtsamExpression();
97  }

149  virtual void transformSensorFrameStateToSensorFrameCorrectedState(DynamicDictionaryContainer& gtsamDynamicExpressionKeys) {
150  // Mutex because we are changing the dynamically allocated graphKeys
151  std::lock_guard<std::mutex> modifyGraphKeysLock(gtsamDynamicExpressionKeys.get<gtsam::Pose3>().mutex());
152 
153  // Initial Guess
154  GTSAM_MEASUREMENT_TYPE initialGuess = GTSAM_MEASUREMENT_TYPE::Identity();
155 
156  // Search for new graph key
157  bool newGraphKeyAddedFlag = false;
158  DynamicVariableType variableType = DynamicVariableType::Global(gmsfBaseUnaryMeasurementPtr_->timeK());
159  DynamicFactorGraphStateKey graphKey = gtsamDynamicExpressionKeys.get<gtsam::Pose3>().getTransformationKey<CALIBRATION_CHAR>(
160  newGraphKeyAddedFlag, gmsfBaseUnaryMeasurementPtr_->sensorFrameName(), gmsfBaseUnaryMeasurementPtr_->sensorFrameCorrectedName(),
161  gmsfBaseUnaryMeasurementPtr_->timeK(), convertToPose3(initialGuess), variableType);
162 
163  // Define expression
164  gtsam::Expression<GTSAM_MEASUREMENT_TYPE> exp_C_sensorFrame_sensorFrameCorrected(graphKey.key());
165 
166  // Apply calibration correction
167  this->applyExtrinsicCalibrationCorrection(exp_C_sensorFrame_sensorFrameCorrected); // Virtual method to be implemented
168 
169  // If key was newly added, just add a value to the new state values to online and offline graph (as completely new)
170  if (newGraphKeyAddedFlag) {
171  // Add value to new state values of online and offline graph
172  newOnlineStateValues_.insert(graphKey.key(), initialGuess);
173  newOfflineStateValues_.insert(graphKey.key(), initialGuess);
174  // Add prior factor to online graph --> make sure it is not fully unconstrained
175  // newOnlineDynamicPriorFactors_.emplace_back(
176  // graphKey.key(), initialGuess,
177  // gtsam::noiseModel::Gaussian::Covariance(gtsam::Matrix::Identity(GTSAM_MEASUREMENT_TYPE::dimension,
178  // GTSAM_MEASUREMENT_TYPE::dimension)));
179  // New key has to be active
180  assert(graphKey.isVariableActive());
181  }
182  // If the key is inactive(), we have to activate it again and add a prior with the previous belief ONLY to the online graph
183  // TODO: Finish up
184  else if (!graphKey.isVariableActive() && graphKey.getNumberStepsOptimized() > 0) {
185  assert(graphKey.getNumberStepsOptimized() > 0);
186  REGULAR_COUT << " GmsfUnaryExpression: Transformation between " << gmsfBaseUnaryMeasurementPtr_->sensorFrameName() << " and "
187  << gmsfBaseUnaryMeasurementPtr_->sensorFrameCorrectedName()
188  << " was inactive. Activating it again and adding a prior to the online graph." << std::endl;
189  // Add prior factor
190  gtsam::Pose3 priorBelief = graphKey.getTransformationAfterOptimization(); // TODO: Does not have to be 6D
191  gtsam::Matrix66 priorCovariance = graphKey.getCovarianceAfterOptimization();
192  // Convert back to GTSAM_MEASUREMENT_TYPE
193  GTSAM_MEASUREMENT_TYPE priorBeliefConverted = convertFromPose3(priorBelief);
194  gtsam::Matrix priorCovarianceConverted = priorCovariance.bottomRightCorner<6, 6>(); // TODO: Not correct
195  // Add prior factor to online graph
196  newOnlineDynamicPriorFactors_.emplace_back(graphKey.key(), priorBeliefConverted,
197  gtsam::noiseModel::Gaussian::Covariance(priorCovarianceConverted));
198  // Add value to new state values of online graph
199  newOnlineStateValues_.insert(graphKey.key(), priorBeliefConverted);
200  // Activate key and set number of steps optimized to 0
201  graphKey.activateVariable();
202  graphKey.resetNumberStepsOptimized();
203  }
204  // If it was never optimized, we can also remove it as we will not be able to reactivate it
205  else if (!graphKey.isVariableActive() && graphKey.getNumberStepsOptimized() == 0) {
206  REGULAR_COUT << " GmsfUnaryExpression: Transformation between " << gmsfBaseUnaryMeasurementPtr_->sensorFrameName() << " and "
207  << gmsfBaseUnaryMeasurementPtr_->sensorFrameCorrectedName()
208  << " is deactivated but was never optimized. Hence, Removing it from the graph." << std::endl;
209  // Remove key
210  DynamicFactorGraphStateKey<gtsam::Pose3> keyToRemoveOrDeactivate;
211  std::ignore = gtsamDynamicExpressionKeys.get<gtsam::Pose3>().removeTransform(gmsfBaseUnaryMeasurementPtr_->sensorFrameName(),
212  gmsfBaseUnaryMeasurementPtr_->sensorFrameCorrectedName(),
213  keyToRemoveOrDeactivate);
214  }
