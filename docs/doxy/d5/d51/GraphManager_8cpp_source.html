71bool GraphManager::initImuIntegrators(const double gravityValue) {
72 // Gravity direction definition
73 imuParamsPtr_ = gtsam::PreintegratedCombinedMeasurements::Params::MakeSharedU(gravityValue); // ROS convention
74
75 // Set noise and bias parameters
77 imuParamsPtr_->setAccelerometerCovariance(gtsam::Matrix33::Identity(3, 3) * std::pow(graphConfigPtr_->accNoiseDensity_, 2));
78 imuParamsPtr_->setIntegrationCovariance(gtsam::Matrix33::Identity(3, 3) *
79 std::pow(graphConfigPtr_->integrationNoiseDensity_, 2)); // error committed in integrating
80 // position from velocities
81 imuParamsPtr_->setUse2ndOrderCoriolis(graphConfigPtr_->use2ndOrderCoriolisFlag_);
83 imuParamsPtr_->setGyroscopeCovariance(gtsam::Matrix33::Identity(3, 3) * std::pow(graphConfigPtr_->gyroNoiseDensity_, 2));
84 imuParamsPtr_->setOmegaCoriolis(gtsam::Vector3(0, 0, 1) * graphConfigPtr_->omegaCoriolis_);
86 imuParamsPtr_->setBiasAccCovariance(gtsam::Matrix33::Identity(3, 3) * std::pow(graphConfigPtr_->accBiasRandomWalkNoiseDensity_, 2));
87 imuParamsPtr_->setBiasOmegaCovariance(gtsam::Matrix33::Identity(3, 3) * std::pow(graphConfigPtr_->gyroBiasRandomWalkNoiseDensity_, 2));
88 imuParamsPtr_->setBiasAccOmegaInit(gtsam::Matrix66::Identity(6, 6) * std::pow(graphConfigPtr_->biasAccOmegaInit_, 2));
89
90 // Use previously defined prior for gyro
91 imuBiasPriorPtr_ = std::make_shared<gtsam::imuBias::ConstantBias>(graphConfigPtr_->accBiasPrior_, graphConfigPtr_->gyroBiasPrior_);
92
93 // Init Pre-integrators
94 imuStepPreintegratorPtr_ = std::make_shared<gtsam::PreintegratedCombinedMeasurements>(imuParamsPtr_, *imuBiasPriorPtr_);
95 imuBufferPreintegratorPtr_ = std::make_shared<gtsam::PreintegratedCombinedMeasurements>(imuParamsPtr_, *imuBiasPriorPtr_);
96 imuParamsPtr_->print("GraphMSF-IMU PreIntegration Parameters:");
97 return true;
98}

100bool GraphManager::initPoseVelocityBiasGraph(const double timeStamp, const gtsam::Pose3& T_W_I0, const gtsam::Pose3& T_O_I0) {
101 // Create Prior factor ----------------------------------------------------
103 auto priorPoseNoise = gtsam::noiseModel::Diagonal::Sigmas(
104 (gtsam::Vector(6) << graphConfigPtr_->initialOrientationNoiseDensity_, graphConfigPtr_->initialOrientationNoiseDensity_,
105 graphConfigPtr_->initialOrientationNoiseDensity_, graphConfigPtr_->initialPositionNoiseDensity_,
106 graphConfigPtr_->initialPositionNoiseDensity_, graphConfigPtr_->initialPositionNoiseDensity_)
107 .finished()); // rad,rad,rad,m, m, m
108 auto priorVelocityNoise = gtsam::noiseModel::Isotropic::Sigma(3, graphConfigPtr_->initialVelocityNoiseDensity_); // m/s
109 auto priorBiasNoise = gtsam::noiseModel::Diagonal::Sigmas((gtsam::Vector(6) << graphConfigPtr_->initialAccBiasNoiseDensity_, // m/s^2
110 graphConfigPtr_->initialAccBiasNoiseDensity_, // m/s^2
111 graphConfigPtr_->initialAccBiasNoiseDensity_, // m/s^2
112 graphConfigPtr_->initialGyroBiasNoiseDensity_, // rad/s
113 graphConfigPtr_->initialGyroBiasNoiseDensity_, // rad/s
114 graphConfigPtr_->initialGyroBiasNoiseDensity_) // rad/s
115 .finished()); // acc, acc, acc, gyro, gyro, gyro
116
117 // Pre-allocate
118 gtsam::NonlinearFactorGraph newRtGraphFactors, newBatchGraphFactors;
119 gtsam::Values newRtGraphValues, newBatchGraphValues;
120 std::shared_ptr<std::map<gtsam::Key, double>> priorKeyTimestampMapPtr = std::make_shared<std::map<gtsam::Key, double>>();
121
122 // Looking up from IMU buffer --> acquire mutex (otherwise values for key might not be set)
123 {
124 const std::lock_guard<std::mutex> operateOnGraphDataLock(operateOnGraphDataMutex_);
125
126 // Initial estimate
127 gtsam::Values valuesEstimate;
128 REGULAR_COUT << " Initial Pose of imu in world frame: " << T_W_I0 << std::endl;
129 valuesEstimate.insert(gtsam::symbol_shorthand::X(propagatedStateKey_), T_W_I0);
130 REGULAR_COUT << " Initial velocity assumed to be: " << gtsam::Vector3(0, 0, 0) << std::endl;
131 valuesEstimate.insert(gtsam::symbol_shorthand::V(propagatedStateKey_), gtsam::Vector3(0, 0, 0));
132 REGULAR_COUT << " Initial bias set to: " << *imuBiasPriorPtr_ << std::endl;
133 valuesEstimate.insert(gtsam::symbol_shorthand::B(propagatedStateKey_), *imuBiasPriorPtr_);
135 rtGraphValuesBufferPtr_->insert(valuesEstimate);
136 batchGraphValuesBufferPtr_->insert(valuesEstimate); // TODO: Check whether velocity can be removed from batch
137 writeValueKeysToKeyTimeStampMap_(valuesEstimate, timeStamp, priorKeyTimestampMapPtr);
138
139 // Initialize graph -------------------------------------------------
140 rtFactorGraphBufferPtr_->resize(0);
141 batchFactorGraphBufferPtr_->resize(0);
142 rtFactorGraphBufferPtr_->emplace_shared<gtsam::PriorFactor<gtsam::Pose3>>(
143 gtsam::symbol_shorthand::X(propagatedStateKey_), T_W_I0,
144 priorPoseNoise); // POSE - PriorFactor format is (key,value,matrix) value
145 batchFactorGraphBufferPtr_->emplace_shared<gtsam::PriorFactor<gtsam::Pose3>>(
146 gtsam::symbol_shorthand::X(propagatedStateKey_), T_W_I0,
147 priorPoseNoise); // POSE - PriorFactor format is (key,value,matrix) value
148 // is same type as type of PriorFactor
149 rtFactorGraphBufferPtr_->emplace_shared<gtsam::PriorFactor<gtsam::Vector3>>(gtsam::symbol_shorthand::V(propagatedStateKey_),
150 gtsam::Vector3(0, 0, 0),
151 priorVelocityNoise); // VELOCITY
152 batchFactorGraphBufferPtr_->emplace_shared<gtsam::PriorFactor<gtsam::Vector3>>(gtsam::symbol_shorthand::V(propagatedStateKey_),
153 gtsam::Vector3(0, 0, 0),
154 priorVelocityNoise); // VELOCITY
155 rtFactorGraphBufferPtr_->emplace_shared<gtsam::PriorFactor<gtsam::imuBias::ConstantBias>>(
156 gtsam::symbol_shorthand::B(propagatedStateKey_), *imuBiasPriorPtr_,
157 priorBiasNoise); // BIAS
158 batchFactorGraphBufferPtr_->emplace_shared<gtsam::PriorFactor<gtsam::imuBias::ConstantBias>>(
159 gtsam::symbol_shorthand::B(propagatedStateKey_), *imuBiasPriorPtr_,
160 priorBiasNoise); // BIAS
161
162 // Add to Time Key Buffer
163 timeToKeyBufferPtr_->addToBuffer(timeStamp, propagatedStateKey_);
164
165 // Copy over
166 // Rt
167 newRtGraphFactors = *rtFactorGraphBufferPtr_;
168 newRtGraphValues = *rtGraphValuesBufferPtr_;
169 rtFactorGraphBufferPtr_->resize(0);
170 rtGraphValuesBufferPtr_->clear();
171 // Batch
172 newBatchGraphFactors = *batchFactorGraphBufferPtr_;
173 newBatchGraphValues = *batchGraphValuesBufferPtr_;
174 batchFactorGraphBufferPtr_->resize(0);
175 batchGraphValuesBufferPtr_->clear();
176 }
177
179 // Rt Smoother
180 addFactorsToRtSmootherAndOptimize(newRtGraphFactors, newRtGraphValues, *priorKeyTimestampMapPtr, graphConfigPtr_, 0);
181 // Batch Smoother
182 addFactorsToBatchSmootherAndOptimize(newBatchGraphFactors, newBatchGraphValues, *priorKeyTimestampMapPtr, graphConfigPtr_);
183
184 // Update Current State ---------------------------------------------------
185 const std::lock_guard<std::mutex> operateOnGraphDataLock(operateOnGraphDataMutex_);
186 optimizedGraphState_.updateNavStateAndBias(propagatedStateKey_, timeStamp, gtsam::NavState(T_W_I0, gtsam::Vector3(0, 0, 0)),
187 gtsam::Vector3(0, 0, 0), *imuBiasPriorPtr_);
188 O_imuPropagatedState_ = gtsam::NavState(T_O_I0, gtsam::Vector3(0, 0, 0));
189 W_imuPropagatedState_ = gtsam::NavState(T_W_I0, gtsam::Vector3(0, 0, 0));
190 T_W_O_ = (T_W_I0.inverse() * T_O_I0).matrix();
191 return true;
192}

194// IMU at the core --------------------------------------------------------------
195void GraphManager::addImuFactorAndGetState(SafeIntegratedNavState& returnPreIntegratedNavState,
196 std::shared_ptr<SafeNavStateWithCovarianceAndBias>& newOptimizedNavStatePtr,
197 const std::shared_ptr<ImuBuffer>& imuBufferPtr, const double imuTimeK, bool createNewStateFlag) {
198 // Logging of latency
199 if (graphConfigPtr_->logLatencyAndUpdateDurationToMemoryFlag_) {
200 latencyStartTime_ = std::chrono::high_resolution_clock::now();
201 }
202
203 // Looking up from IMU buffer --> acquire mutex (otherwise values for key might not be set)
204 const std::lock_guard<std::mutex> operateOnGraphDataLock(operateOnGraphDataMutex_);
205
206 // Part 1 (ALWAYS): Propagate state and imu measurement to pre-integrator -----------------------
207 // 1.1 Get last two measurements from buffer to determine dt
208 TimeToImuMap imuMeas;
209 imuBufferPtr->getLastTwoMeasurements(imuMeas);
210 propagatedStateTime_ = imuTimeK;
211 currentAngularVelocity_ = imuMeas.rbegin()->second.angularVelocity;
212
213 // 1.2 Update IMU Pre-integrator
214 updateImuIntegrators_(imuMeas);
215
216 // 1.3 Predict propagated state via forward integration
217 // Previous pose
218 gtsam::Pose3 T_W_I_beforeInt = W_imuPropagatedState_.pose();
219 if (graphConfigPtr_->usingBiasForPreIntegrationFlag_) {
220 W_imuPropagatedState_ =
221 imuBufferPtr->integrateNavStateFromTimestamp(imuMeas.begin()->first, imuMeas.rbegin()->first, W_imuPropagatedState_,
222 optimizedGraphState_.imuBias(), graphConfigPtr_->W_gravityVector_);
223 } else {
224 W_imuPropagatedState_ =
225 imuBufferPtr->integrateNavStateFromTimestamp(imuMeas.begin()->first, imuMeas.rbegin()->first, W_imuPropagatedState_,
226 gtsam::imuBias::ConstantBias(), graphConfigPtr_->W_gravityVector_);
227 }
228
229 // 1.4 Propagate to state in odom
230 // Delta pose
231 const gtsam::Pose3 T_W_I_afterInt = W_imuPropagatedState_.pose();
232 const gtsam::Pose3 T_Ikm1_Ik = T_W_I_beforeInt.between(T_W_I_afterInt);
233 // Pose of IMU in Odom
234 gtsam::Pose3 T_O_I = O_imuPropagatedState_.pose() * T_Ikm1_Ik;
235 // Get Gravity Alignment right also in Odom frame
236 const Eigen::Matrix3d& R_W_I = T_W_I_afterInt.rotation().matrix();
237 Eigen::Matrix3d R_O_I = T_O_I.rotation().matrix();
238 R_O_I.row(2) = R_W_I.row(2); // Copy gravity direction
239 R_O_I.row(1) = R_O_I.row(2).cross(R_O_I.row(0)).normalized();
240 R_O_I.row(0) = R_O_I.row(1).cross(R_O_I.row(2)).normalized();
241 T_O_I = gtsam::Pose3(gtsam::Rot3(R_O_I), T_O_I.translation());
242 // Velocity
243 gtsam::Vector3 O_v_O_I = T_O_I.rotation() * W_imuPropagatedState_.bodyVelocity();
244 // Update the NavState
245 O_imuPropagatedState_ = gtsam::NavState(T_O_I.rotation(), T_O_I.translation(), O_v_O_I);
246
247 // 1.5 Fill in optimized state container
248 if (optimizedGraphState_.isOptimized()) {
249 newOptimizedNavStatePtr = std::make_shared<SafeNavStateWithCovarianceAndBias>(optimizedGraphState_);
250 } else {
251 newOptimizedNavStatePtr = nullptr;
252 }
253
254 // 1.6 Transformation between World and Odom
255 T_W_O_ = Eigen::Isometry3d((W_imuPropagatedState_.pose() * O_imuPropagatedState_.pose().inverse()).matrix());
256
257 // 1.7 Return pre-integrated state
258 gtsam::NavState& T_O_Ik_nav = O_imuPropagatedState_; // Alias
259 // Assign poses and velocities
260 returnPreIntegratedNavState.update(T_W_O_, Eigen::Isometry3d(T_O_Ik_nav.pose().matrix()), T_O_Ik_nav.bodyVelocity(),
261 optimizedGraphState_.imuBias().correctGyroscope(imuMeas.rbegin()->second.angularVelocity), imuTimeK,
262 false);
263
264 // Check whether new state can be created (in case fixed lag smoother has a short window)
265 if (createNewStateFlag && imuTimeK - lastOptimizedStateTime_ > graphConfigPtr_->realTimeSmootherLag_ - WORST_CASE_OPTIMIZATION_TIME &&
266 lastOptimizedStateTime_ > 0.0) {
267 createNewStateFlag = false;
268 REGULAR_COUT << RED_START
269 << " The current measurement would fall outside of the real-time smoother lag, hence skipping the creation of a new"
270 << "IMU measurement. Not creating new state." << COLOR_END << std::endl;
271 }
272
273 // Part 2 (OPTIONAL): Create new state and add IMU factor to graph ------------------------------
274 if (createNewStateFlag) {
275 // Get new key
276 const gtsam::Key oldKey = propagatedStateKey_;
277 const gtsam::Key newKey = newPropagatedStateKey_();
278
279 // Add to time key buffer
280 timeToKeyBufferPtr_->addToBuffer(imuTimeK, newKey);
281
282 // Add IMU Factor to graph
283 gtsam::CombinedImuFactor imuFactor(gtsam::symbol_shorthand::X(oldKey), gtsam::symbol_shorthand::V(oldKey),
284 gtsam::symbol_shorthand::X(newKey), gtsam::symbol_shorthand::V(newKey),
285 gtsam::symbol_shorthand::B(oldKey), gtsam::symbol_shorthand::B(newKey), *imuStepPreintegratorPtr_);
286 addFactorToRtAndBatchGraph_<const gtsam::CombinedImuFactor*>(&imuFactor, imuTimeK, "imu");
287
288 // Add IMU values
289 gtsam::Values valuesEstimate;
290 valuesEstimate.insert(gtsam::symbol_shorthand::X(newKey), W_imuPropagatedState_.pose());
291 valuesEstimate.insert(gtsam::symbol_shorthand::V(newKey), W_imuPropagatedState_.velocity());
292 valuesEstimate.insert(gtsam::symbol_shorthand::B(newKey), optimizedGraphState_.imuBias());
293 rtGraphValuesBufferPtr_->insert(valuesEstimate);
294 if (graphConfigPtr_->useAdditionalSlowBatchSmootherFlag_) {
295 batchGraphValuesBufferPtr_->insert(valuesEstimate);
296 }
297
298 // Add timestamps
299 writeValueKeysToKeyTimeStampMap_(valuesEstimate, imuTimeK, rtGraphKeysTimestampsMapBufferPtr_);
300 if (graphConfigPtr_->useAdditionalSlowBatchSmootherFlag_) {
301 writeValueKeysToKeyTimeStampMap_(valuesEstimate, imuTimeK, batchGraphKeysTimestampsMapBufferPtr_);
302 }
303
304 // After adding this factor we can again empty the step integrator
305 // Reset IMU Step Pre-integration
306 if (graphConfigPtr_->usingBiasForPreIntegrationFlag_) {
307 imuStepPreintegratorPtr_->resetIntegrationAndSetBias(optimizedGraphState_.imuBias());
308 } else {
309 imuStepPreintegratorPtr_->resetIntegrationAndSetBias(gtsam::imuBias::ConstantBias());
310 }
311 } // End of create new state
312
313 // Potentially log real-time state to container
314 if (graphConfigPtr_->logRealTimeStateToMemoryFlag_) {
315 realTimeWorldPoseContainer_[imuTimeK] = W_imuPropagatedState_.pose();
316 realTimeOdomPoseContainer_[imuTimeK] = O_imuPropagatedState_.pose();
317 }
318
319 // Logging of latency
320 if (graphConfigPtr_->logLatencyAndUpdateDurationToMemoryFlag_) {
321 latencyEndTime_ = std::chrono::high_resolution_clock::now();
322 // Duration in seconds
323 std::chrono::duration<double> latencyDuration = latencyEndTime_ - latencyStartTime_;
324 latencyContainer_[imuTimeK] = latencyDuration.count();
325 }
326}

348 // TODO: Add to buffer and return --> still add it until we are there

350 }
351 // Too far in the future --> do not add it

357 << " ms, being larger than admissible deviation of " << 4 * 1000 * graphConfigPtr_->maxSearchDeviation_
358 << " ms. Not adding to graph." << COLOR_END << std::endl;

360 }
361 }
362 // Measurement coming from the past, but could not find suitable key
363 else { // Otherwise do not add it
364 REGULAR_COUT << RED_START << " Time deviation of " << typeid(unaryMeasurement).name() << " at key " << returnedKey << " is "
365 << 1000 * std::abs(returnedGraphTime - unaryMeasurement.timeK()) << " ms, being larger than admissible deviation of "
366 << 1000 * graphConfigPtr_->maxSearchDeviation_ << " ms. Not adding to graph." << COLOR_END << std::endl;
367 return false;
368 }
369 }
370 // Case 2: Can add immediately
371 else {
372 return true;
373 }

376// Robust Aware Between factors ------------------------------------------------------------------------------------------------------
377gtsam::Key GraphManager::addPoseBetweenFactor(const gtsam::Pose3& deltaPose, const Eigen::Matrix<double, 6, 1>& poseBetweenNoiseDensity,
378 const double timeKm1, const double timeK, const double rate,
379 const RobustNormEnum& robustNormEnum, const double robustNormConstant) {
380 // Find corresponding keys in graph
381 const double maxLidarTimestampDistance = (1.0 / rate) + (2.0 * graphConfigPtr_->maxSearchDeviation_);
382 gtsam::Key closestKeyKm1, closestKeyK;
383 double keyTimeStampDistance{0.0};
384
385 if (!findGraphKeys_(closestKeyKm1, closestKeyK, keyTimeStampDistance, maxLidarTimestampDistance, timeKm1, timeK, "pose between")) {
386 REGULAR_COUT << RED_START << " Current propagated key: " << propagatedStateKey_ << " , PoseBetween factor not added between keys "
387 << closestKeyKm1 << " and " << closestKeyK << COLOR_END << std::endl;
388 return closestKeyK;
389 }
390
391 // Scale delta pose according to timeStampDistance
392 const double scale = keyTimeStampDistance / (timeK - timeKm1);
393 if (graphConfigPtr_->verboseLevel_ > 3) {
394 REGULAR_COUT << " Scaling factor in tangent space for pose between delta pose: " << scale << std::endl;
395 }
396 gtsam::Pose3 scaledDeltaPose = gtsam::Pose3::Expmap(scale * gtsam::Pose3::Logmap(deltaPose));
397
398 // Create noise model
399 assert(poseBetweenNoiseDensity.size() == 6);
400 auto noise = gtsam::noiseModel::Diagonal::Sigmas((gtsam::Vector(poseBetweenNoiseDensity))); // rad,rad,rad,m,m,m
401 boost::shared_ptr<gtsam::noiseModel::Robust> robustErrorFunction;
402 // Pick Robust Error Function
403 switch (robustNormEnum) {
404 case RobustNormEnum::Huber:
405 robustErrorFunction = gtsam::noiseModel::Robust::Create(gtsam::noiseModel::mEstimator::Huber::Create(robustNormConstant), noise);
406 break;
407 case RobustNormEnum::Cauchy:
408 robustErrorFunction = gtsam::noiseModel::Robust::Create(gtsam::noiseModel::mEstimator::Cauchy::Create(robustNormConstant), noise);
409 break;
410 case RobustNormEnum::Tukey:
411 robustErrorFunction = gtsam::noiseModel::Robust::Create(gtsam::noiseModel::mEstimator::Tukey::Create(robustNormConstant), noise);
412 break;
413 case RobustNormEnum::GemanMcClure:
414 robustErrorFunction =
415 gtsam::noiseModel::Robust::Create(gtsam::noiseModel::mEstimator::GemanMcClure::Create(robustNormConstant), noise);
416 break;
417 }
418
419 // Create pose between factor and add it
420 gtsam::BetweenFactor<gtsam::Pose3> poseBetweenFactor;
421 if (robustNormEnum == RobustNormEnum::None) {
422 poseBetweenFactor = gtsam::BetweenFactor<gtsam::Pose3>(gtsam::symbol_shorthand::X(closestKeyKm1),
423 gtsam::symbol_shorthand::X(closestKeyK), scaledDeltaPose, noise);
424 } else {
425 poseBetweenFactor = gtsam::BetweenFactor<gtsam::Pose3>(gtsam::symbol_shorthand::X(closestKeyKm1),
426 gtsam::symbol_shorthand::X(closestKeyK), scaledDeltaPose, robustErrorFunction);
427 }
428
429 // Write to graph
430 addFactorSafelyToRtAndBatchGraph_<const gtsam::BetweenFactor<gtsam::Pose3>*>(&poseBetweenFactor, timeKm1);
431
432 // Print summary
433 if (graphConfigPtr_->verboseLevel_ > 3) {
434 REGULAR_COUT << " Current propagated key: " << propagatedStateKey_ << ", " << YELLOW_START << " PoseBetween factor added between key "
435 << closestKeyKm1 << " and key " << closestKeyK << COLOR_END << std::endl;
436 }
437
438 return closestKeyK;
439}

441gtsam::NavState GraphManager::calculateNavStateAtGeneralKey(bool& computeSuccessfulFlag,
442 const std::shared_ptr<graph_msf::OptimizerBase> optimizerPtr,
443 const gtsam::Key& generalKey, const char* callingFunctionName) {
444 // Nav State (pose and velocity)
445 gtsam::Pose3 resultPose;
446 gtsam::Vector3 resultVelocity;
447 try {
448 resultPose = optimizerPtr->calculateEstimatedPose3(gtsam::symbol_shorthand::X(generalKey)); // auto result = mainGraphPtr_->estimate();
449 resultVelocity = optimizerPtr->calculateEstimatedVelocity3(gtsam::symbol_shorthand::V(generalKey));
450 computeSuccessfulFlag = true;
451 } catch (const std::out_of_range& outOfRangeExeception) {
452 REGULAR_COUT << "Out of Range exeception while optimizing graph: " << outOfRangeExeception.what() << '\n';
453 REGULAR_COUT << RED_START
454 << " This happens if the measurement delay is larger than the graph-smootherLag, i.e. the optimized graph instances are "
455 "not connected. Increase the lag in this case."
456 << COLOR_END << std::endl;
457 REGULAR_COUT << RED_START << " CalculateNavStateAtKey called by " << callingFunctionName << COLOR_END << std::endl;
458 computeSuccessfulFlag = false;
459 }
460 return gtsam::NavState(resultPose, resultVelocity);
461}

463Eigen::Matrix<double, 6, 6> GraphManager::calculatePoseCovarianceAtKeyInWorldFrame(std::shared_ptr<graph_msf::OptimizerBase> graphPtr,
464 const gtsam::Key& graphKey,
465 const char* callingFunctionName,
466 std::optional<const gtsam::NavState> optionalNavState) {
467 // Pose Covariance in Tangent Space
468 std::chrono::high_resolution_clock::time_point start = std::chrono::high_resolution_clock::now();
469 gtsam::Matrix66 resultPoseCovarianceBodyFrame = graphPtr->calculateMarginalCovarianceMatrixAtKey(graphKey);
470 std::chrono::high_resolution_clock::time_point end = std::chrono::high_resolution_clock::now();
471 auto duration = std::chrono::duration_cast<std::chrono::microseconds>(end - start).count();
472 if (!optionalNavState.has_value() && duration > 400) {
473 std::cout << RED_START << "Time to calculate marginal covariance: " << duration << " microseconds, which is too much." << std::endl;
474 }
475 // Transform covariance from I_S_I in body frame, to W_S_W in world frame
476 // Get NavState
477 gtsam::NavState navState;
478 if (optionalNavState.has_value()) {
479 navState = optionalNavState.value();
480 } else {
481 gtsam::Pose3 resultPose = graphPtr->calculateEstimatedPose3(graphKey); // auto result = mainGraphPtr_->estimate();
482 gtsam::Point3 resultVelocity = gtsam::Point3::Identity(); // Assume Zero Velocity, but adjoint below is still fine to map to world
483 navState = gtsam::NavState(resultPose, resultVelocity);
484 }
485 // Compute adjoint matrix
486 gtsam::Matrix66 adjointMatrix = navState.pose().AdjointMap();
487 // Transform covariance
488 gtsam::Matrix66 resultPoseCovarianceWorldFrame = adjointMatrix * resultPoseCovarianceBodyFrame * adjointMatrix.transpose();
489 // Return
490 return resultPoseCovarianceWorldFrame;
491}

621 const char stateCategory = gtsam::Symbol(gtsamKey).chr();
622

624 if (framePairKeyMapIterator.second.isVariableActive()) {
625 // Case 1: All worked ------------------------------------------
626 try { // Obtain estimate and covariance from the extrinsic transformations
627 gtsam::Pose3 T_frame1_frame2;
628 gtsam::Matrix66 T_frame1_frame2_covariance = gtsam::Z_6x6;
629
630 // 6D Transformations

640 std::ignore = batchOptimizerPtr_->updateExistingValues(valuesEstimate);

653 valuesEstimate.insert(gtsamKey, point3Estimate);

712 framePairKeyMapIterator.second.getNumberStepsOptimized() > MIN_ITERATIONS_BEFORE_REMOVING_STATIC_TRANSFORM) {
713 const gtsam::Pose3& T_frame1_frame2_initial =
714 framePairKeyMapIterator.second.getApproximateTransformationBeforeOptimization(); // alias
715 const double errorTangentSpace = gtsam::Pose3::Logmap(T_frame1_frame2_initial.between(T_frame1_frame2)).norm();

726 framePairKeyMapIterator.first.second);

728 }
729 }
730 } // end: try statement
731 // Case 2: Result computation failed -----------------------------------------
732 catch (const std::out_of_range& exception) {

734 if (framePairKeyMapIterator.second.getNumberStepsOptimized() > 0) {

736 << gtsam::Symbol(gtsamKey) << ", for frame pair " << framePairKeyMapIterator.first.first << ","

744 framePairKeyMapIterator.first.second);

746 }
747 // If active but added newly but never optimized
748 else {
749 // If too old but was never optimized --> remove or deactivate

760 framePairKeyMapIterator.first.second);
761 }
762 // Otherwise keep for now and potentially print out
763 else if (graphConfigPtr_->verboseLevel_ >= 1) {
764 REGULAR_COUT

772 }
773 } // catch statement
774 } // end: if active statement
775 } // for loop over all transforms
776 }

779 {
780 // Lock

782

784 optimizedGraphState_.setIsOptimized();
785 // Update Optimized Graph State
786 optimizedGraphState_.updateNavStateAndBias(currentPropagatedKey, currentPropagatedTime, resultNavState,

790 optimizedGraphState_.updateLandmarkTransforms(resultLandmarkTransformations);

793
794 // 2. State in World: Predict from solution to obtain refined propagated state
795 if (graphConfigPtr_->usingBiasForPreIntegrationFlag_) {
796 W_imuPropagatedState_ = imuBufferPreintegratorPtr_->predict(resultNavState, optimizedGraphState_.imuBias());
797 } else {
798 W_imuPropagatedState_ = imuBufferPreintegratorPtr_->predict(resultNavState, gtsam::imuBias::ConstantBias());
799 }
800
801 // 3. State in Odom
802 // Correct rotation only for roll and pitch, keep integrated yaw
803 // gtsam::Rot3 R_O_I_rp_corrected =
804 // gtsam::Rot3::Ypr(O_imuPropagatedState_.pose().rotation().yaw(), W_imuPropagatedState_.pose().rotation().pitch(),
805 // W_imuPropagatedState_.pose().rotation().roll());
806 // TODO: Apply gravity correction
807 // gtsam::Rot3 R_O_I_rp_corrected = O_imuPropagatedState_.pose().rotation();
808 // // Rotate corrected velocity to odom frame
809 // gtsam::Vector3 O_v_O_I = R_O_I_rp_corrected * W_imuPropagatedState_.bodyVelocity();
810 // Update the NavState
811 // O_imuPropagatedState_ = gtsam::NavState(R_O_I_rp_corrected, O_imuPropagatedState_.pose().translation(), O_v_O_I);
812
813 // // 4. Transformation between World and Odom
814 // T_W_O_ = Eigen::Isometry3d((W_imuPropagatedState_.pose() * O_imuPropagatedState_.pose().inverse()).matrix());
815
816 // Update the time of the last optimized state
817 lastOptimizedStateTime_ = currentPropagatedTime;
818 } // end of locking
819
820 // Potentially log real-time state to container
821 if (graphConfigPtr_->logRealTimeStateToMemoryFlag_) {
