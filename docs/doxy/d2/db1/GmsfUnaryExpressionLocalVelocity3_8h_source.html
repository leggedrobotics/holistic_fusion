59 // i) Generate Expression for Basic IMU State in World Frame at Key
60 void generateImuStateInWorldFrameAtKey(const gtsam::Key& closestGeneralKey) override {
61 // Translation (core part)
62 gtsam::Point3_ exp_fixedFrame_v_fixedFrame_sensorFrame_ =
63 gtsam::Expression<gtsam::Vector3>(gtsam::symbol_shorthand::V(closestGeneralKey)); // W_v_W_I at this point
64
65 // Rotation (needed for the transformation to the sensor frame)
66 exp_R_fixedFrame_I_ =
67 gtsam::rotation(gtsam::Expression<gtsam::Pose3>(gtsam::symbol_shorthand::X(closestGeneralKey))); // R_W_I at this point
68
69 // Express velocity in sensor frame
70 exp_sensorFrame_v_fixedFrame_sensorFrame_ =
71 gtsam::rotate(inverseRot3(exp_R_fixedFrame_I_), exp_fixedFrame_v_fixedFrame_sensorFrame_); // I_v_W_I at this point
72 }
73
74 // iii) Transform Measurement to Core Imu Frame
75 void transformImuStateToSensorFrameState() final {
76 // Get relative translation
77 Eigen::Vector3d I_t_I_sensorFrame = T_I_sensorFrameInit_.translation();
78
79 // Angular velocity in the sensor frame
80 // TODO: Add angular velocity bias
81 gtsam::Point3_ I_w_W_I(angularVelocity_);
82
83 // Compute Velocity of the Sensor Frame with Angular Velocity
84 exp_sensorFrame_v_fixedFrame_sensorFrame_ =
85 exp_sensorFrame_v_fixedFrame_sensorFrame_ + gtsam::cross(I_w_W_I, I_t_I_sensorFrame); // I_v_W_sensorFrame at this point
86
87 // Get Rotation from Sensor Frame to Inertial Frame
88 gtsam::Rot3 R_sensorFrame_I = gtsam::Rot3(T_I_sensorFrameInit_.rotation().inverse());
89
90 // Convert to Sensor Frame
91 exp_sensorFrame_v_fixedFrame_sensorFrame_ =
92 gtsam::rotate(R_sensorFrame_I, exp_sensorFrame_v_fixedFrame_sensorFrame_); // sensorFrame_v_fixedFrame_sensorFrame
93 }
