18 explicit OptimizerIsam2Batch(const std::shared_ptr<GraphConfig> graphConfigPtr) : OptimizerIsam2(graphConfigPtr) {
19 // Initialize Slow Bundle Adjustement Smoother (if desired) -----------------------------------------------
20 if (graphConfigPtr_->useAdditionalSlowBatchSmootherFlag_) {
21 // Batch --> slowBatchSmoother: Define whether to use Cholesky factorization
22 if (graphConfigPtr_->slowBatchSmootherUseCholeskyFactorizationFlag_) {
23 isam2Params_.factorization = gtsam::ISAM2Params::CHOLESKY;
24 std::cout << "Using Cholesky factorization for slow batch smoother (ISAM2)." << std::endl;
25 } else {
26 isam2Params_.factorization = gtsam::ISAM2Params::QR;
27 std::cout << "Using QR factorization for slow batch smoother (ISAM2)." << std::endl;
28 }
29 // Initialize Slow Bundle Adjustement Smoother
30 batchSmootherPtr_ = std::make_shared<gtsam::ISAM2>(isam2Params_);
31 batchSmootherPtr_->params().print("GraphMSF: Factor Graph Parameters of Slow Batch Optimization Graph.");
32 }
33 }
34 ~OptimizerIsam2Batch() = default;
35
36 // Empty update call --> do nothing
37 bool update() override { return true; }
38
39 // Add to graph without running optimization
40 bool update(const gtsam::NonlinearFactorGraph& newGraphFactors, const gtsam::Values& newGraphValues,
41 const std::map<gtsam::Key, double>& newGraphKeysTimeStampMap, const int depth = 0) override {
42 // Add Bundle Adjustement Smoother factors to batch without running optimization
43 containerBatchSmootherFactors_.add(newGraphFactors);
44 containerBatchSmootherValues_.insert(newGraphValues);
45 // Add to keyTimestampMap
46 batchSmootherKeyTimestampMap_.insert(newGraphKeysTimeStampMap.begin(), newGraphKeysTimeStampMap.end());
47 return true;
48 }

50 // Optimize
51 void optimize(int maxIterations) override {
52 // Print
53 std::cout << YELLOW_START << "GraphMSF: OptimizerIsam2Batch" << GREEN_START << " Optimizing slow batch smoother." << COLOR_END
54 << std::endl;
55
56 // Optimize
57 batchSmootherIsam2Result_ = batchSmootherPtr_->update(containerBatchSmootherFactors_, containerBatchSmootherValues_);
58 // Reset containers
59 containerBatchSmootherFactors_.resize(0);
60 containerBatchSmootherValues_.clear();
61
62 // Get optimized result
63 batchSmootherOptimizedResult_ = batchSmootherPtr_->calculateEstimate();
64
65 // Flag
66 optimizedAtLeastOnceFlag_ = true;
67 }
68
69 // Optimize bundle adjustement smoother (if desired)
70 const gtsam::Values& getAllOptimizedStates() override {
71 // Check
72 if (!optimizedAtLeastOnceFlag_) {
73 throw std::runtime_error("GraphMSF: OptimizerIsam2Batch: No optimization has been performed yet.");
74 }
75
76 // Return result

78 }

106 return batchSmootherPtr_->calculateEstimate<gtsam::Vector>(key);
107 }
108 // Marginal Covariance
109 gtsam::Matrix calculateMarginalCovarianceMatrixAtKey(const gtsam::Key& key) override {
110 if (!optimizedAtLeastOnceFlag_) {
111 throw std::runtime_error("GraphMSF: OptimizerIsam2Batch: No optimization has been performed yet.");
112 }
113 return batchSmootherPtr_->marginalCovariance(key);
114 }
115
116 private:

122 // Result and keyTimestampMap
123 gtsam::ISAM2Result batchSmootherIsam2Result_;
124 gtsam::Values batchSmootherOptimizedResult_;
125 std::map<gtsam::Key, double> batchSmootherKeyTimestampMap_;
126};
