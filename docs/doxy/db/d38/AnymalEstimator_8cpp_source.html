131// Offline Optimization Service
132bool AnymalEstimator::srvOfflineSmootherOptimizeCallback(graph_msf_ros_msgs::OfflineOptimizationTrigger::Request& req,
133 graph_msf_ros_msgs::OfflineOptimizationTrigger::Response& res) {
134 // Call super class
135 bool success = graph_msf::GraphMsfRos::srvOfflineSmootherOptimizeCallback(req, res);
136
137 // if GNSS is used at all, also log the reference frame
138 if (gnssCallbackCounter_ > 0) {
139 // Get the reference coordinates of the GNSS handler
140 const double referenceLatitude = gnssHandlerPtr_->getGnssReferenceLatitude();
141 const double referenceLongitude = gnssHandlerPtr_->getGnssReferenceLongitude();
142 const double referenceAltitude = gnssHandlerPtr_->getGnssReferenceAltitude();
143 const double referenceHeading = gnssHandlerPtr_->getGnssReferenceHeading();
144 REGULAR_COUT << " GNSS reference (latitude, longitude, altitude, heading): " << referenceLatitude << ", " << referenceLongitude << ", "
145 << referenceAltitude << ", " << referenceHeading << std::endl;
146
147 // File streams
148 std::map<std::string, std::ofstream> fileStreams;
149
150 // Write T_totalStation_totalStationOld_ to file ----------------------------
151 // Get time string by finding latest directory in optimizationResultLoggingPath
152 std::string timeString = graph_msf::getLatestSubdirectory(optimizationResultLoggingPath);
153 if (timeString.empty()) {
154 REGULAR_COUT << " Could not find latest directory in " << optimizationResultLoggingPath << std::endl;
155 return false;
156 } else {
157 REGULAR_COUT << " Found latest directory in " << optimizationResultLoggingPath << ": " << timeString << std::endl;
158 }
159 // Create file stream
160 std::string transformIdentifier = "gnss_reference_lat_lon_alt";
161 graph_msf::createLatLonAltCsvFileStream(fileStreams, optimizationResultLoggingPath, transformIdentifier, timeString);
162 // Add to file
163 graph_msf::writeLatLonAltToCsvFile(fileStreams, Eigen::Vector3d(referenceLatitude, referenceLongitude, referenceAltitude),
164 transformIdentifier, ros::Time::now().toSec());
165 REGULAR_COUT << " Wrote GNSS reference (latitude, longitude, altitude) coordinates to file." << std::endl;
166
167 // If optional GNSS ENU coordinates are logged, write them to pose file
168 if constexpr (logOptionalGnssEnuCoordinatesFlag_) {
169 transformIdentifier = "gnss_enu_trajectory";
170 graph_msf::createPose3CsvFileStream(fileStreams, optimizationResultLoggingPath, transformIdentifier, timeString, false);
171 // Iterate over all optional GNSS ENU coordinates
172 int index = 0;
173 for (const auto& optionalGnssEnuCoordinates : optionalGnssEnuCoordinatesTrajectory_) {
174 // Create Isometry3d with identity rotation
175 Eigen::Isometry3d optionalGnssEnuCoordinatesIsometry = Eigen::Isometry3d::Identity();
176 optionalGnssEnuCoordinatesIsometry.translation() = optionalGnssEnuCoordinates;
177 // Add to file
178 graph_msf::writePose3ToCsvFile(fileStreams, optionalGnssEnuCoordinatesIsometry, transformIdentifier,
179 optionalGnssEnuCoordinatesTimeStamps_[index], false);
180 index++;
181 }
182 }
183 }
184
185 // Return
186 return success;
187}

189// Priority: 1
190void AnymalEstimator::gnssUnaryCallback_(const sensor_msgs::NavSatFix::ConstPtr& gnssMsgPtr) {
191 // Counter
192 ++gnssCallbackCounter_;
193
194 // Convert to Eigen
195 Eigen::Vector3d gnssCoord = Eigen::Vector3d(gnssMsgPtr->latitude, gnssMsgPtr->longitude, gnssMsgPtr->altitude);
196 Eigen::Vector3d estStdDevXYZ(sqrt(gnssMsgPtr->position_covariance[0]), sqrt(gnssMsgPtr->position_covariance[4]),
197 sqrt(gnssMsgPtr->position_covariance[8]));
198
199 // Initialize GNSS Handler
200 if (gnssCallbackCounter_ < NUM_GNSS_CALLBACKS_UNTIL_START) { // Accumulate measurements
201 // Wait until measurements got accumulated
202 accumulatedGnssCoordinates_ += gnssCoord;
203 if ((gnssCallbackCounter_ % 10) == 0) {
204 REGULAR_COUT << " NOT ENOUGH GNSS MESSAGES ARRIVED!" << std::endl;
205 }
206 return;
207 } else if (gnssCallbackCounter_ == NUM_GNSS_CALLBACKS_UNTIL_START) { // Initialize GNSS Handler
208 gnssHandlerPtr_->initHandler(accumulatedGnssCoordinates_ / NUM_GNSS_CALLBACKS_UNTIL_START);
209 REGULAR_COUT << " GNSS Handler initialized." << std::endl;
210 return;
211 }
212
213 // Convert to Cartesian Coordinates
214 Eigen::Vector3d W_t_W_Gnss;
215 // gnssHandlerPtr_->convertNavSatToPosition(gnssCoord, W_t_W_Gnss);
216 gnssHandlerPtr_->convertNavSatToPositionLV03(gnssCoord, W_t_W_Gnss);
217 std::string fixedFrame = staticTransformsPtr_->getWorldFrame(); // Alias
218 // fixedFrame = "east_north_up";
219
220 // // For Debugging: Add Gaussian Noise with 0.1m std deviation
221 // // Random number generator
222 // std::default_random_engine generator(std::chrono::system_clock::now().time_since_epoch().count());
223 // // Add noise
224 // W_t_W_Gnss +=
225 // Eigen::Vector3d(std::normal_distribution<double>(0.0, 0.1)(generator), std::normal_distribution<double>(0.0, 0.1)(generator),
226 // std::normal_distribution<double>(0.0, 0.1)(generator));
227 // // Add to assumed standard deviation (\sigma_{GNSS+noise} = \sqrt{\sigma_{GNSS}^2 + \sigma_{noise}^2})
228 // for (int i = 0; i < 3; ++i) {
229 // estStdDevXYZ(i) = sqrt(estStdDevXYZ(i) * estStdDevXYZ(i) + 0.1 * 0.1);
230 // }
231
232 // Initial world yaw initialization options
233 // Case 1: Initialization
234 if (!areYawAndPositionInited()) {
235 // a: Default
236 double initYaw_W_Base{0.0}; // Default is 0 yaw
237 // b: From file
238 if (gnssHandlerPtr_->getUseYawInitialGuessFromFile()) {
239 initYaw_W_Base = gnssHandlerPtr_->getGlobalYawDegFromFile() / 180.0 * M_PI;
240 }
241 // c: From alignment
242 else if (gnssHandlerPtr_->getUseYawInitialGuessFromAlignment()) {
243 // Adding the GNSS measurement
244 trajectoryAlignmentHandler_->addR3Position(W_t_W_Gnss, gnssMsgPtr->header.stamp.toSec());
245 // In radians
246 Eigen::Isometry3d T_W_Base = Eigen::Isometry3d::Identity();
247 if (!(trajectoryAlignmentHandler_->alignTrajectories(initYaw_W_Base, T_W_Base))) {
248 if (gnssCallbackCounter_ % 10 == 0) {
249 REGULAR_COUT << YELLOW_START << "Trajectory alignment not ready. Waiting for more motion." << COLOR_END << std::endl;
250 }
251 return;
252 }
253 REGULAR_COUT << GREEN_START << "Trajectory Alignment Successful. Obtained Yaw Value of T_W_Base (deg): " << COLOR_END
254 << 180.0 * initYaw_W_Base / M_PI << std::endl;
255 }
256
257 // Actual Initialization
258 if (this->initYawAndPositionInWorld(initYaw_W_Base, W_t_W_Gnss,
259 dynamic_cast<AnymalStaticTransforms*>(staticTransformsPtr_.get())->getBaseLinkFrame(),
260 dynamic_cast<AnymalStaticTransforms*>(staticTransformsPtr_.get())->getGnssFrame())) {
261 REGULAR_COUT << GREEN_START << " GNSS initialization of yaw and position successful." << std::endl;
262
263 } else {
264 REGULAR_COUT << RED_START << " GNSS initialization of yaw and position failed." << std::endl;
265 }
266 } else { // Case 2: Already initialized --> Unary factor
267 const std::string& gnssFrameName = dynamic_cast<AnymalStaticTransforms*>(staticTransformsPtr_.get())->getGnssFrame(); // Alias
268 // Measurement
269 graph_msf::UnaryMeasurementXDAbsolute<Eigen::Vector3d, 3> meas_W_t_W_Gnss(
270 "GnssPosition", int(gnssRate_), gnssFrameName, gnssFrameName + sensorFrameCorrectedNameId, graph_msf::RobustNorm::None(),
271 gnssMsgPtr->header.stamp.toSec(), gnssPositionOutlierThreshold_, W_t_W_Gnss, estStdDevXYZ, fixedFrame,
272 staticTransformsPtr_->getWorldFrame());
273 this->addUnaryPosition3AbsoluteMeasurement(meas_W_t_W_Gnss);
274 }
275
276 // Add _gmsf to the frame
277 if (fixedFrame != staticTransformsPtr_->getWorldFrame()) {
278 fixedFrame += referenceFrameAlignedNameId;
279 }
280
282 addToPathMsg(measGnss_worldGnssPathPtr_, fixedFrame, gnssMsgPtr->header.stamp, W_t_W_Gnss, graphConfigPtr_->imuBufferLength_ * 4);
284 pubMeasWorldGnssPath_.publish(measGnss_worldGnssPathPtr_);
285
286 // Potentially log the GNSS ENU coordinates
287 if (logOptionalGnssEnuCoordinatesFlag_) {
288 optionalGnssEnuCoordinatesTimeStamps_.push_back(gnssMsgPtr->header.stamp.toSec());
289 optionalGnssEnuCoordinatesTrajectory_.push_back(W_t_W_Gnss);
290 }
291}

293// Priority: 2
294void AnymalEstimator::lidarUnaryCallback_(const nav_msgs::Odometry::ConstPtr& odomLidarPtr) {
295 // Counter
296 ++lidarUnaryCallbackCounter_;
297
298 // Prepare Data
299 Eigen::Isometry3d lio_T_M_Lk = Eigen::Isometry3d::Identity();
300 graph_msf::odomMsgToEigen(*odomLidarPtr, lio_T_M_Lk.matrix());
301 double lidarUnaryTimeK = odomLidarPtr->header.stamp.toSec();
302
303 if (useGnssUnaryFlag_ && gnssHandlerPtr_->getUseYawInitialGuessFromAlignment()) {
304 trajectoryAlignmentHandler_->addSe3Position(lio_T_M_Lk.translation(), lidarUnaryTimeK);
305 }
306
307 // Measurement
308 const std::string& lioOdomFrameName = dynamic_cast<AnymalStaticTransforms*>(staticTransformsPtr_.get())->getLioOdometryFrame(); // alias
309 graph_msf::UnaryMeasurementXDAbsolute<Eigen::Isometry3d, 6> unary6DMeasurement(
310 "Lidar_unary_6D", int(lioOdometryRate_), lioOdomFrameName, lioOdomFrameName + sensorFrameCorrectedNameId,
311 graph_msf::RobustNorm::None(), lidarUnaryTimeK, 1.0, lio_T_M_Lk, lioPoseUnaryNoise_, odomLidarPtr->header.frame_id,
312 staticTransformsPtr_->getWorldFrame(), initialSe3AlignmentNoise_, lioSe3AlignmentRandomWalk_);
313
314 if (lidarUnaryCallbackCounter_ <= 2) {
315 return;
316 } else if (!areYawAndPositionInited()) { // Initializing if no GNSS
317 if (!useGnssUnaryFlag_) {
318 REGULAR_COUT << GREEN_START << " LiDAR odometry callback is setting global yaw, as it was not set so far." << COLOR_END << std::endl;
319 this->initYawAndPosition(unary6DMeasurement);
320 }
321 } else { // Already initialized --> unary factor
322 this->addUnaryPose3AbsoluteMeasurement(unary6DMeasurement);
323 }
324
325 // Visualization ----------------------------
326 // Add to path message
327 addToPathMsg(measLio_mapLidarPathPtr_, odomLidarPtr->header.frame_id + referenceFrameAlignedNameId, odomLidarPtr->header.stamp,
328 lio_T_M_Lk.translation(), graphConfigPtr_->imuBufferLength_ * 4);
329
330 // Publish Path
331 pubMeasMapLioPath_.publish(measLio_mapLidarPathPtr_);
332}

334// Priority: 3
335void AnymalEstimator::lidarBetweenCallback_(const nav_msgs::Odometry::ConstPtr& odomLidarPtr) {
336 if (!areRollAndPitchInited()) {
337 return;
338 }
339
340 // Counter
341 ++lidarBetweenCallbackCounter_;
342
343 // Convert
344 Eigen::Isometry3d lio_T_M_Lk = Eigen::Isometry3d::Identity();
345 graph_msf::odomMsgToEigen(*odomLidarPtr, lio_T_M_Lk.matrix());
346 // Get the time
347 double lidarBetweenTimeK = odomLidarPtr->header.stamp.toSec();
348
349 // At start
350 if (lidarBetweenCallbackCounter_ == 0) {
351 lio_T_M_Lkm1_ = lio_T_M_Lk;
352 lidarBetweenTimeKm1_ = lidarBetweenTimeK;
353 }
354
355 // Add to trajectory aligner if needed.
356 if (useGnssUnaryFlag_ && gnssHandlerPtr_->getUseYawInitialGuessFromAlignment()) {
357 trajectoryAlignmentHandler_->addSe3Position(lio_T_M_Lk.translation(), lidarBetweenTimeK);
358 }
359
360 // Frame Name
361 const std::string& lioOdomFrameName = dynamic_cast<AnymalStaticTransforms*>(staticTransformsPtr_.get())->getLioOdometryFrame(); // alias
362
363 // State Machine
364 if (lidarBetweenCallbackCounter_ <= 2) {
365 return;
366 } else if (!areYawAndPositionInited()) { // Initializing
367 if (!useGnssUnaryFlag_ && !useLioUnaryFlag_) {
368 // Measurement
369 graph_msf::UnaryMeasurementXD<Eigen::Isometry3d, 6> unary6DMeasurement(
370 "Lidar_unary_6D", int(lioOdometryRate_), lioOdomFrameName, lioOdomFrameName + sensorFrameCorrectedNameId,
371 graph_msf::RobustNorm::None(), lidarBetweenTimeK, 1.0, lio_T_M_Lk, lioPoseUnaryNoise_);
372 // Add to graph
373 REGULAR_COUT << GREEN_START << " LiDAR odometry callback is setting global yaw, as it was not set so far." << COLOR_END << std::endl;
374 this->initYawAndPosition(unary6DMeasurement);
375 }
376 } else { // Already initialized --> Between factor
377 // Compute Delta
378 const Eigen::Isometry3d T_Lkm1_Lk = lio_T_M_Lkm1_.inverse() * lio_T_M_Lk;
379 // Create measurement
380 graph_msf::BinaryMeasurementXD<Eigen::Isometry3d, 6> delta6DMeasurement(
381 "Lidar_between_6D", int(lioOdometryRate_), lioOdomFrameName, lioOdomFrameName + sensorFrameCorrectedNameId,
382 graph_msf::RobustNorm::None(), lidarBetweenTimeKm1_, lidarBetweenTimeK, T_Lkm1_Lk, lioPoseUnaryNoise_);
383 // Add to graph
384 this->addBinaryPose3Measurement(delta6DMeasurement);
385 }
386 // Provide for next iteration
387 lio_T_M_Lkm1_ = lio_T_M_Lk;
388 lidarBetweenTimeKm1_ = lidarBetweenTimeK;
389
390 // Visualization ----------------------------
391 // Add to path message
392 addToPathMsg(measLio_mapLidarPathPtr_, staticTransformsPtr_->getWorldFrame(), odomLidarPtr->header.stamp, lio_T_M_Lk.translation(),
393 graphConfigPtr_->imuBufferLength_ * 4);
394
395 // Publish Path
396 pubMeasMapLioPath_.publish(measLio_mapLidarPathPtr_);
397}

400void AnymalEstimator::leggedBetweenCallback_(const geometry_msgs::PoseWithCovarianceStamped::ConstPtr& leggedOdometryPoseKPtr) {
401 if (!areRollAndPitchInited()) {
402 return;
403 }
404
405 // Counter
406 ++leggedOdometryPoseCallbackCounter_;
407
408 // Eigen Type
409 Eigen::Isometry3d T_O_Bl_k = Eigen::Isometry3d::Identity();
410 graph_msf::geometryPoseToEigen(*leggedOdometryPoseKPtr, T_O_Bl_k.matrix());
411 double legOdometryTimeK = leggedOdometryPoseKPtr->header.stamp.toSec();
412
413 // At start
414 if (leggedOdometryPoseCallbackCounter_ == 0) {
415 T_O_Bl_km1_ = T_O_Bl_k;
416 legOdometryTimeKm1_ = legOdometryTimeK;
417 return;
418 }
419
420 // Frame Name
421 const std::string& leggedOdometryFrameName =
422 dynamic_cast<AnymalStaticTransforms*>(staticTransformsPtr_.get())->getLeggedOdometryFrame(); // alias
423
424 // State Machine
425 if (!areYawAndPositionInited()) {
426 if (!useGnssUnaryFlag_ && !useLioUnaryFlag_ && !useLioBetweenFlag_) {
427 // Measurement
428 graph_msf::UnaryMeasurementXD<Eigen::Isometry3d, 6> unary6DMeasurement(
429 "Leg_odometry_6D", int(leggedOdometryBetweenRate_), leggedOdometryFrameName, leggedOdometryFrameName + sensorFrameCorrectedNameId,
430 graph_msf::RobustNorm::None(), legOdometryTimeK, 1.0, T_O_Bl_k, legPoseBetweenNoise_);
431 // Add to graph
432 REGULAR_COUT << GREEN_START << " Legged odometry between callback is setting global yaw, as it was not set so far." << COLOR_END
433 << std::endl;
434 this->initYawAndPosition(unary6DMeasurement);
435 }
436 } else {
437 // Only add every 40th measurement
438 int measurementRate = static_cast<int>(leggedOdometryBetweenRate_) / leggedOdometryPoseDownsampleFactor_;
439 // Check
440 if ((leggedOdometryPoseCallbackCounter_ % leggedOdometryPoseDownsampleFactor_) == 0) {
441 // Compute Delta
442 const Eigen::Isometry3d T_Bkm1_Bk = T_O_Bl_km1_.inverse() * T_O_Bl_k;
443 // Create measurement
444 graph_msf::BinaryMeasurementXD<Eigen::Isometry3d, 6> delta6DMeasurement(
445 "Leg_odometry_6D", measurementRate, leggedOdometryFrameName, leggedOdometryFrameName + sensorFrameCorrectedNameId,
446 graph_msf::RobustNorm::None(), legOdometryTimeKm1_, legOdometryTimeK, T_Bkm1_Bk, legPoseBetweenNoise_);
447 // Add to graph
448 this->addBinaryPose3Measurement(delta6DMeasurement);
449
450 // Prepare for next iteration
451 T_O_Bl_km1_ = T_O_Bl_k;
452 legOdometryTimeKm1_ = legOdometryTimeK;
453 }
454 }
455}

465
466 if (!areYawAndPositionInited()) {
467 if (!useGnssUnaryFlag_ && !useLioUnaryFlag_ && !useLioBetweenFlag_ && !useLeggedBetweenFlag_) {
468 // Measurement
469 graph_msf::UnaryMeasurementXD<Eigen::Isometry3d, 6> unary6DMeasurement(
470 "Leg_odometry_6D", int(leggedOdometryVelocityRate_),
471 dynamic_cast<AnymalStaticTransforms*>(staticTransformsPtr_.get())->getLeggedOdometryFrame(),
472 dynamic_cast<AnymalStaticTransforms*>(staticTransformsPtr_.get())->getLeggedOdometryFrame() + sensorFrameCorrectedNameId,

477 << std::endl;
478 this->initYawAndPosition(unary6DMeasurement);
479 }
480 } else {
481 // Only add every nth measurement
482 int measurementRate = static_cast<int>(leggedOdometryVelocityRate_) / leggedOdometryVelocityDownsampleFactor_;

484 // Check

493
494 // Create the unary measurement

496 "LegVelocityUnary", measurementRate, leggedOdometryFrameName, leggedOdometryFrameName + sensorFrameCorrectedNameId,
497 graph_msf::RobustNorm::None(), leggedOdometryKPtr->header.stamp.toSec(), 1.0, legVelocity, legVelocityUnaryNoise_);
498
499 // Add to graph

505void AnymalEstimator::leggedKinematicsCallback_(const graph_msf_anymal_msgs::AnymalState::ConstPtr& anymalStatePtr) {
506 if (!areRollAndPitchInited()) {
507 return;
508 }
509
510 // Counter
511 ++leggedKinematicsCallbackCounter_;
512
513 if (!areYawAndPositionInited()) {
514 if (!useGnssUnaryFlag_ && !useLioUnaryFlag_ && !useLioBetweenFlag_ && !useLeggedBetweenFlag_ && !useLeggedVelocityUnaryFlag_) {
515 // Measurement
516 graph_msf::UnaryMeasurementXD<Eigen::Isometry3d, 6> unary6DMeasurement(
517 "Leg_odometry_6D", int(leggedKinematicsRate_),
518 dynamic_cast<AnymalStaticTransforms*>(staticTransformsPtr_.get())->getLeggedOdometryFrame(),
519 dynamic_cast<AnymalStaticTransforms*>(staticTransformsPtr_.get())->getLeggedOdometryFrame() + sensorFrameCorrectedNameId,
520 graph_msf::RobustNorm::None(), anymalStatePtr->header.stamp.toSec(), 1.0, Eigen::Isometry3d::Identity(),
521 Eigen::Matrix<double, 6, 1>::Identity());
522 // Add to graph
523 REGULAR_COUT << GREEN_START << " Legged kinematics callback is setting global yaw, as it was not set so far." << COLOR_END
524 << std::endl;
525 this->initYawAndPosition(unary6DMeasurement);
526 }
527 }
528 // Normal Operation
529 else {
530 // Only add every nth measurement
531 int measurementRate = static_cast<int>(leggedKinematicsRate_) / leggedKinematicsDownsampleFactor_;
532
533 // Check whether it is time to add a measurement
534 if ((leggedKinematicsCallbackCounter_ % leggedKinematicsDownsampleFactor_) == 0) {
535 // Alias
536 const std::string& leggedOdometryFrameName =
537 dynamic_cast<AnymalStaticTransforms*>(staticTransformsPtr_.get())->getLeggedOdometryFrame();
538
539 // Pose of the base in odom frame
540 Eigen::Isometry3d T_O_B = Eigen::Isometry3d::Identity();
541 graph_msf::geometryPoseToEigen(anymalStatePtr->pose.pose, T_O_B.matrix());
542
543 // Create the unary measurement for each foot (loop unrolling with constexpr arrays) ----------------------------
544 visualization_msgs::MarkerArray footContactMarkers = visualization_msgs::MarkerArray();
545 // Loop
546 for (int legIndex = 0; legIndex < legNames_.size(); ++legIndex) {
547 // Get contact state
548 bool footInContact = anymalStatePtr->contacts[legIndex].state;
549 // Foot name as specified in the message
550 const std::string legName = anymalStatePtr->contacts[legIndex].name;
551
552 // Check whether leg is in contact ----------------------------
553 if (footInContact) {
554 // Increase debounce counter
555 ++legInContactForNSteps_[legIndex];
556
557 // Check whether leg was at least N times in contact to count it as in contact --> debouncing
558 if (legInContactForNSteps_[legIndex] >= legInContactDebounceThreshold_) {
559 // If it was not in contact before
560 if (legInContactForNSteps_[legIndex] == legInContactDebounceThreshold_) {
561 // Increase contact counter
562 ++legTotalContactsCounter_[legIndex];
563 // Print
564 if (graphConfigPtr_->verboseLevel_ > 1) {
565 REGULAR_COUT << " Leg " << legName << " came back into contact for the " << legTotalContactsCounter_[legIndex] << ". time."
566 << std::endl;
567 }
568 }
569
570 // Add measurement ----------------------------
571 // Position of foot in legged odometry frame
572 Eigen::Vector3d O_t_O_foot =
573 Eigen::Vector3d(anymalStatePtr->contacts[legIndex].position.x, anymalStatePtr->contacts[legIndex].position.y,
574 anymalStatePtr->contacts[legIndex].position.z);
575 Eigen::Vector3d B_t_B_foot = T_O_B.inverse() * O_t_O_foot;
576
577 // Create the unary measurement with contact counter
578 std::string legIdentifier = legName;
579 graph_msf::UnaryMeasurementXDLandmark<Eigen::Vector3d, 3> footContactPositionMeasurement(
580 legIdentifier, measurementRate, leggedOdometryFrameName, leggedOdometryFrameName + sensorFrameCorrectedNameId,
581 graph_msf::RobustNorm::None(), anymalStatePtr->header.stamp.toSec(), 1.0, B_t_B_foot, // graph_msf::RobustNorm::Huber(1)
582 legKinematicsFootPositionUnaryNoise_, staticTransformsPtr_->getWorldFrame());
583
584 // Add to graph
585 this->addUnaryPosition3LandmarkMeasurement(footContactPositionMeasurement, legTotalContactsCounter_[legIndex]);
586
587 // Visualize foot contact in RViz
588 visualization_msgs::Marker footContactMarker;
589 createContactMarker(leggedOdometryFrameName, anymalStatePtr->header.stamp, B_t_B_foot, "foot_contact", legIndex,
590 Eigen::Vector3d(0.0, 1.0, 0.0), footContactMarker);
591 footContactMarkers.markers.push_back(footContactMarker);
592 }
593 }
594 // Leg not in contact ----------------------------
595 else if (legInContactForNSteps_[legIndex] > 0) {
596 // Set to 0
597 legInContactForNSteps_[legIndex] = 0;
598 // Print
599 if (graphConfigPtr_->verboseLevel_ > 1) {
600 REGULAR_COUT << " Leg " << legName << " lost contact. " << std::endl;
601 }
602 // Remove marker if it lost contact
603 visualization_msgs::Marker deleteMarker;
604 deleteMarker.header.frame_id = leggedOdometryFrameName;
605 deleteMarker.header.stamp = anymalStatePtr->header.stamp;
606 deleteMarker.ns = "foot_contact";
607 deleteMarker.id = legIndex;
608 deleteMarker.action = visualization_msgs::Marker::DELETE;
609 footContactMarkers.markers.push_back(deleteMarker);
610 }
611 }
612 // Publish foot contact markers
613 if (!footContactMarkers.markers.empty() && pubFootContactMarkers_.getNumSubscribers() > 0) {
614 pubFootContactMarkers_.publish(footContactMarkers);
615 }
616 }
617 }
