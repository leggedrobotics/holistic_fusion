38bool TimeGraphKeyBuffer::getClosestKeyAndTimestamp(double& tInGraph, gtsam::Key& key, const std::string& callingName,
39 const double maxSearchDeviation, const double tK) {
40 std::_Rb_tree_iterator<std::pair<const double, gtsam::Key>> upperIterator;
41 {
42 // Read from IMU buffer --> acquire mutex
43 const std::lock_guard<std::mutex> writeInBufferLock(writeInBufferMutex_);
44 upperIterator = timeToKeyBuffer_.upper_bound(tK);
45 }
46
47 // Empty buffer
48 if (timeToKeyBuffer_.empty()) {
49 std::cerr << YELLOW_START << "GMsf-TimeKeyBuffer " << RED_START << "called from " << callingName << ": Buffer is empty!" << COLOR_END
50 << std::endl;
51 return false;
52 }
53
54 auto lowerIterator = upperIterator;
55 --lowerIterator;
56
57 // Keep key which is closer to tLidar
58 tInGraph = std::abs(tK - lowerIterator->first) < std::abs(upperIterator->first - tK) ? lowerIterator->first : upperIterator->first;
59 key = std::abs(tK - lowerIterator->first) < std::abs(upperIterator->first - tK) ? lowerIterator->second : upperIterator->second;
60 double timeDeviation = tInGraph - tK;
61
62 if (verboseLevel_ >= 2) {
63 std::cout << YELLOW_START << "GMsf-TimeKeyBuffer" << COLOR_END << " " << callingName << std::setprecision(14)
64 << " searched time step: " << tK << std::endl;
65 std::cout << YELLOW_START << "GMsf-TimeKeyBuffer" << COLOR_END << " " << callingName << std::setprecision(14)
66 << " found time step: " << tInGraph << " at key " << key << std::endl;
67 std::cout << YELLOW_START << "GMsf-TimeKeyBuffer" << COLOR_END << " Time Deviation (t_graph-t_request): " << 1000 * timeDeviation
68 << " ms" << std::endl;
69 std::cout << YELLOW_START << "GMsf-TimeKeyBuffer" << COLOR_END << " Latest IMU timestamp: " << tLatestInBuffer_
70 << ", hence absolut delay of measurement is " << 1000 * (tLatestInBuffer_ - tK) << "ms." << std::endl;
71 }
72
73 // Check for error and warn user
74 if (std::abs(timeDeviation) > maxSearchDeviation) {
75 return false;
76 }
