22// Explanation:
23// 1) global variables are constant along the entire operation, hence they are not removed and will get the same gtsam key upon return
24// 2) reference frames can change, but are modelled according to random walk, hence they are only removed once a successor is initialized
25// 3) landmarks are initialized and removed as needed, hence they are removed once they are not needed anymore
26// Frames that are not needed but also not removed are set to be "inactive"
27

41 // Cleanup
42 bool removeTransform(const std::string& frame1, const std::string& frame2,
43 DynamicFactorGraphStateKey<GTSAM_DYNAMIC_STATE_TYPE>& returnRemovedKey) override {
44 // Removed main element
45 bool removedTransformFlag =
46 TransformsDictionary<DynamicFactorGraphStateKey<GTSAM_DYNAMIC_STATE_TYPE>>::removeTransform(frame1, frame2, returnRemovedKey);
47 // If found, we also remove the key from the key-to-frame pair map
48 if (removedTransformFlag) {
49 auto keyFramePairMapIterator = transformGtsamKeyToFramePairMap_.find(returnRemovedKey.key());
50 // Still in map --> remove
51 if (keyFramePairMapIterator != transformGtsamKeyToFramePairMap_.end()) {
52 // transformGtsamKeyToFramePairMap_.erase(keyFramePairMapIterator);
53 // TODO: Remove if offline optimization is disabled, keep otherwise
54 }
55 // Not in map --> throw error
56 else {
57 REGULAR_COUT << RED_START << " Key " << gtsam::Symbol(returnRemovedKey.key()) << " not found in map." << COLOR_END << std::endl;
58 throw std::runtime_error("Key not found in map.");
59 }
60 }
61 return removedTransformFlag;
62 }

64 // Same but with only two arguments
65 bool removeOrDeactivateTransform(const std::string& frame1, const std::string& frame2) {
66 // Check if key is in map
67 const bool isFramePairInDictionary =
68 TransformsDictionary<DynamicFactorGraphStateKey<GTSAM_DYNAMIC_STATE_TYPE>>::isFramePairInDictionary(frame1, frame2);
69
70 // Case: Frame pair is in dictionary
71 if (isFramePairInDictionary) {
72 // Retrieve key and variable type
73 DynamicFactorGraphStateKey<GTSAM_DYNAMIC_STATE_TYPE>& keyToRemoveOrDeactivate =
74 TransformsDictionary<DynamicFactorGraphStateKey<GTSAM_DYNAMIC_STATE_TYPE>>::lv_T_frame1_frame2(frame1, frame2);
75 const DynamicVariableTypeEnum& variableTypeEnum = keyToRemoveOrDeactivate.getVariableTypeEnum();
76
77 // Three different cases
78 switch (variableTypeEnum) {
79 // Case 1: Global Measurement --> Never remove, only deactivate, as it might come back
80 case DynamicVariableTypeEnum::Global:
81 // Currently active, so we can deactivate
82 if (keyToRemoveOrDeactivate.isVariableActive()) {
83 keyToRemoveOrDeactivate.deactivateVariable();
84 return true;
85 }
86 // Already deactivated, so we do not need to do anything
87 else {
88 return false;
89 }
90 // Case 2: Reference Frame --> Do not remove for now, as we will need it for creating the random walk
91 case DynamicVariableTypeEnum::RefFrame:
92 // Currently active, so we can deactivate
93 if (keyToRemoveOrDeactivate.isVariableActive()) {
94 keyToRemoveOrDeactivate.deactivateVariable();
95 return true;
96 }
97 // Already deactivated, so we do not need to do anything
98 else {
99 return false;
100 }
101 // Case 3: Landmark --> Remove, as the landmark is not needed anymore
102 case DynamicVariableTypeEnum::Landmark:
103 return removeTransform(frame1, frame2, keyToRemoveOrDeactivate);
104 // Has to be one of the three cases
105 default:
106 throw std::logic_error("Variable Type not found.");
107 }
108 }
109 // Case: Frame pair is not in dictionary
110 else {
111 return false;
112 }
113 }

169 // Safe addition of new frame pair to dictionary --> checks whether already present
170 template <char SYMBOL_CHAR>
171 bool addNewFramePairSafelyToDictionary(DynamicFactorGraphStateKey<GTSAM_DYNAMIC_STATE_TYPE>& returnStateKey, const std::string& frame1,
172 const std::string& frame2, const double timeK,
173 const gtsam::Pose3& approximateTransformationBeforeOptimization,
174 const DynamicVariableType& variableType) {
175 // Check and modify content --> acquire lock
176 std::lock_guard<std::mutex> lock(internalDictionaryModifierMutex_);
177
178 // Logic
179 // CASE 1: Frame pair is already in dictionary, hence keyframe is not new (measurement either active or not)
180 if (TransformsDictionary<DynamicFactorGraphStateKey<GTSAM_DYNAMIC_STATE_TYPE>>::isFramePairInDictionary(frame1, frame2)) {
181 DynamicFactorGraphStateKey<GTSAM_DYNAMIC_STATE_TYPE>& factorGraphStateKey =
182 TransformsDictionary<DynamicFactorGraphStateKey<GTSAM_DYNAMIC_STATE_TYPE>>::lv_T_frame1_frame2(frame1, frame2);
183 // Update Timestamp and approximate transformation if newer
184 if (timeK > factorGraphStateKey.getTime()) {
185 // std::cout << " Updating timestamp of key " << gtsam::Symbol(factorGraphStateKey.key()) << " from " <<
186 // std::setprecision(14)
187 // << factorGraphStateKey.getTime() << " to " << timeK << "." << std::endl;
188 factorGraphStateKey.setTimeStamp(timeK);
189 factorGraphStateKey.setApproximateTransformationBeforeOptimization(approximateTransformationBeforeOptimization);
190 }
191 returnStateKey = factorGraphStateKey;
192 return false;
193 }
194 // CASE 2: Frame pair is not in dictionary --> Newly added
195 else {
196 returnStateKey =
197 addNewFactorGraphStateKey<SYMBOL_CHAR>(frame1, frame2, timeK, approximateTransformationBeforeOptimization, variableType);
198 return true;
199 }
200 }

213 // Functionality ------------------------------------------------------------
214 template <char SYMBOL_CHAR>
215 DynamicFactorGraphStateKey<GTSAM_DYNAMIC_STATE_TYPE> addNewFactorGraphStateKey(
216 const std::string& frame1, const std::string& frame2, const double timeK,
217 const gtsam::Pose3& approximateTransformationBeforeOptimization, const DynamicVariableType& variableType) {
218 // Compile Time Checks and Variables
219 checkSymbol<SYMBOL_CHAR>();
220 constexpr int symbolIndex = getSymbolIndex<SYMBOL_CHAR>();
221 static_assert(symbolIndex >= 0, "Symbol not found in admissible symbols.");
222
223 // Create New Key
224 gtsam::Key gtsamKey = gtsam::Symbol(SYMBOL_CHAR, numStoredTransformsPerSymbol_[symbolIndex]);
225
226 // Print out for calibrations and reference frames
227 if (variableType.getVariableTypeEnum() != DynamicVariableTypeEnum::Landmark) {
228 REGULAR_COUT << GREEN_START << " New key " << gtsam::Symbol(gtsamKey) << " created for frame pair " << frame1 << " and " << frame2
229 << "." << COLOR_END << std::endl;
230 }
231
232 // Add to main dictionary
233 DynamicFactorGraphStateKey factorGraphStateKey(gtsamKey, timeK, 0, approximateTransformationBeforeOptimization, variableType, frame1,
234 frame2);
235 TransformsDictionary<DynamicFactorGraphStateKey<GTSAM_DYNAMIC_STATE_TYPE>>::set_T_frame1_frame2(frame1, frame2, factorGraphStateKey);
236
237 // Increase the counter of specific symbol
238 ++numStoredTransformsPerSymbol_[symbolIndex];
239
240 // Add to key-to-frame pair map
241 // TODO: Check if necessary for non reference frames
242 transformGtsamKeyToFramePairMap_[gtsamKey] = std::make_pair(frame1, frame2);
243 // Add to key-to-keyframe position map
244 transformGtsamKeyToKeyframePositionMap_[gtsamKey] = variableType.getReferenceFrameKeyframePosition();
245
246 // Return
247 return factorGraphStateKey;
248 }
