19class OptimizerLMFixedLag : public OptimizerLM {
20 public:
21 explicit OptimizerLMFixedLag(const std::shared_ptr<GraphConfig> graphConfigPtr) : OptimizerLM(graphConfigPtr) {
22 // Set Linear Solver Type
23 if (graphConfigPtr_->realTimeSmootherUseCholeskyFactorizationFlag_) {
24 lmParams_.linearSolverType = gtsam::NonlinearOptimizerParams::MULTIFRONTAL_CHOLESKY;
25 REGULAR_COUT << "Using Multifrontal-Cholesky factorization for fixed-lag smoother (LM)." << std::endl;
26 } else {
27 lmParams_.linearSolverType = gtsam::NonlinearOptimizerParams::MULTIFRONTAL_QR;
28 REGULAR_COUT << "Using Multifrontal-QR factorization for fixed-lag smoother (LM)." << std::endl;
29 }
30
31 // Initialize Real-time Smoother -----------------------------------------------
32 fixedLagSmootherPtr_ = std::make_shared<gtsam::BatchFixedLagSmoother>(graphConfigPtr_->realTimeSmootherLag_, lmParams_);
33 // Print
34 fixedLagSmootherPtr_->print("LM FixedLagSmoother initialized with lag: ");
35 }
36 ~OptimizerLMFixedLag() = default;
37
38 bool update() override {
39 fixedLagSmootherPtr_->update();
40 return true;
41 }
42
43 bool update(const gtsam::NonlinearFactorGraph& newGraphFactors, const gtsam::Values& newGraphValues,
44 const std::map<gtsam::Key, double>& newGraphKeysTimeStampMap, const int depth = 0) override {
45 // Try to update
46 try {
47 fixedLagSmootherPtr_->update(newGraphFactors, newGraphValues, newGraphKeysTimeStampMap);
48 } catch (const std::out_of_range& outOfRangeExeception) { // Not specifically catching
49 std::cerr << YELLOW_START << "GMsf-LM-FixedLagSmoother" << RED_START
50 << " OutOfRange-Exception while optimizing graph: " << outOfRangeExeception.what() << COLOR_END << std::endl;
51 std::cout
52 << YELLOW_START << "GMsf-ISAM2" << RED_START
53 << " This usually happens if the measurement delay is larger than the graph-smootherLag, i.e. the optimized graph instances are "
54 "not connected. Increase the lag in this case."
55 << COLOR_END << std::endl;
56 // throw std::out_of_range(outOfRangeExeception.what());
57 return false;
58 } catch (const gtsam::ValuesKeyDoesNotExist& valuesKeyDoesNotExistException) { // Catching, as this can happen in practice (factors
59 // added that are outside the smoother window)
60 std::cout << "----------------------------------------------------------" << std::endl;
61 std::cerr << YELLOW_START << "GMsf-LM-FixedLagSmoother" << RED_START
62 << " Values Key Does Not Exist exeception while optimizing graph: " << valuesKeyDoesNotExistException.what() << COLOR_END
63 << std::endl;
64 std::cout << YELLOW_START << "GMsf-LM-FixedLagSmoother" << RED_START
65 << " This happens if a factor is added to the graph that contains a non-existent state. The most common case is a factor "
66 "containing a graph state that was marginalized out before. To avoid this, increase the factor graph lag or reduce "
67 "the delay of your measurements."
68 << COLOR_END << std::endl;
69 // Filter out the factor that caused the error -------------------------
70 gtsam::NonlinearFactorGraph newGraphFactorsFiltered;
71 gtsam::Key valuesKeyNotExistent = valuesKeyDoesNotExistException.key();
72 for (auto factor : newGraphFactors) {
73 bool factorContainsExistentKeys = true;
74 for (auto key : factor->keys()) {
75 if (key == valuesKeyNotExistent) {
76 std::cout << YELLOW_START << "GMsf-LM-FixedLagSmoother" << RED_START
77 << " Factor contains non-existent key: " << gtsam::Symbol(key) << COLOR_END << std::endl;
78 factorContainsExistentKeys = false;
79 break;
80 }
81 }
82 if (factorContainsExistentKeys) {
83 newGraphFactorsFiltered.add(factor);
84 }
85 }
86 // Filtering the keyTimestampMap
87 std::map<gtsam::Key, double> newGraphKeysTimeStampMapFiltered = newGraphKeysTimeStampMap;
88 newGraphKeysTimeStampMapFiltered.erase(valuesKeyNotExistent);
89 // Try again
90 std::cout << YELLOW_START << "GMsf-LM-FixedLagSmoother" << GREEN_START
91 << " Filtered out the factor that caused the error. Trying to optimize again." << COLOR_END << std::endl;
92 std::cout << "----------------------------------------------------------" << std::endl;
93 return update(newGraphFactorsFiltered, newGraphValues, newGraphKeysTimeStampMapFiltered);
94 } catch (const std::runtime_error& runtimeError) {
95 std::cout << YELLOW_START << "GMsf-ISAM2" << RED_START << " Runtime error while optimizing graph: " << runtimeError.what()
96 << COLOR_END << std::endl;
97 throw std::runtime_error(runtimeError.what());
98 }
99 return true;
100 }

166 }
