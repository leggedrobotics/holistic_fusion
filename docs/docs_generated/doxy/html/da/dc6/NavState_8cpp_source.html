54void NavState::updateYawInWorld(const double yaw_W_Ik) {
55 // Use GTSAM for correcto yaw convention
56 // gtsam::Pose3 T_O_Ik_old(T_O_Ik_gravityAligned_.matrix());
57 gtsam::Pose3 T_W_Ik_old = gtsam::Pose3(T_W_Ik_.matrix());
58 gtsam::Pose3 T_W_Ik_new =
59 gtsam::Pose3(gtsam::Rot3::Ypr(yaw_W_Ik, T_W_Ik_old.rotation().pitch(), T_W_Ik_old.rotation().roll()), T_W_Ik_old.translation());
60 T_W_Ik_ = T_W_Ik_new.matrix();
61}
