156bool GraphMsf::addCoreImuMeasurementAndGetState(
157 const Eigen::Vector3d& linearAcc, const Eigen::Vector3d& angularVel, double imuTimeK,
158 std::shared_ptr<SafeIntegratedNavState>& returnPreIntegratedNavStatePtr,
159 std::shared_ptr<SafeNavStateWithCovarianceAndBias>& returnOptimizedStateWithCovarianceAndBiasPtr,
160 Eigen::Matrix<double, 6, 1>& returnAddedImuMeasurements) {
161 // Setup -------------------------
162 // Increase counter
163 ++imuCallbackCounter_;
164
165 // Adapt time stamp
166 imuTimeK += graphConfigPtr_->imuTimeOffset_;
167
168 // First Iteration
169 if (preIntegratedNavStatePtr_ == nullptr) {
170 preIntegratedNavStatePtr_ = std::make_shared<SafeIntegratedNavState>();
171 preIntegratedNavStatePtr_->updateLatestMeasurementTimestamp(imuTimeK);
172 }
173
174 // Filter out imu messages with same time stamp
175 if (std::abs(imuTimeK - preIntegratedNavStatePtr_->getTimeK()) < 1e-8 && imuCallbackCounter_ > 1) {
176 REGULAR_COUT << RED_START << " Imu time " << std::setprecision(14) << imuTimeK << " was repeated." << COLOR_END << std::endl;
177 return false;
178 }
179
180 // Potentially convert to m/s^2
181 Eigen::Vector3d linearAccInMps2 = linearAcc;
182 if (graphConfigPtr_->isImuAccInG_) {
183 constexpr double gravityConst = 9.80665;
184 linearAccInMps2 *= gravityConst;
185 }
186
187 // Check the norm and spit out warning in case the acceleration is either too small or too big
188 const double accNorm = linearAccInMps2.norm();
189 if (accNorm < 2) {
190 REGULAR_COUT << RED_START << " IMU linear acceleration is too small: " << linearAccInMps2.transpose() << COLOR_END << std::endl;
191 REGULAR_COUT << RED_START << " Check whether isImuAccInG_ is not mistakenly set to false." << COLOR_END << std::endl;
192 } else if (accNorm > 90) {
193 REGULAR_COUT << RED_START << " IMU linear acceleration is too big: " << linearAccInMps2.transpose() << COLOR_END << std::endl;
194 REGULAR_COUT << RED_START << " Check whether isImuAccInG_ is not mistakenly set to true." << COLOR_END << std::endl;
195 }
196
197 // Clamp linear acceleration to 60 m/s^2 to avoid outliers
198 // constexpr double maxAcc = 60.0;
199 // if (accNorm > maxAcc) {
200 // // Iterate through each element and scale
201 // for (int i = 0; i < 3; ++i) {
202 // // If element is too big, clamp
203 // if (std::abs(linearAccInMps2(i)) > maxAcc) {
204 // linearAccInMps2(i) = (linearAccInMps2(i) > 0 ? maxAcc : -maxAcc);
205 // }
206 // }
207 // REGULAR_COUT << YELLOW_START << " Clamped IMU linear acceleration to: " << linearAccInMps2.transpose() << COLOR_END << std::endl;
208 // }
209
210 // Add measurement to buffer
211 returnAddedImuMeasurements = coreImuBufferPtr_->addToImuBuffer(imuTimeK, linearAccInMps2, angularVel);
212
213 // Locking
214 const std::lock_guard<std::mutex> initYawAndPositionLock(initYawAndPositionMutex_);
215
216 // State Machine in form of if-else statements -----------------
217 if (!alignedImuFlag_) { // Case 1: IMU not aligned
218 // Try to align
219 double imuAttitudeRoll, imuAttitudePitch = 0.0;
220 if (!alignImu_(imuAttitudeRoll, imuAttitudePitch)) { // Case 1.1: IMU alignment failed --> try again next time
221 // Print only once per second
222 if (imuCallbackCounter_ % int(graphConfigPtr_->imuRate_) == 0) {
223 REGULAR_COUT << " NOT ENOUGH IMU MESSAGES TO INITIALIZE POSE. WAITING FOR MORE..." << std::endl;
224 }
225 return false;
226 } else { // Case 1.2: IMU alignment succeeded --> continue next call iteration
227 Eigen::Matrix3d R_W_I0_attitude = gtsam::Rot3::Ypr(0.0, imuAttitudePitch, imuAttitudeRoll).matrix();
228 gtsam::Rot3 R_W_Init0_attitude(
229 R_W_I0_attitude *
230 staticTransformsPtr_->rv_T_frame1_frame2(staticTransformsPtr_->getImuFrame(), staticTransformsPtr_->getInitializationFrame())
231 .rotation()
232 .matrix());
233 // Set yaw of base frame to zero
234 // Set yaw of base frame to zero
235 R_W_Init0_attitude = gtsam::Rot3::Ypr(0.0, R_W_Init0_attitude.pitch(), R_W_Init0_attitude.roll());
236 R_W_I0_attitude =
237 R_W_Init0_attitude.matrix() *
238 staticTransformsPtr_->rv_T_frame1_frame2(staticTransformsPtr_->getInitializationFrame(), staticTransformsPtr_->getImuFrame())
239 .rotation()
240 .matrix();
241 Eigen::Isometry3d T_O_Ik_attitude = Eigen::Isometry3d::Identity();
242 T_O_Ik_attitude.matrix().block<3, 3>(0, 0) = R_W_I0_attitude;
243 Eigen::Vector3d O_t_O_Ik =
244 Eigen::Vector3d(0, 0, 0) -
245 R_W_I0_attitude *
246 staticTransformsPtr_->rv_T_frame1_frame2(staticTransformsPtr_->getImuFrame(), staticTransformsPtr_->getInitializationFrame())
247 .translation();
248 T_O_Ik_attitude.matrix().block<3, 1>(0, 3) = O_t_O_Ik;
249 REGULAR_COUT << " Setting zero position of " << staticTransformsPtr_->getInitializationFrame()
250 << ", hence iniital position of IMU is: " << O_t_O_Ik.transpose() << std::endl;
251 Eigen::Vector3d zeroPVeloctiy = Eigen::Vector3d(0, 0, 0);
252 preIntegratedNavStatePtr_ = std::make_shared<SafeIntegratedNavState>(T_O_Ik_attitude, zeroPVeloctiy, zeroPVeloctiy, imuTimeK);
253 REGULAR_COUT << GREEN_START << " IMU aligned. Initial pre-integrated state in odom frame: "
254 << preIntegratedNavStatePtr_->getT_O_Ik_gravityAligned().matrix() << COLOR_END << std::endl;
255 alignedImuFlag_ = true;
256 return false;
257 }
258 } else if (!areYawAndPositionInited()) { // Case 2: IMU aligned, but yaw and position not initialized, waiting for external
259 // initialization, meanwhile publishing initial roll and pitch
260 // Printing every second
261 if (imuCallbackCounter_ % int(graphConfigPtr_->imuRate_) == 0) {
262 REGULAR_COUT << " IMU callback waiting for initialization of global yaw and initial position." << std::endl;
263 }
264 // Publish state with correct roll and pitch, nothing has changed compared to Case 1.2
265 preIntegratedNavStatePtr_->updateLatestMeasurementTimestamp(imuTimeK);
266 returnPreIntegratedNavStatePtr = std::make_shared<SafeIntegratedNavState>(*preIntegratedNavStatePtr_);
267 return true;
268 } else if (!validFirstMeasurementReceivedFlag_) { // Case 3: No valid measurement received yet
269 if (imuCallbackCounter_ % int(graphConfigPtr_->imuRate_) == 0) {
270 REGULAR_COUT << RED_START << " IMU callback waiting for first valid measurement before initializing graph." << COLOR_END << std::endl;
271 }
272 preIntegratedNavStatePtr_->updateLatestMeasurementTimestamp(imuTimeK);
273 returnPreIntegratedNavStatePtr = std::make_shared<SafeIntegratedNavState>(*preIntegratedNavStatePtr_);
274 return true;
275 } else if (!initedGraphFlag_) { // Case 4: IMU aligned, yaw and position initialized, valid measurement received, but graph not yet
276 // initialized
277 preIntegratedNavStatePtr_->updateLatestMeasurementTimestamp(imuTimeK);
278 initGraph_(imuTimeK);
279 returnPreIntegratedNavStatePtr = std::make_shared<SafeIntegratedNavState>(*preIntegratedNavStatePtr_);
280 REGULAR_COUT << GREEN_START << " ...graph is initialized." << COLOR_END << std::endl;
281 return true;
282 }
283
284 // Case 5: Normal operation, meaning predicting the next state via integration -------------
285 // Only create state every n-th measurements (or at first successful iteration)
286 bool createNewStateFlag = imuCallbackCounter_ % graphConfigPtr_->createStateEveryNthImuMeasurement_ == 0 || !normalOperationFlag_;
287 // Add IMU factor and return propagated & optimized state
288 graphMgrPtr_->addImuFactorAndGetState(*preIntegratedNavStatePtr_, returnOptimizedStateWithCovarianceAndBiasPtr, coreImuBufferPtr_,
289 imuTimeK, createNewStateFlag);
290 returnPreIntegratedNavStatePtr = std::make_shared<SafeIntegratedNavState>(*preIntegratedNavStatePtr_);
291
292 // Set to normal operation
293 if (!normalOperationFlag_) {
294 normalOperationFlag_ = true;
295 }
296
297 // Return
298 return true;
299}

377void GraphMsf::optimizeGraph_() {
378 // While loop
379 REGULAR_COUT << " Thread for updating graph is ready." << std::endl;
380 double lastOptimizedTime = std::chrono::duration<double>(std::chrono::system_clock::now().time_since_epoch()).count();
381 double currentTime = std::chrono::duration<double>(std::chrono::system_clock::now().time_since_epoch()).count();
382 bool optimizedAtLeastOnce = false;
383 while (true) {
384 bool optimizeGraphFlag = false;
385 // Mutex for optimizeGraph Flag
386 {
387 // Lock
388 const std::lock_guard<std::mutex> optimizeGraphLock(optimizeGraphMutex_);
389 currentTime = std::chrono::duration<double>(std::chrono::system_clock::now().time_since_epoch()).count();
390 // Optimize at most at the rate of maxOptimizationFrequency but at least every second
391 if ((optimizeGraphFlag_ && ((currentTime - lastOptimizedTime) > (1.0 / graphConfigPtr_->maxOptimizationFrequency_))) ||
392 ((currentTime - lastOptimizedTime) > (1.0 / graphConfigPtr_->minOptimizationFrequency_) && optimizedAtLeastOnce)) {
393 optimizeGraphFlag = true;
394 lastOptimizedTime = currentTime;
395 optimizedAtLeastOnce = true;
396 optimizeGraphFlag_ = false;
397 }
398 }
399
400 // Optimize
401 if (optimizeGraphFlag) {
402 graphMgrPtr_->updateGraph();
403 } // else just sleep for a short amount of time before polling again
404 else {
405 std::this_thread::sleep_for(std::chrono::milliseconds(1));
406 }
407 }
