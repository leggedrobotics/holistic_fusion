25class ImuSignalLowPassFilter {
26 public:
27 ImuSignalLowPassFilter(const double cutoffFrequencyHz, const double samplingTime) {
28 setFilterParameters(cutoffFrequencyHz, samplingTime);
29 };
30 ~ImuSignalLowPassFilter() = default;
31
32 void setFilterParameters(double cutoffFrequencyHz, double samplingTime) {
33 cutoffFrequencyHz_ = cutoffFrequencyHz;
34 samplingTime_ = samplingTime;
35
36 // Calculate second-order Butterworth filter coefficients
37 // Using bilinear transform (Tustin's method) for stability
38 const double nyquistFreq = 1.0 / (2.0 * samplingTime_);
39 const double normalizedCutoff = cutoffFrequencyHz_ / nyquistFreq;
40
41 // Prevent numerical issues for very high cutoff frequencies
42 if (normalizedCutoff >= 1.0) {
43 // If cutoff is at or above Nyquist frequency, disable filtering
44 b0_ = 1.0; b1_ = 0.0; b2_ = 0.0;
45 a1_ = 0.0; a2_ = 0.0;
46 return;
47 }
48
49 // Second-order Butterworth filter design
50 const double wc = std::tan(M_PI * normalizedCutoff);
51 const double wc2 = wc * wc;
52 const double sqrt2 = std::sqrt(2.0);
53 const double k1 = sqrt2 * wc;
54 const double k2 = wc2;
55 const double a0 = k2 + k1 + 1.0;
56
57 // Normalized coefficients
58 b0_ = k2 / a0;
59 b1_ = 2.0 * k2 / a0;
60 b2_ = k2 / a0;
61 a1_ = (2.0 * k2 - 2.0) / a0;
62 a2_ = (k2 - k1 + 1.0) / a0;
63 }
64
65 void reset() {
66 // Reset acceleration filter state
67 x1_accel_ = Eigen::Vector3d::Zero();
68 x2_accel_ = Eigen::Vector3d::Zero();
69 y1_accel_ = Eigen::Vector3d::Zero();
70 y2_accel_ = Eigen::Vector3d::Zero();
71
72 // Reset angular velocity filter state
73 x1_gyro_ = Eigen::Vector3d::Zero();
74 x2_gyro_ = Eigen::Vector3d::Zero();
75 y1_gyro_ = Eigen::Vector3d::Zero();
76 y2_gyro_ = Eigen::Vector3d::Zero();
77 }

79 Eigen::Matrix<double, 6, 1> filter(const Eigen::Vector3d& inputAcceleration_x0, const Eigen::Vector3d& inputAngularVel_x0) {
80 // Apply second-order filter to acceleration
81 Eigen::Vector3d outputAcceleration = applyFilter(inputAcceleration_x0, x1_accel_, x2_accel_, y1_accel_, y2_accel_);
82
83 // Apply second-order filter to angular velocity
84 Eigen::Vector3d outputAngularVel = applyFilter(inputAngularVel_x0, x1_gyro_, x2_gyro_, y1_gyro_, y2_gyro_);
85
86 // Return stacked vectors as 6D vector
87 return (Eigen::Matrix<double, 6, 1>() << outputAcceleration, outputAngularVel).finished();
88 }

94 double getCutoffFrequency() const { return cutoffFrequencyHz_; }

100 double getSamplingTime() const { return samplingTime_; }

112 Eigen::Vector3d applyFilter(const Eigen::Vector3d& input,
113 Eigen::Vector3d& x1, Eigen::Vector3d& x2,
114 Eigen::Vector3d& y1, Eigen::Vector3d& y2) {
115 // Direct Form II implementation of second-order filter
116 // y[n] = b0*x[n] + b1*x[n-1] + b2*x[n-2] - a1*y[n-1] - a2*y[n-2]
117 Eigen::Vector3d output = b0_ * input + b1_ * x1 + b2_ * x2 - a1_ * y1 - a2_ * y2;
118
119 // Update state variables
120 x2 = x1;
121 x1 = input;
122 y2 = y1;
123 y1 = output;
124
125 return output;
126 }
127
128 // Filter parameters
129 double cutoffFrequencyHz_;
130 double samplingTime_;
131
132 // Filter coefficients (second-order Butterworth)
133 double b0_, b1_, b2_; // Numerator coefficients
134 double a1_, a2_; // Denominator coefficients (a0 = 1)
135
136 // Filter state for acceleration (3D)
137 Eigen::Vector3d x1_accel_ = Eigen::Vector3d::Zero(); // Previous input
138 Eigen::Vector3d x2_accel_ = Eigen::Vector3d::Zero(); // Second previous input
139 Eigen::Vector3d y1_accel_ = Eigen::Vector3d::Zero(); // Previous output
140 Eigen::Vector3d y2_accel_ = Eigen::Vector3d::Zero(); // Second previous output
141
142 // Filter state for angular velocity (3D)
143 Eigen::Vector3d x1_gyro_ = Eigen::Vector3d::Zero(); // Previous input
144 Eigen::Vector3d x2_gyro_ = Eigen::Vector3d::Zero(); // Second previous input
145 Eigen::Vector3d y1_gyro_ = Eigen::Vector3d::Zero(); // Previous output
146 Eigen::Vector3d y2_gyro_ = Eigen::Vector3d::Zero(); // Second previous output
147};
