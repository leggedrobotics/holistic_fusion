37// Pose3
38void GraphMsfHolistic::addUnaryPose3AbsoluteMeasurement(const UnaryMeasurementXDAbsolute<Eigen::Isometry3d, 6>& R_T_R_S,
39 const bool addToOnlineSmootherFlag) {
40 // Valid measurement received
41 if (!validFirstMeasurementReceivedFlag_) {
42 validFirstMeasurementReceivedFlag_ = true;
43 }
44
45 // Only take actions if graph has been initialized
46 if (!initedGraphFlag_) { // Graph not yet initialized
47 return;
48 } else { // Graph initialized
49 // Check for covariance violation
50 bool covarianceViolatedFlag = isCovarianceViolated_<6>(R_T_R_S.unaryMeasurementNoiseDensity(), R_T_R_S.covarianceViolationThreshold());
51 if (checkAndPrintCovarianceViolation_(R_T_R_S.measurementName(), covarianceViolatedFlag)) {
52 return;
53 }
54
55 // Create GMSF expression
56 auto gmsfUnaryExpressionPose3Ptr = std::make_shared<GmsfUnaryExpressionAbsolutePose3>(
57 std::make_shared<UnaryMeasurementXDAbsolute<Eigen::Isometry3d, 6>>(R_T_R_S), staticTransformsPtr_->getImuFrame(),
58 staticTransformsPtr_->rv_T_frame1_frame2(staticTransformsPtr_->getImuFrame(), R_T_R_S.sensorFrameName()),
59 graphConfigPtr_->createReferenceAlignmentKeyframeEveryNSeconds_);
60
61 // Add factor to graph
62 graphMgrPtr_->addUnaryGmsfExpressionFactor<GmsfUnaryExpressionAbsolutePose3>(gmsfUnaryExpressionPose3Ptr, addToOnlineSmootherFlag);
63
64 // Optimize ---------------------------------------------------------------
65 {
66 // Mutex for optimizeGraph Flag
67 const std::lock_guard<std::mutex> optimizeGraphLock(optimizeGraphMutex_);
68 optimizeGraphFlag_ = true;
69 }
70 }
71}
