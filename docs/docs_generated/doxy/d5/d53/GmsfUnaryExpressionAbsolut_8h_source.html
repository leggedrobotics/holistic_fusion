38 // ii).A Holistically Optimize over Fixed Frames
39 void transformImuStateFromWorldToReferenceFrame(DynamicDictionaryContainer& gtsamDynamicExpressionKeys,
40 const gtsam::NavState& W_currentPropagatedState,
41 const bool centerMeasurementsAtKeyframePositionBeforeAlignmentFlag) final {
42 if (gmsfUnaryAbsoluteMeasurementPtr_->fixedFrameName() == gmsfUnaryAbsoluteMeasurementPtr_->worldFrameName()) {
43 // Do nothing as this is a world frame measurement
44 return;
45 }
46
47 // Mutex because we are changing the dynamically allocated graphKeys
48 std::lock_guard<std::mutex> modifyGraphKeysLock(gtsamDynamicExpressionKeys.get<gtsam::Pose3>().mutex());
49
50 // Run through steps needed for absolute measurements
51 // If it should be centered --> create keyframe for measurement
52 Eigen::Vector3d measurementOriginPosition = Eigen::Vector3d::Zero();
53 if (centerMeasurementsAtKeyframePositionBeforeAlignmentFlag) {
54 measurementOriginPosition = this->getMeasurementPosition();
55 }
56
57 // Initial Guess
58 gtsam::Pose3 T_W_fixedFrame_initial = this->computeT_W_fixedFrame_initial(W_currentPropagatedState);
59
60 // If initial guess is externally set
61 bool initialGuessSetExternally = false;
62 if (gtsamDynamicExpressionKeys.get<gtsam::Pose3>().getInitialGuessForFramePair(gmsfUnaryAbsoluteMeasurementPtr_->worldFrameName(),
63 gmsfUnaryAbsoluteMeasurementPtr_->fixedFrameName(),
64 T_W_fixedFrame_initial)) {
65 initialGuessSetExternally = true;
66 REGULAR_COUT << GREEN_START << " Initial guess for transform " << gmsfUnaryAbsoluteMeasurementPtr_->worldFrameName() << " to "
67 << gmsfUnaryAbsoluteMeasurementPtr_->fixedFrameName() << " was set externally." << COLOR_END << std::endl;
68 // Remove from map
69 gtsamDynamicExpressionKeys.get<gtsam::Pose3>().removeInitialGuessForFramePair(gmsfUnaryAbsoluteMeasurementPtr_->worldFrameName(),
70 gmsfUnaryAbsoluteMeasurementPtr_->fixedFrameName());
71 }
72
73 // A. Search for the new graph key of T_fixedFrame_W -----------------------------------------------------
74 bool newGraphKeyAddedFlag = false;
75 const DynamicVariableType dynamicVariableType =
76 DynamicVariableType::RefFrame(measurementOriginPosition, gmsfUnaryAbsoluteMeasurementPtr_->timeK());
77 // Search for new graph key, if not found, add it
78 DynamicFactorGraphStateKey<gtsam::Pose3> graphKey = gtsamDynamicExpressionKeys.get<gtsam::Pose3>().getTransformationKey<'r'>(
79 newGraphKeyAddedFlag, gmsfUnaryAbsoluteMeasurementPtr_->worldFrameName(), gmsfUnaryAbsoluteMeasurementPtr_->fixedFrameName(),
80 gmsfUnaryAbsoluteMeasurementPtr_->timeK(), T_W_fixedFrame_initial, dynamicVariableType);
81 // Get age of keyframe
82 const double keyframeAge = graphKey.computeKeyframeAge(gmsfUnaryAbsoluteMeasurementPtr_->timeK());
83
84 // B: Take care of the old keyframe -----------------------------------------------------
85 // If deactivated or too old, we should remove the old keyframe and create a new one (with or without random walk)
86 // In any case, we have to add relation between old and new keyframe
87 // Offline graph: we can always add the delta as all states are part of the optimization (random walk or deterministic keyframe
88 // displacement).
89 // Online graph: we can only add the delta if the old keyframe is still active (random walk or deterministic keyframe displacement. If
90 // it is not active, add prior from old keyframe to new keyframe (including the displacement).
91 bool introducedNewKeyframeDisplacement = false;
92 // Old GTSAM key --> needed for adding constraints between previous and new keyframe
93 const gtsam::Key oldGtsamKey = graphKey.key();
94 const Eigen::Vector3d oldKeyframePosition = graphKey.getReferenceFrameKeyframePosition();
95 bool oldVariableWasActive = graphKey.isVariableActive();
96 // If needed, have a copy of the old key to add back a prior to the online graph
97 std::unique_ptr<DynamicFactorGraphStateKey<gtsam::Pose3>> oldKeyPtr = nullptr;
98 if (!oldVariableWasActive) {
99 REGULAR_COUT << " Old keyframe is not active anymore. Hence keeping the belief and uncertainty of previous key." << std::endl;
100 oldKeyPtr = std::make_unique<DynamicFactorGraphStateKey<gtsam::Pose3>>(graphKey);
101 }
102
103 // C: Check if we need to create a new keyframe -----------------------------------------------------
104 // Case 1: Old keyframe not active --> just reactivate the keyframe (independent of whether new keyframe was added or not)
105 if (!oldVariableWasActive) {
106 // Check whether no new keyframe was added
107 if (newGraphKeyAddedFlag) {
108 throw std::logic_error("GmsfUnaryExpressionAbsolut: Old keyframe was not active, but new keyframe was added.");
109 }
110 // Reactivate the keyframe (have to use reference instead of copy "graphKey" above)
111 gtsamDynamicExpressionKeys.get<gtsam::Pose3>()
112 .lv_T_frame1_frame2(gmsfUnaryAbsoluteMeasurementPtr_->worldFrameName(), gmsfUnaryAbsoluteMeasurementPtr_->fixedFrameName())
113 .activateVariable();
114 // Set number of steps optimized to zero
115 gtsamDynamicExpressionKeys.get<gtsam::Pose3>()
116 .lv_T_frame1_frame2(gmsfUnaryAbsoluteMeasurementPtr_->worldFrameName(), gmsfUnaryAbsoluteMeasurementPtr_->fixedFrameName())
117 .resetNumberStepsOptimized();
118 // Print out
119 REGULAR_COUT << GREEN_START << " Reactivated old keyframe " << gtsam::Symbol(graphKey.key())
120 << " as it was deactivated, but is needed again. Also set number of optimization steps to 0." << COLOR_END << std::endl;
121
122 // If the old keyframe was not active, then it is not part of the optimization anymore (or has never been) --> prior to online graph
123 // Make sure that old key ptr is not null
124 if (oldKeyPtr == nullptr) {
125 throw std::logic_error("GmsfUnaryExpressionAbsolut: Old keyframe was not active, but old key ptr is null.");
126 }
127 // Container for belief of old keyframe
128 gtsam::Pose3 T_W_fixedFrameOld;
129 // Noise model container
130 boost::shared_ptr<gtsam::noiseModel::Base> noiseModelPtr;
131 // Case 1: Has been optimized before
132 if (oldKeyPtr->getNumberStepsOptimized() > 0) {
133 T_W_fixedFrameOld = oldKeyPtr->getTransformationAfterOptimization();
134 Eigen::Matrix<double, 6, 6> oldKeyframeCovariance = oldKeyPtr->getCovarianceAfterOptimization();
135 // Noise model from 6x6 covariance
136 noiseModelPtr = gtsam::noiseModel::Gaussian::Covariance(oldKeyframeCovariance);
137 // Print
138 REGULAR_COUT << " Old keyframe has been optimized before. Adding previous optimization outcome again to online graph." << std::endl;
139 }
140 // Case 2: Has not been optimized before
141 else {
142 T_W_fixedFrameOld = oldKeyPtr->getApproximateTransformationBeforeOptimization();
143 Eigen::Matrix<double, 6, 1> oldKeyframeCovariance = gmsfUnaryAbsoluteMeasurementPtr_->initialSe3AlignmentNoise();
144 // Noise model for prior
145 noiseModelPtr = gtsam::noiseModel::Diagonal::Sigmas(gmsfUnaryAbsoluteMeasurementPtr_->initialSe3AlignmentNoise());
146 // Print
147 REGULAR_COUT << " Old keyframe has not been optimized before. Adding very initial guess that should have been added last time."
148 << std::endl;
149 }
150 // Add information to online graph only (as has been marginalized out), offline still has it
151 // Prior belief
152 this->newOnlinePosePriorFactors_.emplace_back(oldGtsamKey, T_W_fixedFrameOld, noiseModelPtr);
153 // State value
154 this->newOnlineStateValues_.insert(oldGtsamKey, T_W_fixedFrameOld);
155 // Print
156 REGULAR_COUT << GREEN_START << " Added prior to old keyframe " << gtsam::Symbol(oldGtsamKey)
157 << " to the online graph, as it was not active anymore." << COLOR_END << std::endl;
158 }
159
160 // Case 2: Keyframe is too old --> create a new keyframe to model the displacement
161 if (keyframeAge > createReferenceAlignmentKeyframeEveryNSeconds_) {
162 // Remove the old keyframe from memory
163 gtsamDynamicExpressionKeys.get<gtsam::Pose3>().removeTransform(gmsfUnaryAbsoluteMeasurementPtr_->worldFrameName(),
164 gmsfUnaryAbsoluteMeasurementPtr_->fixedFrameName(), graphKey);
165 // Create a new keyframe
166 graphKey = gtsamDynamicExpressionKeys.get<gtsam::Pose3>().getTransformationKey<'r'>(
167 newGraphKeyAddedFlag, gmsfUnaryAbsoluteMeasurementPtr_->worldFrameName(), gmsfUnaryAbsoluteMeasurementPtr_->fixedFrameName(),
168 gmsfUnaryAbsoluteMeasurementPtr_->timeK(), T_W_fixedFrame_initial, dynamicVariableType);
169 // Assert that new keyframe was added and that it is active
170 assert(newGraphKeyAddedFlag && graphKey.isVariableActive());
171 // Set flag that we introduced a new keyframe
172 introducedNewKeyframeDisplacement = true;
173 }
174
175 // D: Shift the measurement to the robot position and recompute initial guess if we create keyframes -----------------------------------
176 // Has to be done here, as we did not know the keyframe position before
177 if (centerMeasurementsAtKeyframePositionBeforeAlignmentFlag && !initialGuessSetExternally) {
178 // Shift the measurement to the robot position
179 this->setMeasurementPosition(this->getMeasurementPosition() - graphKey.getReferenceFrameKeyframePosition());
180 // Recompute initial guess
181 T_W_fixedFrame_initial = this->computeT_W_fixedFrame_initial(W_currentPropagatedState);
182 // Update the initial guess
183 gtsamDynamicExpressionKeys.get<gtsam::Pose3>()
184 .lv_T_frame1_frame2(gmsfUnaryAbsoluteMeasurementPtr_->worldFrameName(), gmsfUnaryAbsoluteMeasurementPtr_->fixedFrameName())
185 .setApproximateTransformationBeforeOptimization(T_W_fixedFrame_initial);
186 }
187
188 // E: Define expression for T_fixedFrame_W -----------------------------------------------------
189 gtsam::Pose3_ exp_T_W_fixedFrame(graphKey.key()); // T_fixedFrame_W
190 // Call main child function
191 transformStateToReferenceFrameMeasurement(exp_T_W_fixedFrame);
192
193 // F: Initial values and priors if needed -----------------------------------------------------
194 if (newGraphKeyAddedFlag) {
195 REGULAR_COUT << " Initial Guess for T_W_" << gmsfUnaryAbsoluteMeasurementPtr_->fixedFrameName()
196 << ", RPY (deg): " << T_W_fixedFrame_initial.rotation().rpy().transpose() * (180.0 / M_PI)
197 << ", t (x, y, z): " << T_W_fixedFrame_initial.translation().transpose() << std::endl;
198 // Insert Values
199 this->newOnlineStateValues_.insert(graphKey.key(), T_W_fixedFrame_initial);
200 this->newOfflineStateValues_.insert(graphKey.key(), T_W_fixedFrame_initial);
201
202 // Case 1: Entirely new keyframe has been added (not just a displacement) --> add prior to online graph
203 if (!introducedNewKeyframeDisplacement) {
204 // Insert Prior ONLY for online graph (offline is observable regardless)
205 this->newOnlinePosePriorFactors_.emplace_back(
206 graphKey.key(), T_W_fixedFrame_initial,
207 gtsam::noiseModel::Diagonal::Sigmas(gmsfUnaryAbsoluteMeasurementPtr_->initialSe3AlignmentNoise()));
208 REGULAR_COUT << GREEN_START << " Adding prior to new keyframe " << gtsam::Symbol(graphKey.key()) << " with noise "
209 << gmsfUnaryAbsoluteMeasurementPtr_->initialSe3AlignmentNoise().transpose()
210 << " to the online graph, as it is an entirely new keyframe." << COLOR_END << std::endl;
211 }
212 // Case 2: New keyframe has been added, but one existed before already --> add relative constraint from old to new keyframe
213 else {
214 // Add relative constraint from old keyframe to new keyframe
215 gtsam::Point3 relativeKeyframeTranslation(graphKey.getReferenceFrameKeyframePosition() - oldKeyframePosition);
216 // Transform between old and new keyframe
217 gtsam::Pose3 T_fixedFrameOld_fixedFrame(gtsam::Rot3::Identity(), relativeKeyframeTranslation);
218 // Define noise model --> either random walk or deterministic displacement
219
220 // a): Random Walk between the two
221 if (gmsfUnaryAbsoluteMeasurementPtr_->modelAsRandomWalkFlag()) {
222 boost::shared_ptr<gtsam::noiseModel::Diagonal> noiseModelPtr =
223 gtsam::noiseModel::Diagonal::Sigmas(gmsfUnaryAbsoluteMeasurementPtr_->se3AlignmentRandomWalk());
224 this->newOnlineAndOfflinePoseBetweenFactors_.emplace_back(oldGtsamKey, graphKey.key(), T_fixedFrameOld_fixedFrame, noiseModelPtr);
225
226 // Add prior from old keyframe to new keyframe
227 REGULAR_COUT << GREEN_START << " Adding relative RANDOM WALK constraint from key " << gtsam::Symbol(oldGtsamKey) << " to key "
228 << gtsam::Symbol(graphKey.key()) << ": " << T_fixedFrameOld_fixedFrame << COLOR_END << std::endl;
229
230 }
231 // b): Deterministic displacement modelled as equality constraint
232 else {
233 boost::shared_ptr<gtsam::noiseModel::Constrained> noiseModelPtr =
234 gtsam::noiseModel::Constrained::MixedSigmas(gmsfUnaryAbsoluteMeasurementPtr_->se3AlignmentRandomWalk());
235 this->newOnlineAndOfflinePoseBetweenFactors_.emplace_back(oldGtsamKey, graphKey.key(), T_fixedFrameOld_fixedFrame, noiseModelPtr);
236
237 // Add prior from old keyframe to new keyframe
238 REGULAR_COUT << GREEN_START << " Adding relative DETERMINISTIC constraint from key " << gtsam::Symbol(oldGtsamKey) << " to key "
239 << gtsam::Symbol(graphKey.key()) << ": " << T_fixedFrameOld_fixedFrame << COLOR_END << std::endl;
240 }
241 }
242 }
243 }
