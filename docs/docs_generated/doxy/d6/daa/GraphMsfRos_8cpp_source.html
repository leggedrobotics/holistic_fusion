124bool GraphMsfRos::srvOfflineSmootherOptimizeCallback(graph_msf_ros_msgs::OfflineOptimizationTrigger::Request& req,
125 graph_msf_ros_msgs::OfflineOptimizationTrigger::Response& res) {
126 // Max Iterations from service call

128 const bool saveCovarianceFlag = req.save_covariance;
129
130 // Trigger offline smoother optimization and create response
131 if (GraphMsf::optimizeSlowBatchSmoother(maxIterations, optimizationResultLoggingPath, saveCovarianceFlag)) {
132 res.success = true;
133 res.message = "Optimization successful.";

135 res.success = false;
136 res.message = "Optimization failed.";
137 }

153void GraphMsfRos::addToPathMsg(const nav_msgs::PathPtr& pathPtr, const std::string& fixedFrameName, const ros::Time& stamp,
154 const Eigen::Vector3d& t, const int maxBufferLength) {
155 geometry_msgs::PoseStamped pose;
156 pose.header.frame_id = fixedFrameName;
157 pose.header.stamp = stamp;
158 pose.pose.position.x = t(0);
159 pose.pose.position.y = t(1);
160 pose.pose.position.z = t(2);
161 pathPtr->header.frame_id = fixedFrameName;
162 pathPtr->header.stamp = stamp;
163 pathPtr->poses.push_back(pose);
164 if (pathPtr->poses.size() > maxBufferLength) {
165 pathPtr->poses.erase(pathPtr->poses.begin());
166 }
167}

169void GraphMsfRos::addToOdometryMsg(const nav_msgs::OdometryPtr& msgPtr, const std::string& fixedFrame, const std::string& movingFrame,
170 const ros::Time& stamp, const Eigen::Isometry3d& T, const Eigen::Vector3d& F_v_W_F,
171 const Eigen::Vector3d& F_w_W_F, const Eigen::Matrix<double, 6, 6>& poseCovariance,
172 const Eigen::Matrix<double, 6, 6>& twistCovariance) {
173 msgPtr->header.frame_id = fixedFrame;
174 msgPtr->child_frame_id = movingFrame;
175 msgPtr->header.stamp = stamp;
176 tf::poseTFToMsg(isometry3ToTf(T), msgPtr->pose.pose);
177 msgPtr->twist.twist.linear.x = F_v_W_F(0);
178 msgPtr->twist.twist.linear.y = F_v_W_F(1);
179 msgPtr->twist.twist.linear.z = F_v_W_F(2);
180 msgPtr->twist.twist.angular.x = F_w_W_F(0);
181 msgPtr->twist.twist.angular.y = F_w_W_F(1);
182 msgPtr->twist.twist.angular.z = F_w_W_F(2);
183
184 for (int i = 0; i < 6; i++) {
185 for (int j = 0; j < 6; j++) {
186 msgPtr->pose.covariance[6 * i + j] = poseCovariance(i, j);
187 msgPtr->twist.covariance[6 * i + j] = twistCovariance(i, j);
188 }
189 }
190}

205void GraphMsfRos::addToLandmarkMarkerArrayMsg(const visualization_msgs::MarkerArrayPtr& markerArrayPtr,
206 const std::string& referenceFrameName, const ros::Time& stamp, const Eigen::Isometry3d& T_R_L,
207 const std::string& transformName) {
208 // Save last 10 markers to know whether to add or not
209 static std::deque<std::string> lastMarkerNames;
210 static int landmarkIndex = 0;
211 // Check whether marker already exists --> return if already added
212 if (std::find(lastMarkerNames.begin(), lastMarkerNames.end(), transformName) != lastMarkerNames.end()) {
213 return;
214 }
215 // Else add
216 else {
217 lastMarkerNames.push_back(transformName);
218 if (lastMarkerNames.size() > NUM_KEEP_MARKER_NAMES) {
219 lastMarkerNames.pop_front();
220 }
221 }
222
223 // Marker
224 visualization_msgs::Marker landmarkMarker;
225 GraphMsfRos::createContactMarker(referenceFrameName, stamp, T_R_L.translation(), "landmarks", landmarkIndex++,
226 Eigen::Vector3d(1.0, 0.0, 0.0), landmarkMarker);
227
228 // Add to array
229 markerArrayPtr->markers.push_back(landmarkMarker);
230 // If too many markers, delete the first one
231 if (markerArrayPtr->markers.size() > NUM_MARKERS_IN_ARRAY) {
232 markerArrayPtr->markers.erase(markerArrayPtr->markers.begin());
233 }
234}

237 Eigen::Matrix<double, 6, 6>& poseCovarianceRos, Eigen::Matrix<double, 6, 6>& twistCovarianceRos,
238 const std::shared_ptr<graph_msf::SafeNavStateWithCovarianceAndBias>& optimizedStateWithCovarianceAndBiasPtr) {

244 } else {
245 poseCovarianceRos.setZero();
246 twistCovarianceRos.setZero();
247 }

251void GraphMsfRos::createVelocityMarker(const std::string& referenceFrameName, const ros::Time& stamp, const Eigen::Vector3d& velocity,
252 const Eigen::Vector3d& colorRgb, visualization_msgs::Marker& marker) {
253 // Arrow
254 marker.header.frame_id = referenceFrameName;
255 marker.header.stamp = stamp;
256 marker.id = 0;
257 marker.type = visualization_msgs::Marker::ARROW;
258 marker.action = visualization_msgs::Marker::ADD;
259 // Scale and Color
260 const double scale = velocity.norm();
261 marker.scale.x = 0.1; // shaft diameter
262 marker.scale.y = 0.2; // head diameter
263 marker.scale.z = 0.2; // head length
264 marker.color.a = 1.0;
265 marker.color.r = colorRgb(0);
266 marker.color.g = colorRgb(1);
267 marker.color.b = colorRgb(2);
268 // Define Arrow through start and end point
269 geometry_msgs::Point startPoint, endPoint;
270 startPoint = geometry_msgs::Point();
271 startPoint.x = 0.0; // origin
272 startPoint.y = 0.0; // origin
273 startPoint.z = 0.0; // origin
274 endPoint = geometry_msgs::Point();
275 endPoint.x = startPoint.x + velocity(0);
276 endPoint.y = startPoint.y + velocity(1);
277 endPoint.z = startPoint.z + velocity(2);
278 marker.points.push_back(startPoint);
279 marker.points.push_back(endPoint);
280 // Quaternion
281 tf::Quaternion q = tf::createQuaternionFromRPY(0, 0, 0);
282 tf::quaternionTFToMsg(q, marker.pose.orientation);
283}

320void GraphMsfRos::imuCallback(const sensor_msgs::Imu::ConstPtr& imuMsgPtr) {
321 // Convert to Eigen
322 Eigen::Vector3d linearAcc(imuMsgPtr->linear_acceleration.x, imuMsgPtr->linear_acceleration.y, imuMsgPtr->linear_acceleration.z);
323 Eigen::Vector3d angularVel(imuMsgPtr->angular_velocity.x, imuMsgPtr->angular_velocity.y, imuMsgPtr->angular_velocity.z);
324 Eigen::Matrix<double, 6, 1> addedImuMeasurements; // accel, gyro
325
326 // Create pointer for carrying state
327 std::shared_ptr<SafeIntegratedNavState> preIntegratedNavStatePtr = nullptr;
328 std::shared_ptr<SafeNavStateWithCovarianceAndBias> optimizedStateWithCovarianceAndBiasPtr = nullptr;
329
330 // Add measurement and get state
331 if (GraphMsf::addCoreImuMeasurementAndGetState(linearAcc, angularVel, imuMsgPtr->header.stamp.toSec(), preIntegratedNavStatePtr,
332 optimizedStateWithCovarianceAndBiasPtr, addedImuMeasurements)) {
333 // Encountered Delay
334 if (ros::Time::now() - ros::Time(preIntegratedNavStatePtr->getTimeK()) > ros::Duration(0.5)) {
335 REGULAR_COUT << RED_START << " Encountered delay of " << std::setprecision(14)
336 << (ros::Time::now() - ros::Time(preIntegratedNavStatePtr->getTimeK())).toSec() << " seconds." << COLOR_END << std::endl;
337 }
338
339 // Publish Odometry
340 this->publishState(preIntegratedNavStatePtr, optimizedStateWithCovarianceAndBiasPtr);
341
342 // Publish Filtered Imu Measurements
343 // this->publishAddedImuMeas_(addedImuMeasurements, imuMsgPtr->header.stamp);
344 } else if (GraphMsf::isGraphInited()) {
345 REGULAR_COUT << RED_START << " Could not add IMU measurement." << COLOR_END << std::endl;
346 }
347}

349// Publish state ---------------------------------------------------------------
350// Higher Level Functions
351void GraphMsfRos::publishState(
352 const std::shared_ptr<graph_msf::SafeIntegratedNavState>& integratedNavStatePtr,
353 const std::shared_ptr<graph_msf::SafeNavStateWithCovarianceAndBias>& optimizedStateWithCovarianceAndBiasPtr) {
354 // Covariances
355 Eigen::Matrix<double, 6, 6> poseCovarianceRos, twistCovarianceRos;
356 extractCovariancesFromOptimizedState(poseCovarianceRos, twistCovarianceRos, optimizedStateWithCovarianceAndBiasPtr);
357
358 // Variances (only digonal elements
359 Eigen::Vector3d positionVarianceRos = poseCovarianceRos.block<3, 3>(0, 0).diagonal();
360 Eigen::Vector3d orientationVarianceRos = poseCovarianceRos.block<3, 3>(3, 3).diagonal();
361
362 // Publish non-time critical data in a separate thread
363 std::thread publishNonTimeCriticalDataThread(&GraphMsfRos::publishNonTimeCriticalData, this, poseCovarianceRos, twistCovarianceRos,
364 positionVarianceRos, orientationVarianceRos, integratedNavStatePtr,
365 optimizedStateWithCovarianceAndBiasPtr);
366 publishNonTimeCriticalDataThread.detach();
367}

369// Copy the arguments in order to be thread safe
370void GraphMsfRos::publishNonTimeCriticalData(
371 const Eigen::Matrix<double, 6, 6> poseCovarianceRos, const Eigen::Matrix<double, 6, 6> twistCovarianceRos,
372 const Eigen::Vector3d positionVarianceRos, const Eigen::Vector3d orientationVarianceRos,
373 const std::shared_ptr<const graph_msf::SafeIntegratedNavState> integratedNavStatePtr,
374 const std::shared_ptr<const graph_msf::SafeNavStateWithCovarianceAndBias> optimizedStateWithCovarianceAndBiasPtr) {
375 // Mutex for not overloading ROS
376 std::lock_guard<std::mutex> lock(rosPublisherMutex_);
377
378 // Time
379 const double& timeK = integratedNavStatePtr->getTimeK(); // Alias
380
381 // Odometry messages
382 publishImuOdoms(integratedNavStatePtr, poseCovarianceRos, twistCovarianceRos);
383
384 // Publish to TF
385 // B_O
386 Eigen::Isometry3d T_B_Ok =
387 staticTransformsPtr_->rv_T_frame1_frame2(staticTransformsPtr_->getBaseLinkFrame(), staticTransformsPtr_->getImuFrame()) *
388 integratedNavStatePtr->getT_O_Ik_gravityAligned().inverse();
389 publishTfTreeTransform(staticTransformsPtr_->getBaseLinkFrame(), staticTransformsPtr_->getOdomFrame(), timeK, T_B_Ok);
390 // O_W
391 Eigen::Isometry3d T_O_W = integratedNavStatePtr->getT_W_O().inverse();
392 publishTfTreeTransform(staticTransformsPtr_->getOdomFrame(), staticTransformsPtr_->getWorldFrame(), timeK, T_O_W);
393
394 // Publish Variances
395 publishDiagVarianceVectors(positionVarianceRos, orientationVarianceRos, timeK);
396
397 // Publish Velocity Markers
398 publishVelocityMarkers(integratedNavStatePtr);
399
400 // Publish paths
401 publishImuPaths(integratedNavStatePtr);
402
403 // Optimized estimate ----------------------
404 publishOptimizedStateAndBias(optimizedStateWithCovarianceAndBiasPtr, poseCovarianceRos, twistCovarianceRos);
405}

407void GraphMsfRos::publishOptimizedStateAndBias(
408 const std::shared_ptr<const graph_msf::SafeNavStateWithCovarianceAndBias> optimizedStateWithCovarianceAndBiasPtr,
409 const Eigen::Matrix<double, 6, 6>& poseCovarianceRos, const Eigen::Matrix<double, 6, 6>& twistCovarianceRos) {
410 if (optimizedStateWithCovarianceAndBiasPtr != nullptr &&
411 optimizedStateWithCovarianceAndBiasPtr->getTimeK() - lastOptimizedStateTimestamp_ > 1e-03) {
412 // Time of this optimized state
413 lastOptimizedStateTimestamp_ = optimizedStateWithCovarianceAndBiasPtr->getTimeK();
414
415 // Odometry messages
416 // world->imu
417 if (pubOptWorldImu_.getNumSubscribers() > 0) {
418 addToOdometryMsg(optWorldImuMsgPtr_, staticTransformsPtr_->getWorldFrame(), staticTransformsPtr_->getImuFrame(),
419 ros::Time(optimizedStateWithCovarianceAndBiasPtr->getTimeK()), optimizedStateWithCovarianceAndBiasPtr->getT_W_Ik(),
420 optimizedStateWithCovarianceAndBiasPtr->getI_v_W_I(), optimizedStateWithCovarianceAndBiasPtr->getI_w_W_I(),
421 poseCovarianceRos, twistCovarianceRos);
422 pubOptWorldImu_.publish(optWorldImuMsgPtr_);
423 }
424
425 // Path
426 // world->imu
427 addToPathMsg(optWorldImuPathPtr_, staticTransformsPtr_->getWorldFrame(), ros::Time(optimizedStateWithCovarianceAndBiasPtr->getTimeK()),
428 optimizedStateWithCovarianceAndBiasPtr->getT_W_Ik().translation(), graphConfigPtr_->imuBufferLength_ * 20);
429 if (pubOptWorldImuPath_.getNumSubscribers() > 0) {
430 pubOptWorldImuPath_.publish(optWorldImuPathPtr_);
431 }
432
433 // Biases
434 // Publish accel bias
435 if (pubAccelBias_.getNumSubscribers() > 0) {
436 Eigen::Vector3d accelBias = optimizedStateWithCovarianceAndBiasPtr->getAccelerometerBias();
437 accelBiasMsgPtr_->header.stamp = ros::Time(optimizedStateWithCovarianceAndBiasPtr->getTimeK());
438 accelBiasMsgPtr_->header.frame_id = staticTransformsPtr_->getImuFrame();
439 accelBiasMsgPtr_->vector.x = accelBias(0);
440 accelBiasMsgPtr_->vector.y = accelBias(1);
441 accelBiasMsgPtr_->vector.z = accelBias(2);
442 pubAccelBias_.publish(accelBiasMsgPtr_);
443 }
444 // Publish gyro bias
445 if (pubGyroBias_.getNumSubscribers() > 0) {
446 Eigen::Vector3d gyroBias = optimizedStateWithCovarianceAndBiasPtr->getGyroscopeBias();
447 gyroBiasMsgPtr_->header.stamp = ros::Time(optimizedStateWithCovarianceAndBiasPtr->getTimeK());
448 gyroBiasMsgPtr_->header.frame_id = staticTransformsPtr_->getImuFrame();
449 gyroBiasMsgPtr_->vector.x = gyroBias(0);
450 gyroBiasMsgPtr_->vector.y = gyroBias(1);
451 gyroBiasMsgPtr_->vector.z = gyroBias(2);
452 pubGyroBias_.publish(gyroBiasMsgPtr_);
453 }
454
455 // Reference Frame TFs in Optimized State
456 for (const auto& framePairTransformMapIterator :
457 optimizedStateWithCovarianceAndBiasPtr->getReferenceFrameTransforms().getTransformsMap()) {
458 // Case 1: If includes world frame --> everything child of world --------------------------------
459 if (framePairTransformMapIterator.first.first == staticTransformsPtr_->getWorldFrame() ||
460 framePairTransformMapIterator.first.second == staticTransformsPtr_->getWorldFrame()) {
461 // A. Get transform
462 Eigen::Isometry3d T_W_M;
463 std::string mapFrameName;
464 const std::string& worldFrameName = staticTransformsPtr_->getWorldFrame();
465
466 // If world is second, then map is second
467 if (framePairTransformMapIterator.first.second == worldFrameName) {
468 T_W_M = framePairTransformMapIterator.second.inverse();
469 mapFrameName = framePairTransformMapIterator.first.first;
470 }
471 // If world is first, then map is second, this is the case for holistic alignment
472 else {
473 T_W_M = framePairTransformMapIterator.second;
474 mapFrameName = framePairTransformMapIterator.first.second;
475
476 // B. Publish TransformStamped for Aligned Frames (dynamically create publisher) --> only for reference frames
477 std::string transformTopic = "/graph_msf/transform_" + worldFrameName + "_to_" + mapFrameName + referenceFrameAlignedNameId;
478 geometry_msgs::PoseWithCovarianceStampedPtr poseWithCovarianceStampedMsgPtr =
479 boost::make_shared<geometry_msgs::PoseWithCovarianceStamped>();
480 addToPoseWithCovarianceStampedMsg(
481 poseWithCovarianceStampedMsgPtr, staticTransformsPtr_->getWorldFrame(),
482 ros::Time(optimizedStateWithCovarianceAndBiasPtr->getTimeK()), T_W_M,
483 optimizedStateWithCovarianceAndBiasPtr->getReferenceFrameTransformsCovariance().rv_T_frame1_frame2(
484 framePairTransformMapIterator.first.first, framePairTransformMapIterator.first.second));
485 // Check whether publisher already exists
486 if (pubPoseStampedByTopicMap_.find(transformTopic) == pubPoseStampedByTopicMap_.end()) {
487 pubPoseStampedByTopicMap_[transformTopic] = privateNode.advertise<geometry_msgs::PoseWithCovarianceStamped>(transformTopic, 1);
488 if (graphConfigPtr_->verboseLevel_ >= VerbosityLevels::kFunctioningAndStateMachine1) {
489 REGULAR_COUT << GREEN_START << " Initialized publisher for " << transformTopic << COLOR_END << std::endl;
490 }
491 }
492 if (pubPoseStampedByTopicMap_[transformTopic].getNumSubscribers() > 0) {
493 pubPoseStampedByTopicMap_[transformTopic].publish(poseWithCovarianceStampedMsgPtr);
494 }
495 }
496
497 // Print for debug
498 if (graphConfigPtr_->verboseLevel_ >= VerbosityLevels::kOperationInformation3) {
499 std::cout << "Transformation from " << mapFrameName << " to " << worldFrameName << std::endl;
500 std::cout << "Uncertainty: " << std::endl
501 << optimizedStateWithCovarianceAndBiasPtr->getReferenceFrameTransformsCovariance().rv_T_frame1_frame2(
502 framePairTransformMapIterator.first.first, framePairTransformMapIterator.first.second)
503 << std::endl;
504 }
505
506 // C. Publish TF Tree --> everything children of world
507 publishTfTreeTransform(worldFrameName, mapFrameName + referenceFrameAlignedNameId,
508 optimizedStateWithCovarianceAndBiasPtr->getTimeK(), T_W_M);
509 }
510 // Case 2: Other transformation (e.g. calibration) --> does usually not include world frame ----------------
511 else {
512 const std::string& sensorFrameName = framePairTransformMapIterator.first.first;
513 const std::string& sensorFrameNameCorrected = framePairTransformMapIterator.first.second;
514 const Eigen::Isometry3d& T_sensor_sensorCorrected = framePairTransformMapIterator.second;
515
516 // A. Publish TransformStamped for Corrected Frames
517 std::string transformTopic = "/graph_msf/transform_" + sensorFrameName + "_to_" + sensorFrameNameCorrected;
518 geometry_msgs::PoseWithCovarianceStampedPtr poseWithCovarianceStampedMsgPtr =
519 boost::make_shared<geometry_msgs::PoseWithCovarianceStamped>();
520 addToPoseWithCovarianceStampedMsg(
521 poseWithCovarianceStampedMsgPtr, sensorFrameName, ros::Time(optimizedStateWithCovarianceAndBiasPtr->getTimeK()),
522 T_sensor_sensorCorrected,
523 optimizedStateWithCovarianceAndBiasPtr->getReferenceFrameTransformsCovariance().rv_T_frame1_frame2(sensorFrameName,
524 sensorFrameNameCorrected));
525 // Check whether publisher already exists
526 if (pubPoseStampedByTopicMap_.find(transformTopic) == pubPoseStampedByTopicMap_.end()) {
527 pubPoseStampedByTopicMap_[transformTopic] = privateNode.advertise<geometry_msgs::PoseWithCovarianceStamped>(transformTopic, 1);
528 REGULAR_COUT << GREEN_START << " Initialized publisher for " << transformTopic << COLOR_END << std::endl;
529 }
530 if (pubPoseStampedByTopicMap_[transformTopic].getNumSubscribers() > 0) {
531 pubPoseStampedByTopicMap_[transformTopic].publish(poseWithCovarianceStampedMsgPtr);
532 }
533
534 // B. Publish TF Tree --> as children of sensor frame
535 publishTfTreeTransform(sensorFrameName, sensorFrameNameCorrected, optimizedStateWithCovarianceAndBiasPtr->getTimeK(),
536 T_sensor_sensorCorrected);
537 }
538 }
539
540 // Landmarks
541 for (const auto& landmarkTransformMapIterator : optimizedStateWithCovarianceAndBiasPtr->getLandmarkTransforms().getTransformsMap()) {
542 // A. Get transform
543 Eigen::Isometry3d T_W_L = landmarkTransformMapIterator.second;
544 std::string worldFrameName = landmarkTransformMapIterator.first.first;
545 std::string landmarkFrameName = landmarkTransformMapIterator.first.second;
546
547 // B. Publish TransformStamped for Landmarks
548 std::string landmarkName = "/graph_msf/landmark_" + staticTransformsPtr_->getWorldFrame() + "_to_" + landmarkFrameName;
549 addToLandmarkMarkerArrayMsg(landmarkMarkersMsgPtr_, staticTransformsPtr_->getWorldFrame(),
550 ros::Time(optimizedStateWithCovarianceAndBiasPtr->getTimeK()), T_W_L, landmarkName);
551 // Publish if there is a subscriber
552 if (pubLandmarks_.getNumSubscribers() > 0) {
553 pubLandmarks_.publish(landmarkMarkersMsgPtr_);
554 }
555 }
556 }
