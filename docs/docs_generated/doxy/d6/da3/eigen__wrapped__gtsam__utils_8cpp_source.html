20// Transformation Functions ---------------------------------------------------
21// Remove roll and pitch
22void inPlaceRemoveRollPitch(Eigen::Isometry3d& T) {
23 // Extract the rotation matrix
24 gtsam::Rot3 R(T.rotation());
25 // Remove the roll and pitch
26 R = gtsam::Rot3::Ypr(R.yaw(), 0.0, 0.0);
27 // Set the rotation matrix
28 T.linear() = R.matrix();
29}
30// Calculate yaw from Isometry
31double getYawFromIsometry(const Eigen::Isometry3d& T) {
32 gtsam::Rot3 R(T.rotation());
33 return R.yaw();
34}
