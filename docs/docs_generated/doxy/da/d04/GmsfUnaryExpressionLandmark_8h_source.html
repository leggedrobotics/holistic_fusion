53 // ii.B) Adding Landmark State in Dynamic Memory
54 void transformLandmarkInWorldToImuFrame(DynamicDictionaryContainer& gtsamDynamicExpressionKeys,
55 const gtsam::NavState& W_currentPropagatedState) final {
56 // Mutex because we are changing the dynamically allocated graphKeys
57 std::lock_guard<std::mutex> modifyGraphKeysLock(gtsamDynamicExpressionKeys.get<gtsam::Pose3>().mutex());
58
59 // Get initial guess (computed geometrically)
60 gtsam::Point3 W_t_W_L_initial = computeW_t_W_L_initial(W_currentPropagatedState);
61
62 // Landmark names
63 const std::string newLandmarkName = landmarkName_ + "_" + std::to_string(landmarkCreationCounter_);
64 const std::string previousLandmarkName = landmarkName_ + "_" + std::to_string(landmarkCreationCounter_ - 1);
65
66 // Create new graph key for landmark dynamically
67 bool newGraphKeyAddedFlag = false;
68 DynamicVariableType variableType = DynamicVariableType::Landmark(this->gmsfBaseUnaryMeasurementPtr_->timeK());
69 const DynamicFactorGraphStateKey newGraphKey = gtsamDynamicExpressionKeys.get<gtsam::Pose3>().getTransformationKey<'l'>(
70 newGraphKeyAddedFlag, gmsfUnaryLandmarkMeasurementPtr_->worldFrameName(), newLandmarkName,
71 this->gmsfBaseUnaryMeasurementPtr_->timeK(), gtsam::Pose3(gtsam::Rot3::Identity(), W_t_W_L_initial), variableType);
72 // Make sure that the variable at the key is active (landmarks always have to be active or removed)
73 assert(newGraphKey.isVariableActive());
74
75 // Remove previous landmark with same landmark name;
76 if (newGraphKeyAddedFlag) {
77 std::ignore = gtsamDynamicExpressionKeys.get<gtsam::Pose3>().removeOrDeactivateTransform(
78 gmsfUnaryLandmarkMeasurementPtr_->worldFrameName(), previousLandmarkName);
79 }
80
81 // Create expression for landmark
82 const gtsam::Point3_ exp_W_t_W_L = gtsam::Point3_(newGraphKey.key());
83
84 // Convert to Imu frame
85 exp_sensorFrame_t_sensorFrame_landmark_ = gtsam::transformTo(exp_T_W_I_, exp_W_t_W_L); // I_t_I_L at this point
86
87 // Add values, if a new state was created
88 if (newGraphKeyAddedFlag) {
89 // Add initial guess
90 this->newOnlineStateValues_.insert(newGraphKey.key(), W_t_W_L_initial);
91 this->newOfflineStateValues_.insert(newGraphKey.key(), W_t_W_L_initial);
92 }
93 }
