58template <typename PointT>
59inline float calcSquaredDiff(const PointT& a, const PointT& b) {
60 float diffX = a.x - b.x;
61 float diffY = a.y - b.y;
62 float diffZ = a.z - b.z;

172inline void invertHomogenousMatrix(const Eigen::Matrix4d& m_in, Eigen::Matrix4d& m_out) {
173 m_out = Eigen::Matrix4d::Identity();
174 Eigen::Matrix3d R = m_in.block<3, 3>(0, 0);
175 Eigen::Vector3d t = m_in.block<3, 1>(0, 3);
176 m_out.block<3, 3>(0, 0) = R.transpose();
177 m_out.block<3, 1>(0, 3) = -R.transpose() * t;
178}
