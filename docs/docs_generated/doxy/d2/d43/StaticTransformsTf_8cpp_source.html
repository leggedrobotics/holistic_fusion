20bool StaticTransformsTf::findTransformations() {
21 // Print to console --------------------------
22 REGULAR_COUT << " Looking up transforms in TF-tree." << std::endl;
23 REGULAR_COUT << " Transforms between the following frames are required: " << imuFrame_ << ", " << baseLinkFrame_ << std::endl;
24 REGULAR_COUT << " Waiting for up to 100 seconds until they arrive..." << std::endl;
25
26 // Temporary variable
27 static tf::StampedTransform transform;
28
29 // Look up transforms ----------------------------
30 // Sleep before subscribing, otherwise sometimes dying in the beginning of rosbag
31 ros::Rate rosRate(10);
32 rosRate.sleep();
33
34 // Imu to Base ---
35 REGULAR_COUT << " Looking up transform from " << imuFrame_ << " to " << baseLinkFrame_ << std::endl;
36 listener_.waitForTransform(imuFrame_, baseLinkFrame_, ros::Time(0), ros::Duration(100.0));
37 listener_.lookupTransform(imuFrame_, baseLinkFrame_, ros::Time(0), transform);
38 // I_Base
39 graph_msf::tfToIsometry3(tf::Transform(transform), lv_T_frame1_frame2(imuFrame_, baseLinkFrame_));
40 REGULAR_COUT << " Translation I_Base: " << rv_T_frame1_frame2(imuFrame_, baseLinkFrame_).translation() << std::endl;
41 // Base_I
42 lv_T_frame1_frame2(baseLinkFrame_, imuFrame_) = rv_T_frame1_frame2(imuFrame_, baseLinkFrame_).inverse();
43
44 // Wrapping up --------------------------
45 REGULAR_COUT << " Transforms looked up successfully." << std::endl;
46 return true;
47}
