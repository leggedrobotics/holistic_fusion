28bool AnymalStaticTransforms::findTransformations() {
29 // Print to console --------------------------
30 REGULAR_COUT << " Looking up transforms in TF-tree." << std::endl;
31 REGULAR_COUT << " Transforms between the following frames are required: " << lioOdometryFrame_ << ", " << gnssFrame_ << ", " << imuFrame_
32 << ", " << baseLinkFrame_ << std::endl;
33 REGULAR_COUT << " Waiting for up to " << TF_WAIT_TIME << " seconds until they arrive..." << std::endl;
34
35 // Temporary variable
36 static tf::StampedTransform transform;
37
38 // Look up transforms ----------------------------
39 // Sleep before subscribing, otherwise sometimes dying in the beginning of rosbag
40 ros::Rate rosRate(10);
41 rosRate.sleep();
42
43 // Imu to LiDAR Link ---
44 REGULAR_COUT << " Looking up transform from " << imuFrame_ << " to " << lioOdometryFrame_ << std::endl;
45 listener_.waitForTransform(imuFrame_, lioOdometryFrame_, ros::Time(0), ros::Duration(TF_WAIT_TIME));
46 listener_.lookupTransform(imuFrame_, lioOdometryFrame_, ros::Time(0), transform);
47 // I_Lidar
48 graph_msf::tfToIsometry3(tf::Transform(transform), lv_T_frame1_frame2(imuFrame_, lioOdometryFrame_));
49 REGULAR_COUT << " Translation I_Lidar: " << rv_T_frame1_frame2(imuFrame_, lioOdometryFrame_).translation() << std::endl;
50 // Lidar_I
51 lv_T_frame1_frame2(lioOdometryFrame_, imuFrame_) = rv_T_frame1_frame2(imuFrame_, lioOdometryFrame_).inverse();
52
53 // Imu to GNSS Link ---
54 try {
55 REGULAR_COUT << " Looking up transform from " << imuFrame_ << " to " << gnssFrame_ << std::endl;
56 listener_.waitForTransform(imuFrame_, gnssFrame_, ros::Time(0), ros::Duration(TF_WAIT_TIME));
57 listener_.lookupTransform(imuFrame_, gnssFrame_, ros::Time(0), transform);
58 // I_Gnss
59 graph_msf::tfToIsometry3(tf::Transform(transform), lv_T_frame1_frame2(imuFrame_, gnssFrame_));
60 REGULAR_COUT << " Translation I_GnssL: " << rv_T_frame1_frame2(imuFrame_, gnssFrame_).translation() << std::endl;
61 // GnssL_I
62 lv_T_frame1_frame2(gnssFrame_, imuFrame_) = rv_T_frame1_frame2(imuFrame_, gnssFrame_).inverse();
63 } catch (const tf2::LookupException& e) {
64 REGULAR_COUT << RED_START << " Could not find transformations for GNSS in TF tree." << COLOR_END << std::endl;
65 }
66
67 REGULAR_COUT << GREEN_START << " Transforms looked up successfully." << COLOR_END << std::endl;
68
69 // Call parent class
70 graph_msf::StaticTransformsTf::findTransformations();
71 return true;
72}
